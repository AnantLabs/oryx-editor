 24 if (!ORYX.Plugins) 
 25     ORYX.Plugins = new Object();
 26 
 27 ORYX.Plugins.TreeGraphSupport = Clazz.extend({
 28 
 29     facade: undefined,
 30     
 31 	/**
 32 	 * 
 33 	 * @param {Object} facade
 34 	 */
 35     construct: function(facade){
 36         
 37 		// Save the facade
 38 		this.facade = facade;
 39         
 40 		// Offer new functionality
 41         this.facade.offer({
 42             'name'				: ORYX.I18N.TreeGraphSupport.syntaxCheckName,
 43             'functionality'		: this.syntaxCheck.bind(this),
 44             'group'				: ORYX.I18N.TreeGraphSupport.group,
 45             'icon'				: ORYX.PATH + "images/checker_syntax.png",
 46             'description'		: ORYX.I18N.TreeGraphSupport.syntaxCheckDesc,
 47             'index'				: 1,
 48             'minShape'			: 0,
 49             'maxShape'			: 0
 50         });
 51         
 52     },
 53     
 54 	/**
 55 	 * 
 56 	 */
 57     syntaxCheck: function() {
 58 		this.facade.raiseEvent({
 59 			type: 			ORYX.CONFIG.EVENT_OVERLAY_HIDE,
 60 			id: 			"treegraph",
 61 		});
 62 		
 63 		
 64          // Send the request to the server.
 65         new Ajax.Request(ORYX.CONFIG.TREEGRAPH_SUPPORT, {
 66             method: 'POST',
 67             asynchronous: false,
 68             parameters: {
 69                 data: this.facade.getERDF()
 70             },
 71             onSuccess: function(request){
 72             	var resp = request.responseText.evalJSON();
 73 
 74 				if (resp instanceof Array ) {
 75 					if (resp.length > 0) {
 76 					
 77 						// Get all Valid ResourceIDs and collect all shapes
 78 						resp.each(function( value ){ 
 79 						
 80 							var sh = this.facade.getCanvas().getChildShapeByResourceId( value );
 81 
 82 							if( sh ){
 83 								
 84 								this.highlightShape(sh);
 85 								
 86 							}
 87 						}.bind(this));
 88 					}
 89 				}
 90 	            Ext.Msg.show({
 91 	            	title	: ORYX.I18N.Oryx.title,
 92 	            	msg		: request.responseText,
 93 	            	icon		: Ext.MessageBox.INFO
 94 	            });
 95             }.bind(this),
 96             	
 97 
 98                         
 99 			onFailure: function(request){
100             	Ext.Msg.show({
101 				   title	: ORYX.I18N.Oryx.title,
102 				   msg		: 'An error occurs while sending request!',
103 				   icon		: Ext.MessageBox.WARNING
104 				});
105             }
106         });
107 
108     },
109     
110 	highlightShape: function(shape){
111 		// Creates overlay for an enabled shape
112 		// display is beeing ignored
113 		if(!(shape instanceof ORYX.Core.Shape)) return;
114 		
115 		var attr;
116 		if(shape instanceof ORYX.Core.Edge) {
117 			attr = {stroke: "red"};
118 		}
119 		else {
120 			attr = {fill: "red", stroke:"black", "stroke-width": 2};
121 		}
122 		
123 		
124 
125 											
126 		this.facade.raiseEvent({
127 				type: 			ORYX.CONFIG.EVENT_OVERLAY_SHOW,
128 				id: 			"treegraph",
129 				shapes: 		[shape],
130 				attributes: 	attr,
131 			});
132 	}
133 });
