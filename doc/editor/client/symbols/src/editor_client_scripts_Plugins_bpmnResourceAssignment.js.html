 25 ORYX.Plugins.ResourceAssignment = Clazz.extend({
 26 
 27     facade: undefined,
 28     
 29     construct: function(facade){
 30 		
 31         this.facade = facade;
 32         
 33 		this.active 		= false;
 34 		this.raisedEventIds = [];
 35 		
 36         this.facade.offer({
 37             'name': ORYX.I18N.ResourceAssignment.name,
 38             'functionality': this.assignResource.bind(this),
 39             'group': ORYX.I18N.ResourceAssignment.group,
 40            // 'dropDownGroupIcon': ORYX.PATH + "images/hr.png",
 41             'icon': ORYX.PATH + "images/hr.png",
 42             'description': ORYX.I18N.ResourceAssignment.desc,
 43             'index': 0,
 44             'toggle': false,
 45             'minShape': 1,
 46             'maxShape': 0
 47         });
 48 		
 49     },
 50     
 51 	assignResource: function(){
 52     	//variable declaration
 53     	var selectedElements = this.facade.getSelection(); //all selected elements
 54     	var taskElements = []; //all selected tasks
 55     	var checkedValues = []; //values checked in resource selection dialog
 56     	var i = 0; //counter
 57     	var len = selectedElements.length; //number of selected elements
 58     	
 59     	//get only selected tasks
 60     	for (var index = 0; index < len; index++) {
 61     		var item = selectedElements[index];
 62     		if(item.properties["oryx-activitytype"] == "Task") {
 63     			taskElements[i] = item;
 64 				if(taskElements[i].properties["oryx-id"] == "") {
 65 					taskElements[i].setProperty("oryx-id", taskElements[i].id);
 66 				}
 67     			i++;
 68     		}
 69     	}
 70 
 71     	//get allocation type and resources to be possibly assigned to the current task
 72 		var allocationTypeData = this.handleAllocationTypeData(taskElements[0].properties["oryx-id"]);
 73 		if(allocationTypeData[0] != null) { //skip following if no resource/ role is selected
 74 			var resourceData = allocationTypeData[0];
 75 			var allocationType = allocationTypeData[1];
 76 			
 77 			if(resourceData != "--automatic execution--") { //skip following if automatic execution shall take place
 78 				//Dialog to select resources respectively roles
 79 				var resourceSelection = new Ext.Window({
 80 					frame: true,
 81 					title: 'Please choose resources for assignment',
 82 					width: 600,
 83 					modal: true,
 84 					closable: false,
 85 					plain: true,
 86 					items: {
 87 						xtype: 'checkbox',
 88 						boxLabel: resourceData.resource[0].resource,
 89 						name: 'checkbox0'
 90 					},
 91 					buttons: [{
 92 						text: 'Select',
 93 						handler: function() {
 94 							resourceSelection.close();
 95 							checkedValues = this.getCheckedValues(resourceSelection);
 96 							this.writeAssignments(checkedValues, taskElements, allocationType);
 97 						}.bind(this)
 98 					},{
 99 						text: 'Close',
100 						handler: function() {
101 							resourceSelection.close();
102 						}.bind(this)
103 					}]
104 				});
105 				var checkbox;
106 				var checkboxName;
107 				//add one checkbox for each entry (resource, role) to dialog window
108 				for (var index = 1; index < resourceData.resource.length; index++) {
109 					checkboxName = "checkbox" + index;
110 					checkbox = new Ext.form.Checkbox ({
111 					boxLabel: resourceData.resource[index].resource,
112 					name: checkboxName
113 					});
114 					resourceSelection.add(checkbox);
115 				}
116 				resourceSelection.show();
117 			} else { //automatic execution
118 				checkedValues[0] = resourceData;
119 				this.writeAssignments(checkedValues, taskElements, allocationType);
120 			}
121 		}
122     },
123 	
124 	getResourceData: function(allocationType) { //Ajax request to get resource/ role information from servlet
125 		var resp;
126 		//get Resource List
127 		new Ajax.Request(ORYX.CONFIG.ROOT_PATH + 'resourceList', {
128 			method: 'POST',
129 			asynchronous: false,
130 			parameters: {
131 				allocation: allocationType
132 			},
133 			onSuccess: function(request){
134 				resp = request.responseText.evalJSON();
135 				returnValue = resp;
136 			}.bind(this)
137     	});
138 		return resp;
139 	},
140 	
141 	getCheckedValues: function(resourceSelection) { //identify check elements and return the value
142 		var selectedValues = [];
143 		var j = 0;
144 		for (var index = 0; index < resourceSelection.items.items.length; index++) {
145 			if(resourceSelection.items.items[index].checked == true) {
146 				selectedValues[j] = resourceSelection.items.items[index].boxLabel;
147 				j++;
148 			}
149 		}
150 		return selectedValues;
151 	},
152 	
153 	writeAssignments: function(newResources, taskElements, allocationType) { //create and write JSON object to task attributes
154 		for (var taskCounter = 0; taskCounter < taskElements.length; taskCounter++) {
155 			var jsonObjectString;
156 			var currentTask = taskElements[taskCounter];
157 			if (currentTask.properties["oryx-resourceassignments"] != "") { //assignments existing?
158 				var jsonObject = currentTask.properties["oryx-resourceassignments"].evalJSON();
159 				var totalCount = parseInt(jsonObject.totalCount) + newResources.length;
160 				var items = jsonObject.items.toArray();
161 				//eliminate duplicates
162 				for (var m = 0; m < newResources.length; m++) {
163 					for (var n = 0; n < items.length; n++) {
164 						if (items[n].assignmentName == newResources[m]) {
165 							if (items[n].assignmentType == allocationType) { //same allocation type, otherwise resources are not equal
166 								//remove duplicate
167 								newResources.splice(m, 1);
168 								totalCount--;
169 								m--;
170 								break;
171 							}
172 						}
173 					}
174 				}
175 				
176 				//write existing assignments
177 				jsonObjectString = "{'totalCount':" + totalCount + ", 'items':[{assignmentType:\"" + items[0].assignmentType + "\", assignmentName:\"" + items[0].assignmentName + "\"}";
178 				for (var c = 1; c < items.length; c++) {
179 					jsonObjectString = jsonObjectString + ", {assignmentType:\"" + items[c].assignmentType + "\", assignmentName:\"" + items[c].assignmentName + "\"}";
180 				}
181 				//write new assignments
182 				for (var c = 0; c < newResources.length; c++) {
183 					jsonObjectString = jsonObjectString + ", {assignmentType:\"" + allocationType + "\", assignmentName:\"" + newResources[c] + "\"}";
184 				}
185 				jsonObjectString = jsonObjectString + "]}";
186 			}
187 			else { //no existing assignments
188 				//write new assignments
189 				jsonObjectString = "{'totalCount':" + newResources.length + ", 'items':[{assignmentType:\"" + allocationType + "\", assignmentName:\"" + newResources[0] + "\"}";
190 				for (var c = 1; c < newResources.length; c++) {
191 					jsonObjectString = jsonObjectString + ", {assignmentType:\"" + allocationType + "\", assignmentName:\"" + newResources[c] + "\"}";
192 				}
193 				jsonObjectString = jsonObjectString + "]}";
194 			}
195 			//complete allocation by writing final entry
196 			currentTask.setProperty("oryx-resourceassignments", jsonObjectString);
197 			
198 		}
199 	},
200 	
201 	handleAllocationTypeData: function(taskId) { //ask for type of allocation via a prompt and collect resource information to return it
202 		var returnValue = [];
203 		var resourceData;
204 		//get possible new assignments for current task
205 		var promptResult = prompt("for task " + taskId + "\n1: direct, \n2: role, \n3: org, \n4: auto");
206 		if(promptResult != "1" && promptResult != "2" && promptResult != "3" && promptResult != "4" && promptResult != null) {
207 			alert("wrong entry, please try again");
208 			returnValue = this.handleAllocationTypeData(taskId);
209 		} else {
210 			if(promptResult == "1") {
211 				resourceData = this.getResourceData("direct");
212 				promptResult = "Single User";
213 				//promptResult = "Direct Allocation";
214 			} else if(promptResult == "2") {
215 				resourceData = this.getResourceData("functional");
216 				promptResult = "Functional Role";
217 				//promptResult = "Functional-Role-based Allocation";
218 			} else if(promptResult == "3") {
219 				resourceData = this.getResourceData("organisational");
220 				promptResult = "Organisational Role";
221 				//promptResult = "Organisational-Role-based Allocation";
222 			} else if(promptResult == "4") {
223 				resourceData = "--automatic execution--";
224 				promptResult = "Automatic Execution";
225 				//promptResult = "Automatic Execution";
226 			} else if(promptResult == null) {
227 				resourceData = null;
228 			}
229 			returnValue[0] = resourceData;
230 			returnValue[1] = promptResult;
231 		}
232 		return returnValue;
233 	},
234     
235     getSavedAssignmentsData: function(taskElement) {
236     	var returnValue = [];
237     	var elementIds = []; //collection of used ids for saved assignments
238 		var totalCount = 0; //number of defined separations for a specific task
239     	var highestId = 1; //next element id for separations of a specific task
240     	var j = 0; //counter
241     	var m = 0; //counter
242 		var assignments = taskElement.properties["oryx-resourceassignments"]; //get saved assignments
243 		assignments = assignments.toString();
244     	//store saved assignments including task ids
245     	var savedAssignmentsString = assignments.substring(0,assignments.indexOf(']'));
246     	//extract number of entries and element ids of these entries
247     	while(assignments.indexOf(',') > -1) {
248     		if(j%2 == 1) { //get element ids as integer
249     			elementIds[m] = parseInt(assignments.substring((assignments.indexOf('"')) + 1));
250     			m++;
251     			assignments = assignments.substring((assignments.indexOf(',')) + 1);
252     		} else if(j == 0) { //get total count as integer
253     			totalCount = parseInt(assignments.substring((assignments.indexOf(':')) + 1));
254     			assignments = assignments.substring((assignments.indexOf(',')) + 1);
255     		} else { //task ids are already saved in "savedSeparations", so that these entries can be discarded
256     			assignments = assignments.substring((assignments.indexOf(',')) + 1);
257     		}
258 	        j++;
259     	}
260     	
261     	//get highest element id
262     	for(var index = 0; index < elementIds.length; index++) {
263     		if(elementIds[index] > highestId) {
264     			highestId = elementIds[index];
265     		}
266     	}
267     	returnValue[0] = savedAssignmentsString;
268     	returnValue[1] = totalCount;
269     	returnValue[2] = highestId;
270     	return returnValue;
271     }
