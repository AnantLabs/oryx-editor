100 /**
101  * Mouse Down Handler
102  */
103 handleMouseDown: function(event, uiObj) {	
104 	
105 	if(this.myLabel && this.myLabel._text != ""){	
106 		//save MousePosition
107 		var MouseX = this.facade.eventCoordinates(event).x;
108 		var MouseY = this.facade.eventCoordinates(event).y;
109 		
110 		//refresh Coordinates
111 		this.calculateLabelCoordinates();
112 		this.calculateRotationPointCoordinates();
113 	
114 		//Set LabelPosition to MousePosition	
115 		if(this.labelSelected == true){				
116 			if(this.myLabel._rotate == 270 || this.myLabel._rotate == 315 || this.myLabel._rotate == 360 || this.myLabel._rotate == 45) {
117 				this.myLabel.x = this.facade.eventCoordinates(event).x+5;
118 				this.myLabel.y = this.facade.eventCoordinates(event).y-5;
119 			}else if (this.myLabel._rotate == 90 || this.myLabel._rotate == 135 || this.myLabel._rotate == 180 || this.myLabel._rotate == 225){
120 				this.myLabel.x = this.facade.eventCoordinates(event).x-5;
121 				this.myLabel.y = this.facade.eventCoordinates(event).y+5;
122 			}
123 			else {
124 				this.myLabel.x = this.facade.eventCoordinates(event).x+10;
125 				this.myLabel.y = this.facade.eventCoordinates(event).y-5;
126 			}
127 			
128 			//refresh real Rotationpoint
129 			if(this.myLabel._rotationPoint){
130 				this.myLabel._rotationPoint.x = MouseX;
131 				this.myLabel._rotationPoint.y = MouseY;
132 			}
133 			
134 			// save the current position of the label in edge.js, mark the label as free moved
135 			this.myLabel.edgePosition = "freeMoved";
136 			this.myLabel.update();			
137 		
138 			//Refresh Coordinates
139 			this.calculateLabelCoordinates();
140 			this.calculateRotationPointCoordinates();				
141 			this.showOverlay("RotationPoint", this.myEdge, this.rotationPointCoordinates, this.rotationPoint);
142 			this.hideOverlay("SettingArrows");
143 			
144 			//Show Moving Arrows		
145 			this.mouseCoordinates = {x:this.labelCoordinates.x, y:this.labelCoordinates.y};			
146 			this.showOverlay("MovingArrows", this.myEdge, this.mouseCoordinates, this.startMovingCross);			
147 			
148 			//show the Association line
149 			this.showLine();
150 		}
151 		else {		
152 			this.hideLine();
153 			this.hideOverlay("RotationPoint");
154 			this.hideOverlay("MovingArrows");
155 			this.hideOverlay("SettingArrows");
156 		}
157 	
158 		//Check if Mouse is in the ClickArea of the Label 
159 		if(	this.labelSelected == false && 
160 				MouseX >= this.labelCoordinates.x-20 && 
161 				MouseX <= this.labelCoordinates.x+20 && 
162 				MouseY <= this.labelCoordinates.y+20 && 
163 				MouseY >= this.labelCoordinates.y-20){		
164 		
165 			//Set Label as Selected
166 			this.labelSelected=true;
167 						
168 			//refresh and show RotationPoint
169 			this.calculateRotationPointCoordinates();
170 			this.showOverlay("RotationPoint", this.myEdge, this.rotationPointCoordinates, this.rotationPoint);
171 		}
172 		else {
173 			//Set Label as not selected
174 			this.labelSelected = false;	
175 		}
176 	
177 		//Check if MouseClick is on RotationPoint
178 		if(	this.rotate == false && 
179 			MouseX >= this.rotationPointCoordinates.x-20 && 
180 			MouseX <= this.rotationPointCoordinates.x+20 && 
181 			MouseY >= this.rotationPointCoordinates.y-20 && 
182 			MouseY <= this.rotationPointCoordinates.y+20){
183 		
184 			//acitvate Rotation
185 			this.rotate = true;
186 			
187 			//Set current RotationState
188 			this.State = 0;
189 			
190 			//show RotationArrows to show that rotation is active
191 			this.showOverlay("RotationPointActive", this.myEdge, this.rotationPointCoordinates, this.moveLeftRight);
192 		}
193 		else{
194 			//deactivate Rotation
195 			this.rotate = false;
196 			
197 			//Hide yellow RotationArrows
198 			this.hideOverlay("RotationPointActive");
199 		}	
200 	}
201 	
202 	//clicking on an Edge saves the label and show the line
203 	if( uiObj instanceof ORYX.Core.Edge){  		
204 		//Identify and set the label of the current Edge	
205 		if(uiObj._labels[uiObj.id+"condition"]){
206 			this.myLabel = uiObj._labels[uiObj.id+"condition"]; //for BPMN
207 		}else if (uiObj._labels[uiObj.id+"name"]){
208 			this.myLabel = uiObj._labels[uiObj.id+"name"]; //for UML
209 		}else {
210 			this.myLabel = uiObj._labels[uiObj.id+"text"];
211 		}
212 	 
213 		//save the edge for adding rotationpoint	
214 		this.myEdge = uiObj;
215 
216 		//Show the RotationPoint and line of the label of the current Edge
217 		if(this.myLabel && this.myLabel._text != ""){
218 			this.calculateLabelCoordinates();
219 			this.calculateRotationPointCoordinates();
220 			this.showLine();			
221 			this.mouseCoordinates = {x:this.labelCoordinates.x, y:this.labelCoordinates.y};			
222 			this.showOverlay("MovingArrows", this.myEdge, this.mouseCoordinates, this.startMovingCross);
223 			this.showOverlay("RotationPoint", this.myEdge, this.rotationPointCoordinates, this.rotationPoint);
224 		}
225 	}

228 /**
229  * Mouse Move Handler
230  */
231 handleMouseMove: function(event, uiObj) {
232 
233 	//if label is selected Posision is set to MousePosition
234 	if(this.labelSelected == true){
235 		if(this.myLabel){			
236 			//Set Label positon for different Rotations
237 			if(this.myLabel._rotate == 270 || this.myLabel._rotate == 315 || this.myLabel._rotate == 360 || this.myLabel._rotate == 45) {
238 				this.myLabel.x = this.facade.eventCoordinates(event).x+5;
239 				this.myLabel.y = this.facade.eventCoordinates(event).y-5;
240 				this.myLabel._rotationPoint.x = this.facade.eventCoordinates(event).x+10;
241 				this.myLabel._rotationPoint.y = this.facade.eventCoordinates(event).y-10;
242 			}else if(this.myLabel._rotate == 90 || this.myLabel._rotate == 135 || this.myLabel._rotate == 180 || this.myLabel._rotate == 225){
243 				this.myLabel.x = this.facade.eventCoordinates(event).x-5;
244 				this.myLabel.y = this.facade.eventCoordinates(event).y+5;
245 				this.myLabel._rotationPoint.x = this.facade.eventCoordinates(event).x-10;
246 				this.myLabel._rotationPoint.y = this.facade.eventCoordinates(event).y+10;
247 			}else {
248 				this.myLabel.x = this.facade.eventCoordinates(event).x+10;
249 				this.myLabel.y = this.facade.eventCoordinates(event).y-5;
250 				this.myLabel._rotationPoint.x = this.facade.eventCoordinates(event).x+10;
251 				this.myLabel._rotationPoint.y = this.facade.eventCoordinates(event).y-10;
252 			}
253 				
254 			this.myLabel.update();
255 		
256 			//refresh Coordinates
257 			this.calculateLabelCoordinates();
258 			this.calculateRotationPointCoordinates();		
259 
260 			this.hideOverlay("RotationPoint");
261 			this.hideOverlay("MovingArrows");
262 			this.hideOverlay("SettingArrows");			
263 			
264 			//refresh Position of the Overlay arrows
265 			if(this.myLabel._rotate == 270 || this.myLabel._rotate == 315 || this.myLabel._rotate == 360 || this.myLabel._rotate == 45) {	
266 				this.mouseCoordinates = {x:this.facade.eventCoordinates(event).x, y:this.facade.eventCoordinates(event).y};	
267 				this.showOverlay("SettingArrows", this.myEdge, this.mouseCoordinates, this.endMovingCross);
268 			}else {
269 				this.mouseCoordinates = {x:this.facade.eventCoordinates(event).x, y:this.facade.eventCoordinates(event).y};	
270 				this.showOverlay("SettingArrows", this.myEdge, this.mouseCoordinates, this.endMovingCross);
271 			}			
272 			//Refresh the Line
273 			this.hideLine();
274 			this.showLine();
275 		}
276 	}
277 	
278 	//perform the Statevalidation for Rotation
279 	if(this.rotate == true) {
280 		
281 		//save MouseCoordinates
282 		var MouseX = this.facade.eventCoordinates(event).x;
283 		var MouseY = this.facade.eventCoordinates(event).y;
284 
285 		//refresh Coordinates
286 		this.calculateLabelCoordinates();
287 		this.calculateRotationPointCoordinates();
288 		
289 		//defines the States for Rotation(every 20px there is a new State)
290 		if(MouseX < this.rotationPointCoordinates.x-150 ){
291 			this.State = -8;
292 		}
293 		else if (MouseX < this.rotationPointCoordinates.x-130 && MouseX >= this.rotationPointCoordinates.x-150){
294 			this.State = -7;	
295 		}
296 		else if (MouseX < this.rotationPointCoordinates.x-110 && MouseX >= this.rotationPointCoordinates.x-130){
297 			this.State =- 6;	
298 		}
299 		else if (MouseX < this.rotationPointCoordinates.x-90 && MouseX >= this.rotationPointCoordinates.x-110){
300 			this.State = -5;	
301 		}
302 		else if (MouseX < this.rotationPointCoordinates.x-70 && MouseX >= this.rotationPointCoordinates.x-90){
303 			this.State = -4;	
304 		}
305 		else if (MouseX < this.rotationPointCoordinates.x-50 && MouseX >= this.rotationPointCoordinates.x-70){
306 			this.State = -3;	
307 		}
308 		else if (MouseX < this.rotationPointCoordinates.x-30 && MouseX >= this.rotationPointCoordinates.x-50){
309 			this.State = -2;	
310 		}
311 		else if (MouseX < this.rotationPointCoordinates.x-10 && MouseX >= this.rotationPointCoordinates.x-30){
312 			this.State = -1;
313 		}
314 		else if (MouseX < this.rotationPointCoordinates.x+10 && MouseX >= this.rotationPointCoordinates.x-10){
315 			this.State = 0;	
316 		}
317 		else if (MouseX < this.rotationPointCoordinates.x+30 && MouseX >= this.rotationPointCoordinates.x+10){
318 			this.State = 1;	
319 		}
320 		else if (MouseX < this.rotationPointCoordinates.x+50 && MouseX >= this.rotationPointCoordinates.x+30){
321 			this.State = 2;
322 		}
323 		else if (MouseX < this.rotationPointCoordinates.x+70 && MouseX >= this.rotationPointCoordinates.x+50){
324 			this.State = 3;
325 		}
326 		else if ( MouseX < this.rotationPointCoordinates.x+90 && MouseX >= this.rotationPointCoordinates.x+70){
327 			this.State = 4;	
328 		}
329 		else if ( MouseX < this.rotationPointCoordinates.x+110 && MouseX >= this.rotationPointCoordinates.x+90){
330 			this.State = 5;	
331 		}
332 		else if ( MouseX < this.rotationPointCoordinates.x+130 && MouseX >= this.rotationPointCoordinates.x+110){
333 			this.State = 6;	
334 		}
335 		else if ( MouseX < this.rotationPointCoordinates.x+150 && MouseX >= this.rotationPointCoordinates.x+130){
336 			this.State7;	
337 		}
338 		else if(MouseX >= this.rotationPointCoordinates.x+150){
339 			this.State = 8;
340 		}
341 		
342 		//checks the way of moving the Mouse through the states and rotate
343 		if(this.State > this.prevState){
344 			this.rotate_right();
345 			this.prevState = this.State;
346 		}
347 		else if(this.State < this.prevState){
348 			this.rotate_left();
349 			this.prevState = this.State;
350 		}
351 	}

355  * rotate the label to the right with 45Â° (clockwise)

359 
360 	var myRotPoint = {x:this.labelCoordinates.x, y:this.labelCoordinates.y};

367 		this.myLabel.rotate(90, myRotPoint);
368 	}
369 	else if(myRotation == 315 || myRotation > 315 && myRotation < 360){
370 		this.myLabel.rotate(360,myRotPoint);
371 	}
372 	else if(myRotation == 270 || myRotation > 270 && myRotation < 315){
373 		this.myLabel.rotate(315,myRotPoint);
374 	}
375 	else if(myRotation == 90 || myRotation < 135 && myRotation > 90){
376 		this.myLabel.rotate(135,myRotPoint);
377 	}
378 	else if(myRotation == 135 || myRotation < 180 && myRotation > 135){
379 		this.myLabel.rotate(180,myRotPoint);
380 	}
381 	else if(myRotation == 180 || myRotation < 225 && myRotation > 180){
382 		this.myLabel.rotate(225,myRotPoint);		
383 	}
384 	else if(myRotation == 225 || myRotation < 270 && myRotation > 225){
385 		this.myLabel.rotate(270,myRotPoint);

425 /**
426  * set the Coordinates of the RotationPoint for different degree values
427  * 
428  */
429 calculateRotationPointCoordinates: function(){	
430 	if(this.rotate == false) {
431 		this.labelLength = this.myLabel._estimateTextWidth(this.myLabel._text,14);
432 	
433 		if(this.myLabel._rotate == 360){
434 			this.rotationPointCoordinates.x = this.labelCoordinates.x-8 + this.labelLength/3;
435 			this.rotationPointCoordinates.y = this.labelCoordinates.y-35;	
436 		}
437 		else if(this.myLabel._rotate == 90) {
438 			this.rotationPointCoordinates.x = this.labelCoordinates.x + 35;
439 			this.rotationPointCoordinates.y = this.labelCoordinates.y-8 + this.labelLength/3;			
440 		}
441 		else if(this.myLabel._rotate == 180) {
442 			this.rotationPointCoordinates.x = this.labelCoordinates.x - this.labelLength/2;
443 			this.rotationPointCoordinates.y = this.labelCoordinates.y+35;
444 		}
445 		else if(this.myLabel._rotate == 270) {
446 			this.rotationPointCoordinates.x = this.labelCoordinates.x-35;
447 			this.rotationPointCoordinates.y = this.labelCoordinates.y+8 - this.labelLength/2;	
448 		}
449 		else if(this.myLabel._rotate == 45) {
450 			this.rotationPointCoordinates.x = this.labelCoordinates.x+40;
451 			this.rotationPointCoordinates.y = this.labelCoordinates.y;	
452 		}
453 		else if(this.myLabel._rotate == 135) {
454 			this.rotationPointCoordinates.x = this.labelCoordinates.x;
455 			this.rotationPointCoordinates.y = this.labelCoordinates.y+40;	
456 		}
457 		else if(this.myLabel._rotate == 225) {
458 			this.rotationPointCoordinates.x = this.labelCoordinates.x-40;
459 			this.rotationPointCoordinates.y = this.labelCoordinates.y;	
460 		}
461 		else if(this.myLabel._rotate == 315) {
462 			this.rotationPointCoordinates.x = this.labelCoordinates.x;
463 			this.rotationPointCoordinates.y = this.labelCoordinates.y-40;	
464 		}
465 		else {
466 			//Default
467 			this.rotationPointCoordinates.x = this.labelCoordinates.x-8 + this.labelLength/3;
468 			this.rotationPointCoordinates.y = this.labelCoordinates.y-35;	
469 		}
470 	}
