 26 ORYX.Plugins.Validator = ORYX.Plugins.AbstractPlugin.extend({
 27     construct: function(facade){
 28         this.facade = facade;
 29         
 30         this.active = false;
 31         this.raisedEventIds = [];
 32         
 33         this.buttonId = ORYX.Editor.provideId();
 34         
 35         this.facade.offer({
 36             'name': ORYX.I18N.Validator.name,
 37             'id': this.buttonId,
 38             'functionality': this.load.bind(this),
 39             'group': "Verification",
 40             'icon': ORYX.PATH + "images/checker_validation.png",
 41             'description': ORYX.I18N.Validator.description,
 42             'index': 1,
 43             'toggle': true,
 44             'minShape': 0,
 45             'maxShape': 0
 46         });
 47     },
 48     
 49     load: function(button, pressed){
 50         if (!pressed) {
 51             this.hideOverlays();
 52             this.active = !this.active;
 53         }
 54         else {
 55             this.validate(button);
 56         }
 57     },
 58     
 59     setActive: function(active){
 60         this.active = active;
 61         this.facade.raiseEvent({
 62             type: ORYX.CONFIG.EVENT_BUTTON_UPDATE,
 63             id: this.buttonId,
 64             pressed: active
 65         });
 66     },
 67     
 68     hideOverlays: function(){
 69         this.raisedEventIds.each(function(id){
 70             this.facade.raiseEvent({
 71                 type: ORYX.CONFIG.EVENT_OVERLAY_HIDE,
 72                 id: id
 73             });
 74         }
 75 .bind(this));
 76         
 77         this.raisedEventIds = [];
 78     },
 79     validate: function(button){
 80         this.facade.raiseEvent({
 81             type: ORYX.CONFIG.EVENT_LOADING_ENABLE,
 82             text: ORYX.I18N.Validator.checking
 83         });
 84       
 85         // Send the request to the server.
 86         new Ajax.Request(ORYX.CONFIG.VALIDATOR_URL, {
 87             method: 'POST',
 88             asynchronous: false,
 89             parameters: {
 90                 resource: location.href,
 91                 data: this.getRDFFromDOM()
 92             },
 93             onSuccess: function(request){
 94                 var result = Ext.decode(request.responseText);
 95                 
 96                 this.facade.raiseEvent({
 97                     type: ORYX.CONFIG.EVENT_LOADING_DISABLE
 98                 });
 99                 
100                 // This should be implemented by child instances of validator 
101                 this.handleValidationResponse(result, button);
102             }
103 .bind(this),
104             onFailure: function(){
105                 this.facade.raiseEvent({
106                     type: ORYX.CONFIG.EVENT_LOADING_DISABLE
107                 });
108                 Ext.Msg.alert(ORYX.I18N.Validator.error, ORYX.I18N.Validator.errorDesc);
109             }.bind(this)
110         });
111     },
112     
113     showOverlay: function(shape, errorMsg, errorTitle){
114     
115         var id = "syntaxchecker." + this.raisedEventIds.length;
116         
117         var crossId = ORYX.Editor.provideId();
118         
119         var cross = ORYX.Editor.graft("http://www.w3.org/2000/svg", null, ['path', {
120         	"id":crossId,
121         	"title":"",
122             "stroke-width": 5.0,
123             "stroke": "red",
124             "d": "M20,-5 L5,-20 M5,-5 L20,-20",
125             "line-captions": "round"
126         }]);
127         
128         this.facade.raiseEvent({
129             type: ORYX.CONFIG.EVENT_OVERLAY_SHOW,
130             id: id,
131             shapes: [shape],
132             node: cross,
133             nodePosition: shape instanceof ORYX.Core.Edge ? "START" : "NW"
134         });
135         
136         this.raisedEventIds.push(id);
137         
138         var tooltip = new Ext.ToolTip({
139         	showDelay:50,
140         	title:errorTitle,
141         	html:errorMsg,
142         	target:crossId
143         });   
144         
145         return cross;
146     },
147     
148     /**
149      * Registers handler for deactivating syntax checker as soon as somewhere is clicked...
150      * @param {Ext.Button} Toolbar button
151      */
152     enableDeactivationHandler: function(button){
153         var deactivate = function(){
154             this.setActive(false);
155             this.facade.unregisterOnEvent(ORYX.CONFIG.EVENT_MOUSEDOWN, deactivate);
156         };
157         
158         this.facade.registerOnEvent(ORYX.CONFIG.EVENT_MOUSEDOWN, deactivate.bind(this));
159     }
160 });

162 ORYX.Plugins.BPMNValidator = Ext.extend(ORYX.Plugins.Validator, {
163     handleValidationResponse: function(result, button){
164         var conflictingNodes = result.conflictingNodes;
165         var leadsToEnd = result.leadsToEnd;
166         var unsafeNode = result.unsafeNode;
167         // Only stay active if there is anything to visualize
168         this.setActive(conflictingNodes.size() > 0);
169         
170 //        if (!leadsToEnd) {
171 //            Ext.Msg.alert("Validation Result", "The process will never reach a final state!");
172 //        }
173         if (conflictingNodes.size() > 0) {
174             conflictingNodes.each(function(node){
175                 sh = this.facade.getCanvas().getChildShapeByResourceId(node.id);
176                 if (sh) {
177                     this.showOverlay(sh, ORYX.I18N.Validator.bpmnDeadlock,  ORYX.I18N.Validator.bpmnDeadlockTitle);
178                 }
179             }.bind(this));
180         }
181         if(unsafeNode) {
182         	var shape = this.facade.getCanvas().getChildShapeByResourceId(unsafeNode);
183         	if (shape) {
184                 this.showOverlay(shape, ORYX.I18N.Validator.bpmnUnsafe, ORYX.I18N.Validator.bpmnUnsafeTitle);
185             }
186         }
187         if(leadsToEnd && conflictingNodes.size() === 0 && !unsafeNode) {
188         	this.facade.raiseEvent({
189     			type:ORYX.CONFIG.EVENT_LOADING_STATUS,
190     			text:ORYX.I18N.Validator.noErrors,
191     			timeout:10000
192     		});
193         	//Ext.Msg.alert(ORYX.I18N.Validator.result, ORYX.I18N.Validator.noErrors);
194         } else if(!leadsToEnd && conflictingNodes.size() === 0 && !unsafeNode){
195         	Ext.Msg.alert(ORYX.I18N.Validator.bpmnLeadsToNoEndTitle, ORYX.I18N.Validator.bpmnLeadsToNoEnd);
196         } else {
197         	this.enableDeactivationHandler(button);
198         	//show a status message with a hint to the error messages in the tooltip
199             this.facade.raiseEvent({
200     			type:ORYX.CONFIG.EVENT_LOADING_STATUS,
201     			text:ORYX.I18N.SyntaxChecker.notice,
202     			timeout:10000
203     		});
204         }
205     }
206 });

208 ORYX.Plugins.EPCValidator = Ext.extend(ORYX.Plugins.Validator, {
209   getLabelOfShape: function(node){
210     if(node.properties["oryx-title"] === ""){
211       return node.id;
212     } else {
213       return node.properties["oryx-title"];
214     }
215   },
216   findShapeById: function(id){
217     return this.facade.getCanvas().getChildShapeByResourceId(id);
218   },
219   
220     handleValidationResponse: function(result, button){
221       //TODO use Ext XTemplate
222         var isSound = result.isSound;
223         var badStartArcs = result.badStartArcs;
224         var badEndArcs = result.badEndArcs;
225         var goodInitialMarkings = result.goodInitialMarkings;
226         var goodFinalMarkings = result.goodFinalMarkings;
227         
228         var message = "";
229         
230         if (isSound) {
231           message += ORYX.I18N.Validator.epcIsSound;
232         } else {
233           message += ORYX.I18N.Validator.epcNotSound;
234         }
235         
236         message += "<hr />";
237         
238         var arrayOfArraysToMessage = function(arrayOfArrays, formatter){
239           var message = "<ul>"
240           arrayOfArrays.each(function(array){
241             message += "<li> - ";
242             array.each(function(element){
243               message += '"' + formatter(element) + '" ';
244             });
245             message += "</li>";
246           });
247           message += "</ul>";
248           return message;
249         }
250         var arrayToMessage = function(array, formatter){
251           var message = "<ul>"
252           array.each(function(element){
253             message += "<li> - " + formatter(element) + "</li>";
254           });
255           message += "</ul>";
256           return message;
257         }
258         
259         message += "<p>There are "+ goodInitialMarkings.length +" initial markings which does not run into a deadlock.</p>";
260         message += arrayOfArraysToMessage(goodInitialMarkings, function(arc){
261           return this.getLabelOfShape(this.findShapeById(arc.id).getIncomingShapes()[0]);
262         }.createDelegate(this));
263         message += "<p>The initial markings do not include "+ badStartArcs.length +" start nodes.</p>";
264         message += arrayToMessage(badStartArcs, function(arc){
265           return this.getLabelOfShape(this.findShapeById(arc.id).getIncomingShapes()[0]);
266         }.createDelegate(this));
267         
268         message += "<hr />";
269         
270         message += "<p>There are "+ goodFinalMarkings.length +" final markings which can be reached from the initial markings.</p>";
271         message += arrayOfArraysToMessage(goodFinalMarkings, function(arc){
272           return this.getLabelOfShape(this.findShapeById(arc.id).getOutgoingShapes()[0]);
273         }.createDelegate(this));
274         message += "<p>The final markings do not include "+ badEndArcs.length +" end nodes.</p>";
275         message += arrayToMessage(badEndArcs, function(arc){
276           return this.getLabelOfShape(this.findShapeById(arc.id).getOutgoingShapes()[0]);
277         }.createDelegate(this))
278         
279         message += "<hr />";
280         
281         message += "<p><i>Remark: Set titles of functions and events to get some nicer output (names instead of ids)</i></p>"
282         
283         Ext.Msg.alert('Validation Result', message);
284         
285         this.setActive(false);
286     }
287 });
