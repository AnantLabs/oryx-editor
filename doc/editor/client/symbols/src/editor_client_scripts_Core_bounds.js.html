 31 /**
 32  * @classDescription With Bounds you can set and get position and size of UIObjects.
 33  */
 34 ORYX.Core.Bounds = {
 35 
 36 	/**
 37 	 * Constructor
 38 	 */
 39 	construct: function() {
 40 		this._changedCallbacks = []; //register a callback with changedCallacks.push(this.method.bind(this));
 41 		this.a = {};
 42 		this.b = {};
 43 		this.set.apply(this, arguments);
 44 		this.suspendChange = false;
 45 		this.changedWhileSuspend = false;
 46 	},
 47 	
 48 	/**
 49 	 * Calls all registered callbacks.
 50 	 */
 51 	_changed: function(sizeChanged) {
 52 		if(!this.suspendChange) {
 53 			this._changedCallbacks.each(function(callback) {
 54 				callback(this, sizeChanged);
 55 			}.bind(this));
 56 			this.changedWhileSuspend = false;
 57 		} else
 58 			this.changedWhileSuspend = true;
 59 	},
 60 	
 61 	/**
 62 	 * Registers a callback that is called, if the bounds changes.
 63 	 * @param callback {Function} The callback function.
 64 	 */
 65 	registerCallback: function(callback) {
 66 		if(!this._changedCallbacks.member(callback)) {
 67 			this._changedCallbacks.push(callback);	
 68 		}
 69 	},
 70 	
 71 	/**
 72 	 * Unregisters a callback.
 73 	 * @param callback {Function} The callback function.
 74 	 */
 75 	unregisterCallback: function(callback) {
 76 			this._changedCallbacks = this._changedCallbacks.without(callback);
 77 	},
 78 	
 79 	/**
 80 	 * Sets position and size of the shape dependent of four coordinates
 81 	 * (set(ax, ay, bx, by);), two points (set({x: ax, y: ay}, {x: bx, y: by});)
 82 	 * or one bound (set({a: {x: ax, y: ay}, b: {x: bx, y: by}});).
 83 	 */
 84 	set: function() {
 85 		
 86 		var changed = false;
 87 		
 88 		switch (arguments.length) {
 89 		
 90 			case 1:
 91 				if(this.a.x !== arguments[0].a.x) {
 92 					changed = true;
 93 					this.a.x = arguments[0].a.x;
 94 				}
 95 				if(this.a.y !== arguments[0].a.y) {
 96 					changed = true;
 97 					this.a.y = arguments[0].a.y;
 98 				}
 99 				if(this.b.x !== arguments[0].b.x) {
100 					changed = true;
101 					this.b.x = arguments[0].b.x;
102 				}
103 				if(this.b.y !== arguments[0].b.y) {
104 					changed = true;
105 					this.b.y = arguments[0].b.y;
106 				}
107 				break;
108 			
109 			case 2:
110 				var ax = Math.min(arguments[0].x, arguments[1].x);
111 				var ay = Math.min(arguments[0].y, arguments[1].y);
112 				var bx = Math.max(arguments[0].x, arguments[1].x);
113 				var by = Math.max(arguments[0].y, arguments[1].y);
114 				if(this.a.x !== ax) {
115 					changed = true;
116 					this.a.x = ax;
117 				}
118 				if(this.a.y !== ay) {
119 					changed = true;
120 					this.a.y = ay;
121 				}
122 				if(this.b.x !== bx) {
123 					changed = true;
124 					this.b.x = bx;
125 				}
126 				if(this.b.y !== by) {
127 					changed = true;
128 					this.b.y = by;
129 				}
130 				break;
131 			
132 			case 4:
133 				var ax = Math.min(arguments[0], arguments[2]);
134 				var ay = Math.min(arguments[1], arguments[3]);
135 				var bx = Math.max(arguments[0], arguments[2]);
136 				var by = Math.max(arguments[1], arguments[3]);
137 				if(this.a.x !== ax) {
138 					changed = true;
139 					this.a.x = ax;
140 				}
141 				if(this.a.y !== ay) {
142 					changed = true;
143 					this.a.y = ay;
144 				}
145 				if(this.b.x !== bx) {
146 					changed = true;
147 					this.b.x = bx;
148 				}
149 				if(this.b.y !== by) {
150 					changed = true;
151 					this.b.y = by;
152 				}
153 				break;
154 		}
155 		
156 		if(changed) {
157 			this._changed(true);
158 		}
159 	},
160 	
161 	/**
162 	 * Moves the bounds so that the point p will be the new upper left corner.
163 	 * @param {Point} p
164 	 * or
165 	 * @param {Number} x
166 	 * @param {Number} y
167 	 */
168 	moveTo: function() {
169 		
170 		var currentPosition = this.upperLeft();
171 		switch (arguments.length) {
172 			case 1:
173 				this.moveBy({
174 					x: arguments[0].x - currentPosition.x,
175 					y: arguments[0].y - currentPosition.y
176 				});
177 				break;
178 			case 2:
179 				this.moveBy({
180 					x: arguments[0] - currentPosition.x,
181 					y: arguments[1] - currentPosition.y
182 				});
183 				break;
184 			default:
185 				//TODO error
186 		}
187 		
188 	},
189 	
190 	/**
191 	 * Moves the bounds relatively by p.
192 	 * @param {Point} p
193 	 * or
194 	 * @param {Number} x
195 	 * @param {Number} y
196 	 * 
197 	 */
198 	moveBy: function() {
199 		var changed = false;
200 		
201 		switch (arguments.length) {
202 			case 1:
203 				var p = arguments[0];
204 				if(p.x !== 0 || p.y !== 0) {
205 					changed = true;
206 					this.a.x += p.x;
207 					this.b.x += p.x;
208 					this.a.y += p.y;
209 					this.b.y += p.y;
210 				}
211 				break;	
212 			case 2:
213 				var x = arguments[0];
214 				var y = arguments[1];
215 				if(x !== 0 || y !== 0) {
216 					changed = true;
217 					this.a.x += x;
218 					this.b.x += x;
219 					this.a.y += y;
220 					this.b.y += y;
221 				}
222 				break;	
223 			default:
224 				//TODO error
225 		}
226 		
227 		if(changed) {
228 			this._changed();
229 		}
230 	},
231 	
232 	/***
233 	 * Includes the bounds b into the current bounds.
234 	 * @param {Bounds} b
235 	 */
236 	include: function(b) {
237 		
238 		if( (this.a.x === undefined) && (this.a.y === undefined) &&
239 			(this.b.x === undefined) && (this.b.y === undefined)) {
240 			return b;
241 		};
242 		
243 		var cx = Math.min(this.a.x,b.a.x);
244 		var cy = Math.min(this.a.y,b.a.y);
245 		
246 		var dx = Math.max(this.b.x,b.b.x);
247 		var dy = Math.max(this.b.y,b.b.y);
248 
249 		
250 		this.set(cx, cy, dx, dy);
251 	},
252 	
253 	/**
254 	 * Relatively extends the bounds by p.
255 	 * @param {Point} p
256 	 */
257 	extend: function(p) {
258 		
259 		if(p.x !== 0 || p.y !== 0) {
260 			// this is over cross for the case that a and b have same coordinates.
261 			//((this.a.x > this.b.x) ? this.a : this.b).x += p.x;
262 			//((this.b.y > this.a.y) ? this.b : this.a).y += p.y;
263 			this.b.x += p.x;
264 			this.b.y += p.y;
265 			
266 			this._changed(true);
267 		}
268 	},
269 	
270 	/**
271 	 * Widens the scope of the bounds by x.
272 	 * @param {Number} x
273 	 */
274 	widen: function(x) {
275 		if (x !== 0) {
276 			this.suspendChange = true;
277 			this.moveBy({x: -x, y: -x});
278 			this.extend({x: 2*x, y: 2*x});
279 			this.suspendChange = false;
280 			if(this.changedWhileSuspend) {
281 				this._changed(true);
282 			}
283 		}
284 	},
285 	
286 	/**
287 	 * Returns the upper left corner's point regardless of the
288 	 * bound delimiter points.
289 	 */
290 	upperLeft: function() {
291 		
292 		return {x:this.a.x, y:this.a.y};
293 	},
294 	
295 	/**
296 	 * Returns the lower Right left corner's point regardless of the
297 	 * bound delimiter points.
298 	 */
299 	lowerRight: function() {
300 		
301 		return {x:this.b.x, y:this.b.y};
302 	},
303 	
304 	/**
305 	 * @return {Number} Width of bounds.
306 	 */
307 	width: function() {
308 		return this.b.x - this.a.x;
309 	},
310 	
311 	/**
312 	 * @return {Number} Height of bounds.
313 	 */
314 	height: function() {
315 		return this.b.y - this.a.y;
316 	},
317 	
318 	/**
319 	 * @return {Point} The center point of this bounds.
320 	 */
321 	center: function() {
322 		return {
323 			x: (this.a.x + this.b.x)/2.0, 
324 			y: (this.a.y + this.b.y)/2.0
325 		};
326 	},
327 
328 	
329 	/**
330 	 * @return {Point} The center point of this bounds relative to upperLeft.
331 	 */
332 	midPoint: function() {
333 		return {
334 			x: (this.b.x - this.a.x)/2.0, 
335 			y: (this.b.y - this.a.y)/2.0
336 		};
337 	},
338 		
339 	/**
340 	 * Moves the center point of this bounds to the new position.
341 	 * @param p {Point} 
342 	 * or
343 	 * @param x {Number}
344 	 * @param y {Number}
345 	 */
346 	centerMoveTo: function() {
347 		var currentPosition = this.center();
348 		
349 		switch (arguments.length) {
350 			
351 			case 1:
352 				this.moveBy(arguments[0].x - currentPosition.x,
353 							arguments[0].y - currentPosition.y);
354 				break;
355 			
356 			case 2:
357 				this.moveBy(arguments[0] - currentPosition.x,
358 							arguments[1] - currentPosition.y);
359 				break;
360 		}
361 	},
362 	
363 	isIncluded: function(point, offset) {
364 		
365 		var pointX, pointY, offset;
366 
367 		// Get the the two Points	
368 		switch(arguments.length) {
369 			case 1:
370 				pointX = arguments[0].x;
371 				pointY = arguments[0].y;
372 				offset = 0;
373 				
374 				break;
375 			case 2:
376 				if(arguments[0].x && arguments[0].y) {
377 					pointX = arguments[0].x;
378 					pointY = arguments[0].y;
379 					offset = Math.abs(arguments[1]);
380 				} else {
381 					pointX = arguments[0];
382 					pointY = arguments[1];
383 					offset = 0;
384 				}
385 				break;
386 			case 3:
387 				pointX = arguments[0];
388 				pointY = arguments[1];
389 				offset = Math.abs(arguments[2]);
390 				break;
391 			default:
392 				throw "isIncluded needs one, two or three arguments";
393 		}
394 				
395 		var ul = this.upperLeft();
396 		var lr = this.lowerRight();
397 		
398 		if(pointX >= ul.x - offset 
399 			&& pointX <= lr.x + offset && pointY >= ul.y - offset 
400 			&& pointY <= lr.y + offset)
401 			return true;
402 		else 
403 			return false;
404 	},
405 	
406 	/**
407 	 * @return {Bounds} A copy of this bounds.
408 	 */
409 	clone: function() {
410 		
411 		//Returns a new bounds object without the callback
412 		// references of the original bounds
413 		return new ORYX.Core.Bounds(this);
414 	},
415 	
416 	toString: function() {
417 		
418 		return "( "+this.a.x+" | "+this.a.y+" )/( "+this.b.x+" | "+this.b.y+" )";
419 	},
420 	
421 	serializeForERDF: function() {
422 
423 		return this.a.x+","+this.a.y+","+this.b.x+","+this.b.y;
424 	}
425  };
