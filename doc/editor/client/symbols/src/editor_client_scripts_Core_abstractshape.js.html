 42 	construct: function(options, stencil) {
 43 		
 44 		arguments.callee.$.construct.apply(this, arguments);
 45 		
 46 		this.resourceId = ORYX.Editor.provideId(); //Id of resource in DOM
 47 		
 48 		// stencil reference
 49 		this._stencil = stencil;
 50 		// if the stencil defines a super stencil that should be used for its instances, set it.
 51 		if (this._stencil._jsonStencil.superId){
 52 			stencilId = this._stencil.id()
 53 			superStencilId = stencilId.substring(0, stencilId.indexOf("#") + 1) + stencil._jsonStencil.superId;
 54 			stencilSet =  this._stencil.stencilSet();
 55 			this._stencil = stencilSet.stencil(superStencilId);
 56 		}
 57 		
 58 		//Hash map for all properties. Only stores the values of the properties.
 59 		this.properties = new Hash();
 60 		this.propertiesChanged = new Hash();
 61 
 62 		// List of properties which are not included in the stencilset, 
 63 		// but which gets (de)serialized
 64 		this.hiddenProperties = new Hash();
 65 		
 66 		
 67 		//Initialization of property map and initial value.
 68 		this._stencil.properties().each((function(property) {
 69 			var key = property.prefix() + "-" + property.id();
 70 			this.properties[key] = property.value();
 71 			this.propertiesChanged[key] = true;
 72 		}).bind(this));
 73 		
 74 		// if super stencil was defined, also regard stencil's properties:
 75 		if (stencil._jsonStencil.superId) {
 76 			stencil.properties().each((function(property) {
 77 				var key = property.prefix() + "-" + property.id();
 78 				var value = property.value();
 79 				var oldValue = this.properties[key];
 80 				this.properties[key] = value;
 81 				this.propertiesChanged[key] = true;
 82 
 83 				// Raise an event, to show that the property has changed
 84 				// required for plugins like processLink.js
 85 				//window.setTimeout( function(){
 86 
 87 					this._delegateEvent({
 88 							type	: ORYX.CONFIG.EVENT_PROPERTY_CHANGED, 
 89 							name	: key, 
 90 							value	: value,
 91 							oldValue: oldValue
 92 						});
 93 
 94 				//}.bind(this), 10)
 95 
 96 			}).bind(this));
 97 		}

211 	//TODO deep iterator
212 	getAbstractShapesAtPosition: function() {
213 		var x, y;
214 		switch (arguments.length) {
215 			case 1:
216 				x = arguments[0].x;
217 				y = arguments[0].y;
218 				break;
219 			case 2:	//two or more arguments
220 				x = arguments[0];
221 				y = arguments[1];
222 				break;
223 			default:
224 				throw "getAbstractShapesAtPosition needs 1 or 2 arguments!"
225 		}
226 
227 		if(this.isPointIncluded(x, y)) {
228 
229 			var result = [];
230 			result.push(this);
231 
232 			//check, if one child is at that position						
233 			
234 			
235 			var childNodes = this.getChildNodes();
236 			var childEdges = this.getChildEdges();
237 			
238 			[childNodes, childEdges].each(function(ne){
239 				var nodesAtPosition = new Hash();
240 				
241 				ne.each(function(node) {
242 					if(!node.isVisible){ return }
243 					var candidates = node.getAbstractShapesAtPosition( x , y );
244 					if(candidates.length > 0) {
245 						var nodesInZOrder = $A(node.node.parentNode.childNodes);
246 						var zOrderIndex = nodesInZOrder.indexOf(node.node);
247 						nodesAtPosition[zOrderIndex] = candidates;
248 					}
249 				});
250 				
251 				nodesAtPosition.keys().sort().each(function(key) {
252 					result = result.concat(nodesAtPosition[key]);
253 				});
254  			});
255 						
256 			return result;
257 			
258 		} else {
259 			return [];
260 		}

268 	setProperty: function(key, value, force) {
269 		var oldValue = this.properties[key];
270 		if(oldValue !== value || force === true) {
271 			this.properties[key] = value;
272 			this.propertiesChanged[key] = true;
273 			this._changed();
274 			
275 			// Raise an event, to show that the property has changed
276 			//window.setTimeout( function(){
277 
278 			if (!this._isInSetProperty) {
279 				this._isInSetProperty = true;
280 				
281 				this._delegateEvent({
282 						type	: ORYX.CONFIG.EVENT_PROPERTY_CHANGED, 
283 						elements : [this],
284 						name	: key, 
285 						value	: value,
286 						oldValue: oldValue
287 					});
288 				
289 				delete this._isInSetProperty;
290 			}
291 			//}.bind(this), 10)
292 		}

328 	serialize: function() {
329 		var serializedObject = [];
330 		
331 		// Add the type
332 		serializedObject.push({name: 'type', prefix:'oryx', value: this.getStencil().id(), type: 'literal'});	
333 	
334 		// Add hidden properties
335 		this.hiddenProperties.each(function(prop){
336 			serializedObject.push({name: prop.key.replace("oryx-", ""), prefix: "oryx", value: prop.value, type: 'literal'});
337 		}.bind(this));
338 		
339 		// Add all properties
340 		this.getStencil().properties().each((function(property){
341 			
342 			var prefix = property.prefix();	// Get prefix
343 			var name = property.id();		// Get name
344 			
345 			//if(typeof this.properties[prefix+'-'+name] == 'boolean' || this.properties[prefix+'-'+name] != "")
346 				serializedObject.push({name: name, prefix: prefix, value: this.properties[prefix+'-'+name], type: 'literal'});
347 
348 		}).bind(this));
349 		
350 		return serializedObject;
351 	},

354 	deserialize: function(serialize){
355 		// Search in Serialize
356 		var initializedDocker = 0;
357 		
358 		// Sort properties so that the hidden properties are first in the list
359 		serialize = serialize.sort(function(a,b){ return Number(this.properties.keys().member(a.prefix+"-"+a.name)) > Number(this.properties.keys().member(b.prefix+"-"+b.name)) ? -1 : 0 }.bind(this));
360 		
361 		serialize.each((function(obj){
362 			
363 			var name 	= obj.name;
364 			var prefix 	= obj.prefix;
365 			var value 	= obj.value;
366             
367             // Complex properties can be real json objects, encode them to a string
368             if(Ext.type(value) === "object") value = Ext.encode(value);
369 
370 			switch(prefix + "-" + name){
371 				case 'raziel-parent': 
372 							// Set parent
373 							if(!this.parent) {break};
374 							
375 							// Set outgoing Shape
376 							var parent = this.getCanvas().getChildShapeByResourceId(value);
377 							if(parent) {
378 								parent.add(this);
379 							}
380 							
381 							break;											
382 				default:
383 							// Set property
384 							if(this.properties.keys().member(prefix+"-"+name)) {
385 								this.setProperty(prefix+"-"+name, value);
386 							} else if(!(name === "bounds"||name === "parent"||name === "target"||name === "dockers"||name === "docker"||name === "outgoing"||name === "incoming")) {
387 								this.setHiddenProperty(prefix+"-"+name, value);
388 							}
389 					
390 			}
391 		}).bind(this));
392 	},

396     /**
397      * Converts the shape to a JSON representation.
398      * @return {Object} A JSON object with included ORYX.Core.AbstractShape.JSONHelper and getShape() method.
399      */
400     toJSON: function(){
401         var json = {
402             resourceId: this.resourceId,
403             properties: Ext.apply({}, this.properties, this.hiddenProperties).inject({}, function(props, prop){
404               var key = prop[0];
405               var value = prop[1];
406                 
407               //If complex property, value should be a json object
408               if(this.getStencil().property(key)
409                 && this.getStencil().property(key).type() === ORYX.CONFIG.TYPE_COMPLEX 
410                 && Ext.type(value) === "string"){
411                   try {value = Ext.decode(value);} catch(error){}
412               }
413               
414               //Takes "my_property" instead of "oryx-my_property" as key
415               key = key.replace(/^[\w_]+-/, "");
416               props[key] = value;
417               
418               return props;
419             }.bind(this)),
420             stencil: {
421                 id: this.getStencil().idWithoutNs()
422             },
423             childShapes: this.getChildShapes().map(function(shape){
424                 return shape.toJSON()
425             })
426         };
427         
428         if(this.getOutgoingShapes){
429             json.outgoing = this.getOutgoingShapes().map(function(shape){
430                 return {
431                     resourceId: shape.resourceId
432                 };
433             });
434         }
435         
436         if(this.bounds){
437             json.bounds = { 
438                 lowerRight: this.bounds.lowerRight(), 
439                 upperLeft: this.bounds.upperLeft() 
440             };
441         }
442         
443         if(this.dockers){
444             json.dockers = this.dockers.map(function(docker){
445                 var d = docker.getDockedShape() && docker.referencePoint ? docker.referencePoint : docker.bounds.center();
446                 d.getDocker = function(){return docker;};
447                 return d;
448             })
449         }
450         
451         Ext.apply(json, ORYX.Core.AbstractShape.JSONHelper);
452         
453         // do not pollute the json attributes (for serialization), so put the corresponding
454         // shape is encapsulated in a method
455         json.getShape = function(){
456             return this;
457         }.bind(this);
458         
459         return json;
460     }

463 /**
464  * @namespace Collection of methods which can be used on a shape json object (ORYX.Core.AbstractShape#toJSON()).
465  * @example
466  * Ext.apply(shapeAsJson, ORYX.Core.AbstractShape.JSONHelper);
467  */
468 ORYX.Core.AbstractShape.JSONHelper = {
469      /**
470       * Iterates over each child shape.
471       * @param {Object} iterator Iterator function getting a child shape and his parent as arguments.
472       * @param {boolean} [deep=false] Iterate recursively (childShapes of childShapes)
473       * @param {boolean} [modify=false] If true, the result of the iterator function is taken as new shape, return false to delete it. This enables modifying the object while iterating through the child shapes.
474       * @example
475       * // Increases the lowerRight x value of each direct child shape by one. 
476       * myShapeAsJson.eachChild(function(shape, parentShape){
477       *     shape.bounds.lowerRight.x = shape.bounds.lowerRight.x + 1;
478       *     return shape;
479       * }, false, true);
480       */
481      eachChild: function(iterator, deep, modify){
482          if(!this.childShapes) return;
483          
484          var newChildShapes = []; //needed if modify = true
485          
486          this.childShapes.each(function(shape){
487              var res = iterator(shape, this);
488              if(res) newChildShapes.push(res); //if false is returned, and modify = true, current shape is deleted.
489              
490              if(deep) shape.eachChild(iterator, deep, modify);
491          }.bind(this));
492          
493          if(modify) this.childShapes = newChildShapes;
494      },
495      
496      getChildShapes: function(deep){
497          var allShapes = this.childShapes;
498          
499          if(deep){
500              this.eachChild(function(shape){
501                  allShapes = allShapes.concat(shape.getChildShapes(deep));
502              }, true);
503          }
504          
505          return allShapes;
506      },
507      
508      /**
509       * @return {String} Serialized JSON object
510       */
511      serialize: function(){
512          return Ext.encode(this);
513      }
