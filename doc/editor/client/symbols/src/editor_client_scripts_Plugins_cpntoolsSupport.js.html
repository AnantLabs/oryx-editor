 60 	resetTokenPosition: function()
 61 	{
 62 		// Get selected places		
 63 		var allplaces = this.facade.getSelection().findAll(function(selectedItem) {
 64 			return (selectedItem.getStencil().id() === "http://b3mn.org/stencilset/coloredpetrinet#Place");
 65 		});
 66 		
 67 		if (allplaces.length > 0)
 68 		{
 69 			allplaces.each(function(place) {
 70 				
 71 				var placeBounds = place.absoluteBounds();
 72 				var placeCenter = placeBounds.center();
 73 				
 74 				// Calculate radius in order to check if a token is in the place
 75 				var radiusY = placeCenter.y - placeBounds.upperLeft().y;
 76 				var radiusX = placeCenter.x - placeBounds.upperLeft().x;
 77 				var radius = Math.min(radiusY,radiusX);
 78 				var c = radius / 2;
 79 				
 80 				// Get all tokens inside the place 
 81 				var alltokens = place.getChildNodes(false).findAll(function(child) {
 82 					return (child.getStencil().id() === "http://b3mn.org/stencilset/coloredpetrinet#Token");
 83 				});
 84 				
 85 				if (alltokens.length > 0)
 86 				{
 87 					var i = 0;
 88 					var x = 0;
 89 					var y = 0;
 90 					
 91 					alltokens.each(function(token) {
 92 						var tokenBounds = token.absoluteBounds();
 93 						var tokenCenter = tokenBounds.center();
 94 						
 95 						// Calculate the distance between token and center of the place
 96 						var diffX = placeCenter.x - tokenCenter.x;
 97 						var diffY = placeCenter.y - tokenCenter.y;
 98 						var distanceToPlaceCenter= diffX*diffX + diffY*diffY; // take care it's squared
 99 						
100 						// Check if the token is in the place
101 						if (radius*radius <= distanceToPlaceCenter)
102 						{	// if the token is out of the place, calculate the position for the token
103 							// the token are positioned in circle which is in the place
104 							y = Math.round(Math.sin((Math.PI / 6) * i) * c);
105 							x = Math.round(Math.cos((Math.PI / 6) * i) * c);
106 							// take care centerMoveTo is referred to the position in the selected place (not absolute) 
107 							token.bounds.centerMoveTo(place.bounds.width() / 2  + x, place.bounds.height() / 2 + y);
108 							token.update();
109 							i = i + 1;
110 						}
111 					});					
112 				}
113 			});
114 
115 		}			
116 		this.facade.getCanvas().update();
117 	},

121 	_sendRequest: function( url, method, params, successcallback, failedcallback ){
122 
123 		var suc = false;
124 
125 		new Ajax.Request(
126 		url, 
127 		{
128            method			: method,
129            asynchronous		: false,
130            parameters		: params,
131 		   onSuccess		: function(transport) 
132 		   {
133 				suc = true;
134 		
135 				if(successcallback)
136 				{
137 					successcallback( transport.responseText )	
138 				}
139 		
140 		   }.bind(this),
141 		   onFailure		: function(transport) 
142 		   {
143 				if(failedcallback)
144 				{							
145 					failedcallback();							
146 				} 
147 				else 
148 				{
149 					this._showErrorMessageBox(ORYX.I18N.Oryx.title, ORYX.I18N.cpntoolsSupport.serverConnectionFailed);
150 					ORYX.log.warn("Communication failed: " + transport.responseText);	
151 				}					
152 		   }.bind(this)		
153 		});
154 		
155 		return suc;		
156 	},

286 	_getAllPages: function(cpnXML, loadMask)
287 	{
288 		var parser = new DOMParser();
289 		var xmlDoc = parser.parseFromString(cpnXML,"text/xml");
290 		var allPages = xmlDoc.getElementsByTagName("page");
291 		
292 		// If there are no pages then it is propably that cpnXML is not a cpn - File
293 		if (allPages.length == 0)
294 		{
295 			loadMask.hide();
296 			this._showErrorMessageBox(ORYX.I18N.cpntoolsSupport.title, ORYX.I18N.cpntoolsSupport.wrongCPNFile);
297 			
298 			return;
299 		}
300 		
301 		
302 		if (allPages.length == 1)
303 		{
304 			pageAttr = allPages[0].children[0];
305 			pageName = pageAttr.attributes[0].nodeValue;
306 			
307 			this._sendRequest(
308 					ORYX.CONFIG.CPNTOOLSIMPORTER,
309 					'POST',
310 					{ 
311 						'pagesToImport': pageName,
312 						'data' : cpnXML 
313 					},
314 					function( arg )
315 					{
316 						if (arg.startsWith("error:"))
317 						{
318 							this._showErrorMessageBox(ORYX.I18N.Oryx.title, arg);
319 							loadMask.hide();
320 						}
321 						else
322 						{
323 							this.facade.importJSON(arg); 
324 							loadMask.hide();							
325 						}
326 					}.bind(this),
327 					function()
328 					{
329 						loadMask.hide();
330 						this._showErrorMessageBox(ORYX.I18N.Oryx.title, ORYX.I18N.cpntoolsSupport.serverConnectionFailed);
331 					}.bind(this)
332 				);
333 			
334 			return;
335 		}
336 		
337 		var i, pageName, data = [];
338 		for (i = 0; i < allPages.length; i++)
339 		{
340 			pageAttr = allPages[i].children[0];
341 			pageName = pageAttr.attributes[0].nodeValue;
342 			data.push([pageName]);
343 		}
344 		
345 		loadMask.hide();
346 		this.showPageDialog(data, cpnXML);		
347 	},
