 27 /**
 28  * Enables exporting and importing current model in JSON.
 29  */
 30 ORYX.Plugins.JSONSupport = ORYX.Plugins.AbstractPlugin.extend({
 31 
 32     construct: function(){
 33         // Call super class constructor
 34         arguments.callee.$.construct.apply(this, arguments);
 35         
 36         this.facade.offer({
 37             'name': ORYX.I18N.JSONSupport.exp.name,
 38             'functionality': this.exportJSON.bind(this),
 39             'group': ORYX.I18N.JSONSupport.exp.group,
 40             dropDownGroupIcon: ORYX.PATH + "images/export2.png",
 41 			'icon': ORYX.PATH + "images/page_white_javascript.png",
 42             'description': ORYX.I18N.JSONSupport.exp.desc,
 43             'index': 0,
 44             'minShape': 0,
 45             'maxShape': 0
 46         });
 47         
 48         this.facade.offer({
 49             'name': ORYX.I18N.JSONSupport.imp.name,
 50             'functionality': this.showImportDialog.bind(this),
 51             'group': ORYX.I18N.JSONSupport.imp.group,
 52             dropDownGroupIcon: ORYX.PATH + "images/import.png",
 53 			'icon': ORYX.PATH + "images/page_white_javascript.png",
 54             'description': ORYX.I18N.JSONSupport.imp.desc,
 55             'index': 1,
 56             'minShape': 0,
 57             'maxShape': 0
 58         });
 59     },
 60     
 61     exportJSON: function(){
 62         var json = this.facade.getSerializedJSON();
 63         this.openDownloadWindow(window.document.title + ".json", json);
 64     },
 65     
 66     /**
 67      * Opens a upload dialog.
 68      *
 69      */
 70     showImportDialog: function(successCallback){
 71     
 72         var form = new Ext.form.FormPanel({
 73             baseCls: 'x-plain',
 74             labelWidth: 50,
 75             defaultType: 'textfield',
 76             items: [{
 77                 text: ORYX.I18N.JSONSupport.imp.selectFile,
 78                 style: 'font-size:12px;margin-bottom:10px;display:block;',
 79                 anchor: '100%',
 80                 xtype: 'label'
 81             }, {
 82                 fieldLabel: ORYX.I18N.JSONSupport.imp.file,
 83                 name: 'subject',
 84                 inputType: 'file',
 85                 style: 'margin-bottom:10px;display:block;',
 86                 itemCls: 'ext_specific_window_overflow'
 87             }, {
 88                 xtype: 'textarea',
 89                 hideLabel: true,
 90                 name: 'msg',
 91                 anchor: '100% -63'
 92             }]
 93         });
 94         
 95         // Create the panel
 96         var dialog = new Ext.Window({
 97             autoCreate: true,
 98             layout: 'fit',
 99             plain: true,
100             bodyStyle: 'padding:5px;',
101             title: ORYX.I18N.JSONSupport.imp.name,
102             height: 350,
103             width: 500,
104             modal: true,
105             fixedcenter: true,
106             shadow: true,
107             proxyDrag: true,
108             resizable: true,
109             items: [form],
110             buttons: [{
111                 text: ORYX.I18N.JSONSupport.imp.btnImp,
112                 handler: function(){
113                 
114                     var loadMask = new Ext.LoadMask(Ext.getBody(), {
115                         msg: ORYX.I18N.JSONSupport.imp.progress
116                     });
117                     loadMask.show();
118                     
119                     window.setTimeout(function(){
120                         var json = form.items.items[2].getValue();
121                         try {
122                             this.facade.importJSON(json, true);
123                             dialog.close();
124                         } 
125                         catch (error) {
126                             Ext.Msg.alert(ORYX.I18N.JSONSupport.imp.syntaxError, error.message);
127                         }
128                         finally {
129                             loadMask.hide();
130                         }
131                     }.bind(this), 100);
132                     
133                 }.bind(this)
134             }, {
135                 text: ORYX.I18N.JSONSupport.imp.btnClose,
136                 handler: function(){
137                     dialog.close();
138                 }.bind(this)
139             }]
140         });
141         
142         // Show the panel
143         dialog.show();
144         
145         // Adds the change event handler to 
146         form.items.items[1].getEl().dom.addEventListener('change', function(evt){
147             var text = evt.target.files[0].getAsText('UTF-8');
148             form.items.items[2].setValue(text);
149         }, true)
150         
151     }
152     
153 });
