 30 ORYX.Plugins.TransformationDownloadDialogForBPEL4Chor = {
 31 		
 32 	construct: function() {
 33 		arguments.callee.$.construct.apply(this, arguments);
 34 	},
 35 	/**
 36 	 * Opens a dialog that presents the results of a transformation.
 37 	 * The dialog shows a list containing the resulting XML files.
 38 	 * Each file can be shown in a new window or downloaded.
 39      *
 40 	 * @param {Object} data The data to be shown in the dialog
 41 	 * Format: array with three elements: 
 42 	 *   * file - the file
 43 	 *   * result - the content of file, may also be an error message.
 44 	 *   * info - status of the result: "success" or "error"
 45 	 */
 46 	openResultDialog: function(data) {
 47 
 48 		var ds = new Ext.data.Store({
 49 	        proxy: new Ext.data.MemoryProxy(data),
 50 	        reader: new Ext.data.ArrayReader({}, [
 51 	               {name: 'file', type: 'string'},
 52 	               {name: 'result', type: 'string'},
 53 	               {name: 'info', type: 'string'}
 54 	        	])
 55 		});
 56 		
 57 		ds.load();
 58 
 59 		// renderer
 60 		var infoRenderer = function (val){
 61             if(val == "success"){
 62                 return '<span style="color:green;">' + val + '</span>';
 63             }else if(val == "error"){
 64                 return '<span style="color:red;">' + val + '</span>';
 65             }
 66             return val;
 67         };
 68 	
 69 		var cm = new Ext.grid.ColumnModel([
 70 		    {id:'file',header: "File", width: 200, sortable: false, dataIndex: 'file', resizable: false},
 71 		    {header: "Info", width: 75, sortable: false, dataIndex: 'info', renderer: infoRenderer, resizable: false} 
 72 		]);
 73 				
 74 		var grid = new Ext.grid.GridPanel({
 75 			store:ds,
 76 	        cm: cm,
 77 	        sm: new Ext.grid.RowSelectionModel({ 	singleSelect:true }),
 78 			autoWidth: true
 79 	    });
 80 		
 81 	    var toolbar = new Ext.Toolbar();
 82 		
 83 		var dialog = new Ext.Window({ 
 84 			autoCreate: true, 
 85 			title: ORYX.I18N.TransformationDownloadDialog.transResult, 
 86 			autoHeight: true, 
 87 			width: 297, 
 88 			modal:true,
 89 			collapsible:false,
 90 			fixedcenter: true, 
 91 			shadow:true, 
 92 			proxyDrag: true,
 93 			resizable:false,
 94 			items:[toolbar, grid]
 95 		});
 96 	
 97 		dialog.on('hide', function(){
 98 			dialog.destroy(true);
 99 			grid.destroy(true);
100 			delete dialog;
101 			delete grid;
102 		});
103 		dialog.show();
104 		
105 		toolbar.add({
106 			icon: 'images/view.png', // icons can also be specified inline
107 	        cls: 'x-btn-icon',
108     	    tooltip: ORYX.I18N.TransformationDownloadDialog.showFile,
109 			handler: function() {
110 				var ds = grid.getStore();
111 				var selection = grid.getSelectionModel().getSelected();
112 				if (selection == undefined) {
113 					return;
114 				}
115 				var show = selection.get("result");
116 				if (selection.get("info") == "success") {
117 					this.openXMLWindow(show);
118 				} else {
119 					this.openErrorWindow(show);
120 				}
121 			}.bind(this)
122 		});
123 		toolbar.add({
124 			icon: 'images/disk.png', // icons can also be specified inline
125 	        cls: 'x-btn-icon',
126     	    tooltip: ORYX.I18N.TransformationDownloadDialog.downloadFile,
127 			handler: function() {
128 				var ds = grid.getStore();
129 				var selection = grid.getSelectionModel().getSelected();
130 				if (selection == undefined) {
131 					return;
132 				}
133 				this.openDownloadWindow(selection, false);
134 			}.bind(this)
135 		});
136 		toolbar.add({
137 			icon: 'images/disk_multi.png', // icons can also be specified inline
138 	        cls: 'x-btn-icon',
139     	    tooltip: ORYX.I18N.TransformationDownloadDialog.downloadAll,
140 			handler: function() {
141 				var ds = grid.getStore();				
142 				this.openDownloadWindow(ds.getRange(0, ds.getCount()), true);
143 			}.bind(this)
144 		});			
145 
146 		// Select the first row
147 		grid.getSelectionModel().selectFirstRow();
148 
149 	},
150 	
151 	/**
152 	 * Opens a download window for downloading the given content.
153 	 * 
154 	 * Creates a submit form to send the contents to the 
155 	 * Oryx Legacy File Download Servlet (MultiDownloader).
156 	 * 
157 	 * @param {Object} content The content to be downloaded. If it is a zip 
158 	 *                         file, then this should be an array of contents.
159 	 * @param {Object} zip     True, if it is a zip file, false otherwise
160 	 */
161 	openDownloadWindow: function(content, zip) {
162 		var win = window.open("");
163 		if (win != null) {
164 			win.document.open();
165 			win.document.write("<html><body>");
166 			var submitForm = win.document.createElement("form");
167 			win.document.body.appendChild(submitForm);
168 			
169 			try {
170 				if (zip) {
171 					for (var i = 0; i < content.length; i++) {
172 						var file = this.addFileExtension(content[i].get("file"));
173 						if(file.include("-wsdl")){
174 							file = file.replace("-wsdl","");
175 						}	
176 						submitForm.appendChild( this.createHiddenElement("download_" + i, content[i].get("result")));
177 						submitForm.appendChild( this.createHiddenElement("file_" + i, file));
178 					}
179 				} else {
180 					var file = this.addFileExtension(content.get("file"));
181 					if(file.include("-wsdl")){
182 						file = file.replace("-wsdl","");
183 					}
184 					submitForm.appendChild( this.createHiddenElement("download", content.get("result")));
185 					submitForm.appendChild( this.createHiddenElement("file", file));
186 				}
187 			} catch (error){
188 				this.facade.raiseEvent({type:ORYX.CONFIG.EVENT_LOADING_DISABLE});
189 				Ext.Msg.alert(ORYX.I18N.Oryx.title, error);
190 			}
191 			
192 			submitForm.method = "POST";
193 			win.document.write("</body></html>");
194 			win.document.close();
195 			submitForm.action= "download";
196 			submitForm.submit();
197 		}		
198 	},
199 	
200 	/**
201 	 * Adds a file extension to the given file name. If the file
202 	 * include the name "wsdl", an .wsdl extension will
203 	 * be added. Otherwise a .bpel extension will be added
204 	 * 
205 	 * @param {Object} file The file name to add the extension to.
206 	 */
207 	addFileExtension: function(file) {
208 		if (file.include("wsdl")) {
209 			return file + ".wsdl";
210 		} else {
211 			return file + ".bpel";
212 		}
213 	},
214 	
215 	/**
216 	 * Determines if the result is an XML file or not.
217 	 * For this purpose it is determined if the given
218 	 * result starts with "<?xml".
219 	 * 
220 	 * @param {Object} result The result to be checked.
221 	 * @return "success" if it is an XML file, "error" otherwise
222 	 */
223 	getResultInfo: function(result) {
224 		if (!result) {
225 			return "error";
226 		} else if (result.substr(0, 5) == "<?xml") {
227 			return "success";
228 		}
229 		
230 		return "error";
231 	},
232 	
233 	/**
234 	 * Determines the process name for a given bpel
235 	 * string. 
236 	 * 
237 	 * @param {String} process The BPEL process.
238 	 */
239 	getBPELName: function(bpel) {
240 		var parser	= new DOMParser();
241 		var doc		= parser.parseFromString(bpel,"text/xml");
242 		var name 	= doc.documentElement.getAttribute("name");
243 		return name;
244 	}
245 }
