 27 /**
 28  * Supports creating and editing ad-hoc completion conditions

 33 	facade: undefined,
 34 	UNSAVED_RESOURCE: "unsaved",
 35 
 36 	/**
 37 	 * Offers the plugin functionality:
 38 	 * 
 39 	 */
 40 	construct: function(facade) {
 41 		this.facade = facade;
 42 		
 43 		this.facade.offer({
 44 			'name':ORYX.I18N.AdHocCC.compl,
 45 			'functionality': this.editCC.bind(this),
 46 			'group': ORYX.I18N.AdHocCC.group,
 47 			'icon': ORYX.PATH + "images/adhoc.gif",
 48 			'description': ORYX.I18N.AdHocCC.complDesc,
 49 			'index': 0,
 50 			'minShape': 1,
 51 			'maxShape': 1
 52 			// ISSUE: Should the Context Area this Plugin is creating be removed?
 53 		});
 54 	},

 57 	/**
 58 	 * Opens a Dialog that can be used to edit an ad-hoc activity's completion condition
 59 	 * 
 60 	 */
 61 	editCC: function(){	
 62 	
 63 		/*
 64 		 * 	check pre conditions
 65 		 */ 
 66 		var elements = this.facade.getSelection();
 67 		if (elements.length != 1) {
 68 			// Should not happen!
 69 			this.openErroDialog(ORYX.I18N.AdHocCC.notOne);
 70 			return ; 
 71 		}
 72 		var adHocActivity = elements[0];
 73 		if (adHocActivity._stencil.id() != "http://b3mn.org/stencilset/bpmnexec#Subprocess" || !adHocActivity.properties['oryx-isadhoc']){
 74 			this.openErroDialog(ORYX.I18N.AdHocCC.nodAdHocCC); 
 75 			return ;
 76 		}
 77 	
 78 		/*
 79 		 * 	load relevant data
 80 		 */ 	
 81 		var oldCC = adHocActivity.properties['oryx-adhoccompletioncondition'];
 82 		var taskArrayFields = ['resourceID', 'resourceName'];
 83 		var taskArray = []; 
 84 		var stateArrayFields = ['state'];
 85 		var stateArray = [ ['ready'], ['skipped'], ['completed'] ];
 86 		var dataArrayFields = ['resourceID_FieldName', 'dataNameAndFieldName'];
 87 		var dataArray = [];
 88 
 89 		var parser = new DOMParser();
 90 		
 91 		var childNodes = adHocActivity.getChildNodes();
 92 		for (var i = 0; i < childNodes.length; i++) {
 93 			var child = childNodes[i];
 94 			if (child._stencil.id() == "http://b3mn.org/stencilset/bpmnexec#Task") {
 95 				var resourceName = child.properties['oryx-name'];
 96 				var resourceID = child.resourceId;
 97 				if (typeof resourceID == "undefined") {
 98 					DataManager.__persistDOM(this.facade);
 99 					resourceID = child.resourceId;
100 					if (typeof resourceID == "undefined") {
101 						resourceID = this.UNSAVED_RESOURCE;
102 						resourceName = resourceName + " (unsaved)";
103 					}
104 				}
105 				taskArray.push([resourceID, resourceName]);
106 			}
107 			else if (child._stencil.id() == "http://b3mn.org/stencilset/bpmnexec#DataObject") {
108 				var resourceName = child.properties['oryx-name'];
109 				var resourceID = child.resourceId;
110 				if (typeof resourceID == "undefined") {
111 					DataManager.__persistDOM(this.facade);
112 					resourceID = child.resourceId;
113 					if (typeof resourceID == "undefined") {
114 						resourceID = this.UNSAVED_RESOURCE;
115 						resourceName = resourceName + " (unsaved)";
116 					}
117 				}
118 				var dataModelString = child.properties['oryx-datamodel'];
119 				var dataModel = parser.parseFromString(dataModelString,"text/xml");
120 				var rootXMLNode = dataModel.childNodes[0];
121 				if (rootXMLNode != null){
122 					var childXMLNodes = rootXMLNode.childNodes;
123 					for (var j = 0; j < childXMLNodes.length; j++) {
124 						var dataFieldTagName = childXMLNodes[j].attributes['name'].nodeValue;
125 						if (dataFieldTagName != null) {
126 							dataArray.push([[resourceID, dataFieldTagName], resourceName + "/" + dataFieldTagName]);
127 						}
128 					}
129 				}
130 			}
131 		}
132 				
133 		/*
134 		 * 	initialiaze UI
135 		 */ 
136 		var taskStore = new Ext.data.SimpleStore({
137    			fields: taskArrayFields,
138     		data : taskArray
139 		});
140 		
141 		var stateStore = new Ext.data.SimpleStore({
142    			fields: stateArrayFields,
143     		data : stateArray
144 		});
145 		
146 		var dataStore = new Ext.data.SimpleStore({
147    			fields: dataArrayFields,
148     		data : dataArray
149 		});
150 		
151 		var taskCombo = new Ext.form.ComboBox({
152     		store: taskStore,
153 			valueField: taskArrayFields[0],
154 			displayField: taskArrayFields[1],
155     		emptyText: ORYX.I18N.AdHocCC.selectTask,
156 			typeAhead: true,
157     		mode: 'local',
158     		triggerAction: 'all',
159    			selectOnFocus: true,
160 			editable: false,
161 			width: 180
162 		});
163 		
164 		var stateCombo = new Ext.form.ComboBox({
165     		store: stateStore,
166     		displayField: stateArrayFields[0],
167     		emptyText: ORYX.I18N.AdHocCC.selectState,
168 			typeAhead: true,
169     		mode: 'local',
170     		triggerAction: 'all',
171    			selectOnFocus: true,
172 			editable: false,
173 			width: 180
174 		});
175 
176 		var addStateExprButton = new Ext.Button({
177 			text: ORYX.I18N.AdHocCC.addExp,
178 			handler: function(){
179 				var task = taskCombo.getValue();
180 				var state = stateCombo.getValue();
181 				if (task != this.UNSAVED_RESOURCE && task != "" && state != "") {
182 					this.addStringToTextArea(textArea, "stateExpression('"+task+"', '"+state+"')");
183 					taskCombo.setValue("");
184 					stateCombo.setValue("");
185 				}
186 			}.bind(this)
187 		});
188 		
189 		var dataCombo = new Ext.form.ComboBox({
190     		store: dataStore,
191 			valueField: dataArrayFields[0],
192     		displayField: dataArrayFields[1],
193     		emptyText: ORYX.I18N.AdHocCC.selectDataField,
194 			typeAhead: true,
195     		mode: 'local',
196     		triggerAction: 'all',
197    			selectOnFocus: true,
198 			editable: false,
199 			width: 180
200 		});
201 		
202 		var valueField = new Ext.form.TextField({
203 			width: 180,
204 			emptyText: ORYX.I18N.AdHocCC.enterEqual,
205 		});
206 		
207 		var addDataExprButton = new Ext.Button({
208 			text: ORYX.I18N.AdHocCC.addExp,
209 			handler: function(){
210 				var data = dataCombo.getValue();
211 				var value = valueField.getValue();
212 				if (data != null && data[0] != this.UNSAVED_RESOURCE && value != "") {
213 					this.addStringToTextArea(textArea, "dataExpression('"+data[0]+"', '"+data[1]+"', '"+value+"')");
214 					dataCombo.setValue("");
215 					valueField.setValue("");
216 				}
217 			}.bind(this)
218 		});
219 		
220 		var addAndButton = new Ext.Button({
221 			text: ORYX.I18N.AdHocCC.and, 
222 			minWidth: 50,
223 			handler: function(){
224 				this.addStringToTextArea(textArea, "&");
225 			}.bind(this)
226 		});
227 					
228 		var addOrButton = new Ext.Button({
229 			text: ORYX.I18N.AdHocCC.or, 
230 			minWidth: 50,
231 			handler: function(){
232 				this.addStringToTextArea(textArea, "|");
233 			}.bind(this)
234 		});
235 		
236 		var addLPButton = new Ext.Button({
237 			text: "(", 
238 			minWidth: 50,
239 			handler: function(){
240 				this.addStringToTextArea(textArea, "(");
241 			}.bind(this)
242 		});
243 					
244 		var addRPButton = new Ext.Button({
245 			text: ")", 
246 			minWidth: 50,
247 			handler: function(){
248 				this.addStringToTextArea(textArea, ")");
249 			}.bind(this)
250 		});
251 		
252 		var addNotButton = new Ext.Button({
253 			text: ORYX.I18N.AdHocCC.not, 
254 			minWidth: 50,
255 			handler: function(){
256 				this.addStringToTextArea(textArea, "!");
257 			}.bind(this)
258 		});
259 		
260 		var textArea = new Ext.form.TextArea({
261 			width: 418,
262 			height: 100,
263 			value: oldCC
264 		});
265 		
266 		var clearButton = new Ext.Button({
267 			text: ORYX.I18N.AdHocCC.clearCC,
268 			handler: function(){
269 				textArea.setValue("");
270 			}
271 		});
272 		
273 		var win = new Ext.Window({ 
274 			width: 450,
275 			//minWidth: 400,
276 			height: 485,
277 			//minHeight: 450,
278 			resizable: false,
279 			minimizable: false,
280 			modal: true,
281 			autoScroll: true,
282 			title: ORYX.I18N.AdHocCC.editCC,
283 			layout: 'table',
284 			defaults: {
285 		        bodyStyle:'padding:3px;background-color:transparent;border-width:0px'
286 		    },
287 			layoutConfig: {
288 		        columns: 7
289 		    },
290 			items: [
291 				{ items: [new Ext.form.Label({text: ORYX.I18N.AdHocCC.addExecState, style: 'font-size:12px;'})], colspan: 7},
292 				{}, {items: [taskCombo], colspan: 6},
293 				{}, {items: [stateCombo], colspan: 4}, {items: [addStateExprButton]}, {},
294 				{colspan: 7},
295 				{ items: [new Ext.form.Label({text: ORYX.I18N.AdHocCC.addDataExp, style: 'font-size:12px;'})], colspan: 7},	
296 				{}, {items: [dataCombo], colspan: 6},
297 				{}, {items: [valueField], colspan: 4}, {items: [addDataExprButton]}, {},
298 				{colspan: 7},
299 				{ items: [new Ext.form.Label({text: ORYX.I18N.AdHocCC.addLogOp, style: 'font-size:12px;'})], colspan: 7},	
300 				{}, {items: [addAndButton]}, {items: [addOrButton]}, {items: [addLPButton]}, {items: [addRPButton]}, {items: [addNotButton]}, {},
301 				{colspan: 7},
302 				{ items: [new Ext.form.Label({text: ORYX.I18N.AdHocCC.curCond, style: 'font-size:12px;'})], colspan: 7},
303 				{}, {items: [textArea], colspan: 5}, {},
304 				{colspan: 5}, {items: [clearButton]}, {}
305 			],
306 			buttons: [
307 				{
308 		        	text: 'Apply',
309 		        	handler: function(){
310 		            	win.hide();
311 						adHocActivity.properties['oryx-adhoccompletioncondition'] = textArea.getValue();
312 						// ISSUE: This might be done more elegant using a refresh-event implemented in the property window plugin
313 						this.facade.setSelection([]);
314 						this.facade.setSelection(elements);
315 		        	}.bind(this)
316 		    	},
317 				{
318 		        	text: 'Cancel',
319 		        	handler: function(){win.hide();}
320 		    	}
321 			],
322 	    	keys: [{
323 	        	key: 27,  // Esc
324 	        	fn: function(){win.hide();}
325 	    	}]
326 		});
327 		win.show();	
328 	},

331 	/**
332 	 * Adds an string into a text area
333 	 * 
334 	 * NOTE: This implementation does only work with Gecko browsers (e.g. Mozilla Firefox)
335 	 * 
336 	 * @param {TextField} textArea
337 	 * @param {String} string
338 	 */
339 	addStringToTextArea: function(textArea, string){
340 		var selectionStart = textArea.getEl().dom.selectionStart;
341 		var selectionEnd = textArea.getEl().dom.selectionEnd;
342 		var currentValue = textArea.getValue();
343 		textArea.setValue(currentValue.substring(0, selectionStart)+string+currentValue.substring(selectionEnd));
344 		textArea.getEl().dom.selectionStart = selectionStart + string.length;
345 		textArea.getEl().dom.selectionEnd = textArea.getEl().dom.selectionStart;
346 	},
347 	
348 	/**
349 	 * Opens an error dialog
350 	 * 
351 	 * @param {String} errorMsg
352 	 */
353 	openErroDialog: function(errorMsg){
354 		Ext.MessageBox.show({
355            title: 'Error',
356            msg: errorMsg,
357            buttons: Ext.MessageBox.OK,
358            icon: Ext.MessageBox.ERROR
359        });
360 	}
