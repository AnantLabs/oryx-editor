105 	/**
106 	 * 
107 	 * 
108 	 * @param {Object} url
109 	 * @param {Object} params
110 	 * @param {Object} successcallback
111 	 */
112 	sendRequest: function( url, params, successcallback, failedcallback ){
113 
114 		var suc = false;
115 
116 		new Ajax.Request(url, {
117             method			: 'POST',
118             asynchronous	: false,
119             parameters		: params,
120 			onSuccess		: function(transport) {
121 				
122 				suc = true;
123 				
124 				if(successcallback){
125 					successcallback( transport.result )	
126 				}
127 				
128 			}.bind(this),
129 			
130 			onFailure		: function(transport) {
131 
132 				if(failedcallback){
133 					
134 					failedcallback();
135 					
136 				} else {
137 					Ext.Msg.alert(ORYX.I18N.Oryx.title, ORYX.I18N.ERDFSupport.impFailed);
138 					ORYX.log.warn("Import ERDF failed: " + transport.responseText);	
139 				}
140 				
141 			}.bind(this)		
142 		});
143 		
144 		
145 		return suc;
146 							
147 	},

150 	loadERDF: function( erdfString, success, failed ){
151 		
152 		var s 	= erdfString;
153 		s 		= s.startsWith('<?xml') ? s : '<?xml version="1.0" encoding="utf-8"?>'+s+'';	
154 						
155 		var parser	= new DOMParser();			
156 		var doc 	=  parser.parseFromString( s ,"text/xml");
157 							
158 		if( doc.firstChild.tagName == "parsererror" ){
159 
160 			Ext.MessageBox.show({
161 					title: 		ORYX.I18N.ERDFSupport.error,
162  					msg: 		ORYX.I18N.ERDFSupport.impFailed2 + doc.firstChild.textContent.escapeHTML(),
163 					buttons: 	Ext.MessageBox.OK,
164 					icon: 		Ext.MessageBox.ERROR
165 				});
166 																
167 			if(failed)
168 				failed();
169 				
170 		} else if( !this.hasStencilSet(doc) ){
171 			
172 			if(failed)
173 				failed();		
174 		
175 		} else {
176 			
177 			this.facade.importERDF( doc );
178 			
179 			if(success)
180 				success();
181 		
182 		}
183 	},
184 
185 	hasStencilSet: function( doc ){
186 		
187 		var getElementsByClassNameFromDiv 	= function(doc, id){ return $A(doc.getElementsByTagName('div')).findAll(function(el){ return $A(el.attributes).any(function(attr){ return attr.nodeName == 'class' && attr.nodeValue == id }) })	}
188 
189 		// Get Canvas Node
190 		var editorNode 		= getElementsByClassNameFromDiv( doc, '-oryx-canvas')[0];
191 		
192 		if( !editorNode ){
193 			this.throwWarning(ORYX.I18N.ERDFSupport.noCanvas);
194 			return false
195 		}
196 		
197 		var stencilSetNode 	= $A(editorNode.getElementsByTagName('a')).find(function(node){ return node.getAttribute('rel') == 'oryx-stencilset'});
198 
199 		if( !stencilSetNode ){
200 			this.throwWarning(ORYX.I18N.ERDFSupport.noSS);
201 			return false
202 		}
203 		
204 		var stencilSetUrl	= stencilSetNode.getAttribute('href').split("/")
205 		stencilSetUrl		= stencilSetUrl[stencilSetUrl.length-2] + "/" + stencilSetUrl[stencilSetUrl.length-1];
206 		
207 //		var isLoaded = this.facade.getStencilSets().values().any(function(ss){ return ss.source().endsWith( stencilSetUrl ) })
208 //		if( !isLoaded ){
209 //			this.throwWarning(ORYX.I18N.ERDFSupport.wrongSS);
210 //			return false
211 //		}
212 				
213 		return true;
214 	},

239 	/**
240 	 * Opens a download window for downloading the given content.
241 	 * 
242 	 */
243 	openDownloadWindow: function(file, content) {
244 		var win = window.open("");
245 		if (win != null) {
246 			win.document.open();
247 			win.document.write("<html><body>");
248 			var submitForm = win.document.createElement("form");
249 			win.document.body.appendChild(submitForm);
250 			
251 			submitForm.appendChild( this.createHiddenElement("download", content));
252 			submitForm.appendChild( this.createHiddenElement("file", file));
253 			
254 			
255 			submitForm.method = "POST";
256 			win.document.write("</body></html>");
257 			win.document.close();
258 			submitForm.action= ORYX.PATH + "/download";
259 			submitForm.submit();
260 		}		
261 	},

263 	/**
264 	 * Creates a hidden form element to communicate parameter values.
265 	 * 
266 	 * @param {Object} name  The name of the hidden field
267 	 * @param {Object} value The value of the hidden field
268 	 */
269 	createHiddenElement: function(name, value) {
270 		var newElement = document.createElement("input");
271 		newElement.name=name;
272 		newElement.type="hidden";
273 		newElement.value = value;
274 		return newElement
275 	},
