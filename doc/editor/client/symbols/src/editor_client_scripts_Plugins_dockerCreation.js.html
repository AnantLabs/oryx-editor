 24 if(!ORYX.Plugins)
 25 	ORYX.Plugins = new Object();
 26 
 27 ORYX.Plugins.DockerCreation = Clazz.extend({
 28 	
 29 	construct: function( facade ){
 30 		this.facade = facade;		
 31 		this.active = false; //true-> a ghostdocker is shown; false->ghostdocker is hidden
 32 		this.point = {x:0, y:0}; //Ghostdocker
 33 		
 34 		//visual representation of the Ghostdocker
 35 		this.circle = ORYX.Editor.graft("http://www.w3.org/2000/svg", null ,
 36 				['g', {"pointer-events":"none"},
 37 					['circle', {cx: "8", cy: "8", r: "3", fill:"yellow"}]]); 	
 38 		
 39 		//Event registrations
 40 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_MOUSEDOWN, this.handleMouseDown.bind(this));
 41 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_MOUSEOVER, this.handleMouseOver.bind(this));
 42 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_MOUSEOUT, this.handleMouseOut.bind(this));
 43 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_MOUSEMOVE, this.handleMouseMove.bind(this));
 44 		/*
 45 		 * Double click is reserved for label access, so abort action
 46 		 */
 47 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_DBLCLICK,function(){window.clearTimeout(this.timer)}.bind(this));
 48 		/*
 49 		 * click is reserved for selecting, so abort action when mouse goes up
 50 		 */
 51 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_MOUSEUP,function(){window.clearTimeout(this.timer)}.bind(this));
 52 
 53 },
 54 
 55 /**
 56  * MouseOut Handler
 57  * 
 58  *hide the Ghostpoint when Leaving the mouse from an edge
 59  */
 60 handleMouseOut: function(event, uiObj) {
 61 	
 62 	if (this.active) {		
 63 		this.hideOverlay();
 64 		this.active = false;
 65 	}	
 66 },
 67 
 68 /**
 69  * MouseOver Handler
 70  * 
 71  *show the Ghostpoint if the edge is selected
 72  */
 73 handleMouseOver: function(event, uiObj) {
 74 	
 75 	this.point.x = this.facade.eventCoordinates(event).x;
 76 	this.point.y = this.facade.eventCoordinates(event).y;
 77 
 78 	//show the Ghostdocker on the edge
 79 	if( uiObj instanceof ORYX.Core.Edge){		
 80 		this.showOverlay( uiObj, this.point )
 81 	}
 82 	//ghostdocker is active
 83 	this.active = true;
 84 	
 85 },
 86 
 87 /**
 88  * MouseDown Handler
 89  * 
 90  *create a Docker when clicking on a selected edge
 91  */
 92 handleMouseDown: function(event, uiObj) {	
 93 	if (event.which==1 && uiObj instanceof ORYX.Core.Edge){
 94 		//Timer for Doubleclick to be able to create a label
 95 		window.clearTimeout(this.timer);
 96 		
 97 		this.timer = window.setTimeout(function () {
 98 			/*
 99 			 * Give the event to enable one click creation and drag
100 			 */
101 			this.addDockerCommand({
102 	            edge: uiObj,
103 				event: event,
104 	            position: this.facade.eventCoordinates(event)
105 	        });
106 
107 		}.bind(this),200);
108 		this.hideOverlay();
109 
110 	}
111 },
112 
113 //
114 /**
115  * MouseMove Handler
116  * 
117  *refresh the ghostpoint when moving the mouse over an edge
118  */
119 handleMouseMove: function(event, uiObj) {		
120 		if (uiObj instanceof ORYX.Core.Edge){
121 			this.point.x = this.facade.eventCoordinates(event).x;
122 			this.point.y = this.facade.eventCoordinates(event).y;
123 
124 			if (this.active) {	
125 				//refresh Ghostpoint
126 				this.hideOverlay();			
127 				this.showOverlay( uiObj, this.point);
128 			}else{
129 				this.showOverlay( uiObj, this.point);	
130 			}		
131 		}	
132 },
133 
134 
135 /**
136  * Command for creating a new Docker
137  * 
138  * @param {Object} options
139  */
140 addDockerCommand: function(options){
141     if(!options.edge)
142         return;
143     
144     var commandClass = ORYX.Core.Command.extend({
145         construct: function(edge, docker, pos, facade, options){            
146             this.edge = edge;
147             this.docker = docker;
148             this.pos = pos;
149             this.facade = facade;
150 			this.options= options;
151         },
152         execute: function(){
153             this.docker = this.edge.addDocker(this.pos, this.docker);
154 			this.index = this.edge.dockers.indexOf(this.docker);                                    
155             this.facade.getCanvas().update();
156             this.facade.updateSelection();
157             this.options.docker=this.docker;
158 
159         },
160         rollback: function(){
161           
162              if (this.docker instanceof ORYX.Core.Controls.Docker) {
163                     this.edge.removeDocker(this.docker);
164              }             
165             this.facade.getCanvas().update();
166             this.facade.updateSelection(); 
167         }
168     });
169     var command = new commandClass(options.edge, options.docker, options.position, this.facade, options);    
170     this.facade.executeCommands([command]);
171 
172     
173 	this.facade.raiseEvent({
174 		uiEvent:	options.event,
175 		type:		ORYX.CONFIG.EVENT_DOCKERDRAG}, options.docker );
176     
177 },
178 
179 /**
180  *show the ghostpoint overlay
181  *
182  *@param {Shape} edge
183  *@param {Point} point
184  */
185 showOverlay: function(edge, point){
186 
187 	this.facade.raiseEvent({
188 			type: 			ORYX.CONFIG.EVENT_OVERLAY_SHOW,
189 			id: 			"ghostpoint",
190 			shapes: 		[edge],
191 			node:			this.circle,
192 			ghostPoint:		point,
193 			dontCloneNode:	true
194 		});			
195 },
196 
197 /**
198  *hide the ghostpoint overlay
199  */
200 hideOverlay: function() {
201 	
202 	this.facade.raiseEvent({
203 		type: ORYX.CONFIG.EVENT_OVERLAY_HIDE,
204 		id: "ghostpoint"
205 	});	
206 }
207 
208 });
