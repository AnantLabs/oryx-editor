  7 if(!ORYX){ var ORYX = {} }
  8 if(!ORYX.Plugins){ ORYX.Plugins = {} }

 10 /**
 11    This abstract plugin implements the core behaviour of layout
 12    
 13    @class ORYX.Plugins.AbstractLayouter
 14    @constructor Creates a new instance
 15    @author Willi Tscheschner
 16 */
 17 ORYX.Plugins.AbstractLayouter = ORYX.Plugins.AbstractPlugin.extend({
 18 	
 19 	/**
 20 	 * 'layouted' defined all types of shapes which will be layouted. 
 21 	 * It can be one value or an array of values. The value
 22 	 * can be a Stencil ID (as String) or an class type of either 
 23 	 * a ORYX.Core.Node or ORYX.Core.Edge
 24      * @type Array|String|Object
 25      * @memberOf ORYX.Plugins.AbstractLayouter.prototype
 26 	 */
 27 	layouted : [],
 28 	
 29 	/**
 30 	 * Constructor
 31 	 * @param {Object} facade
 32 	 * @memberOf ORYX.Plugins.AbstractLayouter.prototype
 33 	 */
 34 	construct: function( facade ){
 35 		arguments.callee.$.construct.apply(this, arguments);
 36 			
 37 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_LAYOUT, this._initLayout.bind(this));
 38 	},
 39 	
 40 	/**
 41 	 * Proofs if this shape should be layouted or not
 42 	 * @param {Object} shape
 43      * @memberOf ORYX.Plugins.AbstractLayouter.prototype
 44 	 */
 45 	isIncludedInLayout: function(shape){
 46 		if (!(this.layouted instanceof Array)){
 47 			this.layouted = [this.layouted].compact();
 48 		}
 49 		
 50 		// If there are no elements
 51 		if (this.layouted.length <= 0) {
 52 			// Return TRUE
 53 			return true;
 54 		}
 55 		
 56 		// Return TRUE if there is any correlation between 
 57 		// the 'layouted' attribute and the shape themselve.
 58 		return this.layouted.any(function(s){
 59 			if (typeof s == "string") {
 60 				return shape.getStencil().id().include(s);
 61 			} else {
 62 				return shape instanceof s;
 63 			}
 64 		})
 65 	},
 66 	
 67 	/**
 68 	 * Callback to start the layouting
 69 	 * @param {Object} event Layout event
 70 	 * @param {Object} shapes Given shapes
 71      * @memberOf ORYX.Plugins.AbstractLayouter.prototype
 72 	 */
 73 	_initLayout: function(event){
 74 		
 75 		// Get the shapes
 76 		var shapes = [event.shapes].flatten().compact();
 77 		
 78 		// Find all shapes which should be layouted
 79 		var toLayout = shapes.findAll(function(shape){
 80 			return this.isIncludedInLayout(shape) 
 81 		}.bind(this))
 82 		
 83 		// If there are shapes left 
 84 		if (toLayout.length > 0){
 85 			// Do layout
 86 			this.layout(toLayout);
 87 		}
 88 	},
 89 	
 90 	/**
 91 	 * Implementation of layouting a set on shapes
 92 	 * @param {Object} shapes Given shapes
 93      * @memberOf ORYX.Plugins.AbstractLayouter.prototype
 94 	 */
 95 	layout: function(shapes){
 96 		throw new Error("Layouter has to implement the layout function.")
 97 	}
 98 });
