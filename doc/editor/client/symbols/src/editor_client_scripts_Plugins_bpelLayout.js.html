 37 	construct: function(facade) {
 38 		this.facade = facade;
 39 		
 40 		this.isEnabled = true;
 41 		
 42 		this.facade.offer({
 43 			'name':ORYX.I18N.BPELSupport.enable,
 44 			'functionality': this.enableBpelLayout.bind(this),
 45 			'group': ORYX.I18N.BPELLayout.group,
 46 			'icon': ORYX.PATH + "images/bpel_layout_enable.png",
 47 			'description': ORYX.I18N.BPELLayout.enDesc,
 48 			'index': 0,
 49 			'minShape': 0,
 50 			'maxShape': 0,
 51 			'isEnabled': function(){ return !(this.isEnabled)}.bind(this)
 52 		});
 53 		
 54 		this.facade.offer({
 55 			'name':ORYX.I18N.BPELSupport.disable,
 56 			'functionality': this.disableBpelLayout.bind(this),
 57 			'group': ORYX.I18N.BPELLayout.group,
 58 			'icon': ORYX.PATH + "images/bpel_layout_disable.png",
 59 			'description': ORYX.I18N.BPELLayout.disDesc,
 60 			'index': 1,
 61 			'minShape': 0,
 62 			'maxShape': 0,
 63 			'isEnabled': function(){ return this.isEnabled}.bind(this)
 64 		});
 65 	
 66 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_LAYOUT_BPEL, this.handleLayoutEvent.bind(this));
 67 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_LAYOUT_BPEL_VERTICAL, this.handleLayoutVerticalEvent.bind(this));
 68 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_LAYOUT_BPEL_HORIZONTAL, this.handleLayoutHorizontalEvent.bind(this));
 69 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_LAYOUT_BPEL_SINGLECHILD, this.handleSingleChildLayoutEvent.bind(this));
 70 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_LAYOUT_BPEL_AUTORESIZE, this.handleAutoResizeLayoutEvent.bind(this));

 80 	enableBpelLayout : function(){
 81 		
 82 		this.isEnabled = true;
 83 		
 84 		this.facade.raiseEvent({type:ORYX.CONFIG.EVENT_LOADING_ENABLE,text: 'Auto Layouting...'});
 85 		
 86 		//adjust all immediate child nodes(grand-children are adjusted recursively)
 87 		nodes = this.facade.getCanvas().getChildNodes();
 88 		for (var i = 0; i < nodes.size(); i++) {
 89 			node = nodes[i];
 90 			if (node.getStencil().id() == node.getStencil().namespace() + "process"){
 91 				this._adjust_node(node);
 92 			}
 93 		}
 94 		
 95 		this.facade.raiseEvent({type:ORYX.CONFIG.EVENT_LOADING_DISABLE});
 96 	},

 98 	_adjust_node : function (node){
 99 		
100 		// handle children first
101 		// that means, the innermost children should be at first arranged,
102 		var nodes = node.getChildNodes();
103 		for (var i = 0; i < nodes.size(); i++) {
104 			this._adjust_node(nodes[i]);
105 		};
106 		
107 		// handle the current node
108 		this._handleLayoutEventAdapter (node);
109 		//alert (node.getStencil().id());
110

113 	_handleLayoutEventAdapter : function(node){
114 		
115 		if (node.getStencil().id() == node.getStencil().namespace() + "process"
116 	 		|| node.getStencil().id() == node.getStencil().namespace() + "invoke" 
117 	    	|| node.getStencil().id() == node.getStencil().namespace() + "scope"){
118 	    		
119 			this._handleLayoutEvent (node);
120 			
121 		} else if (node.getStencil().id() == node.getStencil().namespace() + "assign"
122 			|| node.getStencil().id() == node.getStencil().namespace() + "eventHandlers"
123 			|| node.getStencil().id() == node.getStencil().namespace() + "faultHandlers"
124 			|| node.getStencil().id() == node.getStencil().namespace() + "compensationHandler"
125 			|| node.getStencil().id() == node.getStencil().namespace() + "terminationHandler"){
126 			
127 			this._handleLayoutVerticalEvent (node);
128 			
129 		} else if  (node.getStencil().id() == node.getStencil().namespace() + "if"
130 			|| node.getStencil().id() == node.getStencil().namespace() + "sequence"
131 			|| node.getStencil().id() == node.getStencil().namespace() + "pick"){
132 			
133 			this._handleLayoutHorizontalEvent (node);
134 			
135 		} else if (node.getStencil().id() == node.getStencil().namespace() + "onMessage"
136 			|| node.getStencil().id() == node.getStencil().namespace() + "if_branch"
137 			|| node.getStencil().id() == node.getStencil().namespace() + "else_branch"
138 			|| node.getStencil().id() == node.getStencil().namespace() + "while"
139 			|| node.getStencil().id() == node.getStencil().namespace() + "repeatUntil"
140 			|| node.getStencil().id() == node.getStencil().namespace() + "forEach"
141 			|| node.getStencil().id() == node.getStencil().namespace() + "onAlarm"
142 			|| node.getStencil().id() == node.getStencil().namespace() + "onEvent"
143 			|| node.getStencil().id() == node.getStencil().namespace() + "catch"
144 			|| node.getStencil().id() == node.getStencil().namespace() + "catchAll"){
145 			
146 			this._handleSingleChildLayoutEvent (node);
147 			
148 		} else if (node.getStencil().id() == node.getStencil().namespace() + "flow"){
149 			
150 			this._handleAutoResizeLayoutEvent (node);
151 		} else {
152 			// other shapes cannot contain any children shapes.
153 			return;
154 		}

184 	/**
185 	 *  realize special BPEL layouting:
186 	 *  main activity: placed left,
187 	 *  Handler: placed right.
188 	 */
189 	_handleLayoutEvent: function(shape) {
190 		
191 		if (this.isEnabled == false) {
192 			return;
193 		}
194 		
195      	var elements = shape.getChildShapes(false);
196      	
197      	// if Autolayout is not required, do nothing.
198 		if (!this._requiredAutoLayout (shape)){
199      		return;
200      	};
201      	
202 		// If there are no elements
203 		if(!elements || elements.length == 0) {
204 			this._resetBounds(shape);
205 			this._update(shape);
206 			return;
207 		};
208 		
209      	var eventHandlers = elements.find(function(node) {
210 				return (node.getStencil().id() == node.getStencil().namespace() + "eventHandlers");
211 			});
212 		
213 		var faultHandlers = elements.find(function(node) {
214 				return (node.getStencil().id() == node.getStencil().namespace() + "faultHandlers");
215 			});
216 			
217 		var compensationHandler = elements.find(function(node) {
218 				return (node.getStencil().id() == node.getStencil().namespace() + "compensationHandler");
219 			});	
220 
221 		var terminationHandler = elements.find(function(node) {
222 				return (node.getStencil().id() == node.getStencil().namespace() + "terminationHandler");
223 			});
224 			
225 		var otherElements = elements.findAll(function(node){
226 				return (node !== eventHandlers && node !== faultHandlers 
227 				&& node !== compensationHandler && node !== terminationHandler)
228 			});
229 		
230 		var nextLeftBound = 30;
231 		var nextUpperBound = 30;
232 		
233 		// handle Activity
234 		if (otherElements){
235 			
236 			// Sort top-down
237 			otherElements = otherElements.sortBy(function(element){
238 				return element.bounds.upperLeft().y;
239 			});
240 			
241 			// move some certain elements to the last child position
242 			// if it "true" returns, that means, the arrangement of elements
243 			// is changed, we should sort all elements again
244 			if (this._moveSomeElementToLastPosition(otherElements)){
245 				// Sort again
246 				otherElements = otherElements.sortBy(function(element){
247 					return element.bounds.upperLeft().y;
248 				});
249 			}
250 			
251 			var lastUpperYPosition = 0;
252 			var elementWidth;
253 			var maxElementWidth = 0;
254 			
255 			// Arrange shapes like Layout-Vertical
256 			otherElements.each (function(element){
257 		
258 				var ul = element.bounds.upperLeft();
259 				var oldUlY = ul.y;
260 			
261 				ul.y = lastUpperYPosition + 30;
262 				lastUpperYPosition = ul.y + element.bounds.height();
263 			
264 				if (ul.y != oldUlY) {
265 					element.bounds.moveTo(30, ul.y);
266 				};
267 				
268 				elementWidth = element.bounds.width();
269 				if (elementWidth > maxElementWidth){
270 					maxElementWidth = elementWidth;
271 				}
272 			});
273 			
274 			nextLeftBound = 30 + maxElementWidth + 30;
275 		
276 		}
277 		
278 		var width;
279 		var maxWidth = 0;
280 		
281 		// handle EventHanlders
282 		if (eventHandlers){
283 			eventHandlers.bounds.moveTo(nextLeftBound, nextUpperBound);
284 			nextUpperBound = eventHandlers.bounds.lowerRight().y + 10;
285 			
286 			// record maximal width
287 			width = this._getRightestBoundOfAllChildren(eventHandlers)+ 30;
288 			if (width > maxWidth){
289 				maxWidth = width;
290 			}
291 		}
292 		// handle FaultHandlers
293 		if (faultHandlers){
294 			faultHandlers.bounds.moveTo(nextLeftBound, nextUpperBound);
295 			nextUpperBound = faultHandlers.bounds.lowerRight().y + 10;
296 			
297 			// record maximal width
298 			width = this._getRightestBoundOfAllChildren(faultHandlers)+ 30;
299 			if (width > maxWidth){
300 				maxWidth = width;
301 			}
302 		}
303 		// handle CompensationHandler
304 		if (compensationHandler){
305 			compensationHandler.bounds.moveTo(nextLeftBound, nextUpperBound);
306 			nextUpperBound = compensationHandler.bounds.lowerRight().y + 10;
307 			
308 			// record maximal width
309 			width = this._getRightestBoundOfAllChildren(compensationHandler)+ 30;
310 			if (width > maxWidth){
311 				maxWidth = width;
312 			}
313 		}
314 		
315 		// handle TerminationHandler
316      	if (terminationHandler){
317 			terminationHandler.bounds.moveTo(nextLeftBound, nextUpperBound);
318 			
319 			// record maximal width
320 			width = this._getRightestBoundOfAllChildren(terminationHandler)+ 30;
321 			if (width > maxWidth){
322 				maxWidth = width;
323 			}
324 		}
325 		
326 		// resize all the handlers with the same width
327 		if (width > 0){
328 			var ul;
329 			var lr;
330 			
331 			if (eventHandlers){	
332 				width = eventHandlers.bounds.width();
333 				if (width !== maxWidth){
334 					ul = eventHandlers.bounds.upperLeft();
335 					lr = eventHandlers.bounds.lowerRight();
336 					eventHandlers.bounds.set(ul.x, ul.y, ul.x + maxWidth, lr.y);
337 					//eventHandlers._changed();
338 				}
339 			}
340 
341 			if (faultHandlers){
342 				width = faultHandlers.bounds.width();
343 				if (width !== maxWidth){
344 					ul = faultHandlers.bounds.upperLeft();
345 					lr = faultHandlers.bounds.lowerRight();
346 					faultHandlers.bounds.set(ul.x, ul.y, ul.x + maxWidth, lr.y);
347 					//faultHandlers._changed();
348 				}
349 			}
350 
351 			if (compensationHandler){
352 				width = compensationHandler.bounds.width();
353 				if (width !== maxWidth){
354 					ul = compensationHandler.bounds.upperLeft();
355 					lr = compensationHandler.bounds.lowerRight();
356 					compensationHandler.bounds.set(ul.x, ul.y, ul.x + maxWidth, lr.y);
357 					//compensationHandler._changed();
358 				}
359 			}
360 			
361 	     	if (terminationHandler){
362 				width = terminationHandler.bounds.width();
363 				if (width !== maxWidth){
364 					ul = terminationHandler.bounds.upperLeft();
365 					lr = terminationHandler.bounds.lowerRight();
366 					terminationHandler.bounds.set(ul.x, ul.y, ul.x + maxWidth, lr.y);
367 					//terminationHandler._changed();
368 				}
369 			}
370 		}
371 		
372 		this._autoResizeLayout(shape);
373 		
374 		this._update(shape);
375 		
376 		return;
377 		
378 	},
379 	
380 	_getRightestBoundOfAllChildren : function(shape){
381 		var elements = shape.getChildShapes(false);
382      	
383 		// If there are no elements
384 		if(!elements || elements.length == 0) {
385 			// 160 is the default width of hanlders
386 			return 130;
387 		};
388 			
389 		// Sort left-right
390 		elements = elements.sortBy(function(element){
391 			return element.bounds.lowerRight().x;
392 		});
393 		
394 		return elements.last().bounds.lowerRight().x;
395 	},

397 	_handleLayoutVerticalEvent: function(shape) {
398 		
399 		if (this.isEnabled == false) {
400 			return;
401 		}
402 		
403 		var elements = shape.getChildShapes(false);
404 		
405 		// if Autolayout is not required, do nothing.
406 		if (!this._requiredAutoLayout (shape)){
407      		return;
408      	};
409 		
410 		// If there are no elements
411 		if(!elements || elements.length == 0) {
412 			this._resetBounds(shape);
413 			return;
414 		};
415 		
416 		// Sort top-down
417 		elements = elements.sortBy(function(element){
418 			return element.bounds.upperLeft().y;
419 		});
420 		
421 					
422 		// move some certain elements to the last child position
423 		// if it "true" returns, that means, the arrangement of elements
424 		// is changed, we should sort all elements again
425 		if (this._moveSomeElementToLastPosition(elements)){
426 			// Sort again
427 			elements = elements.sortBy(function(element){
428 				return element.bounds.upperLeft().y;
429 			});
430 		}
431 		
432 		var lastUpperYPosition = 0;
433 		// Arrange shapes
434 		elements.each(function(element){
435 		
436 			var ul = element.bounds.upperLeft();
437 			var oldUlY = ul.y;
438 			
439 			ul.y = lastUpperYPosition + 30;
440 			lastUpperYPosition = ul.y + element.bounds.height();
441 			
442 			if ((ul.y != oldUlY)) {
443 				element.bounds.moveTo(30, ul.y);
444 			}
445 		});
446 		
447 		this._autoResizeLayout(shape);
448 	
449 		return;
450 	},

452 	_handleLayoutHorizontalEvent: function(shape) {
453 
454 		if (this.isEnabled == false) {
455 			return;
456 		}
457 		
458 		var elements = shape.getChildShapes(false);
459 		
460 		// if Autolayout is not required, do nothing.
461 		if (!this._requiredAutoLayout (shape)){
462      		return;
463      	};
464 		
465 		// If there are no elements
466 		if(!elements || elements.length == 0) {
467 			this._resetBounds(shape);
468 			return;
469 		};
470 					
471 		
472 		// Sort left-right
473 		elements = elements.sortBy(function(element){
474 			return element.bounds.upperLeft().x;
475 		});
476 		
477 		// move some certain elements to the last child position
478 		// if it "true" returns, that means, the arrangement of elements
479 		// is changed, we should sort all elements again
480 		if (this._moveSomeElementToLastPosition(elements)){
481 			// Sort again
482 			elements = elements.sortBy(function(element){
483 				return element.bounds.upperLeft().x;
484 			});
485 		}
486 		
487 		var lastLeftXPosition = 0;
488 		
489 		// Arrange shapes on rows (align left)
490 		elements.each(function(element){
491 		
492 			var ul = element.bounds.upperLeft();
493 			var oldUlX = ul.x;
494 			
495 			ul.x = lastLeftXPosition + 30;
496 			lastLeftXPosition = ul.x + element.bounds.width();
497 
498 			if ((ul.x != oldUlX)) {
499 				element.bounds.moveTo(ul.x, 30);
500 			}
501 		});
502 		
503 		this._autoResizeLayout(shape);
504 			
505 		return;
506 	},

511      	
512 		if (this.isEnabled == false) {
513 			return;
514 		}
515 		
516 		var elements = shape.getChildShapes(false);
517 		
518 		// if Autolayout is not required, do nothing.
519 		if (!this._requiredAutoLayout (shape)){
520      		return;
521      	};
522 		
523 		// If there are no elements
524 		if(!elements || elements.length == 0) {
525 			this._resetBounds(shape);
526 			return;
527 		};
528 		
529 		elements.first().bounds.moveTo(30, 30);
530 		
531 		this._autoResizeLayout(shape);

536 	_handleAutoResizeLayoutEvent: function(shape) {
537 		
538 		if (this.isEnabled == false) {
539 			return;
540 		};
541 		
542 		var elements = shape.getChildShapes(false);
543 		
544 		// if Autolayout is not required, do nothing.
545 		if (!this._requiredAutoLayout (shape)){
546      		return;
547      	};
548 		
549 		elements.each(function(element){
550 		
551 			var ul = element.bounds.upperLeft();
552 			
553 			if ((ul.x < 30)) {
554 				element.bounds.moveTo(30, ul.y);
555 				ul = element.bounds.upperLeft();
556 			}
557 			
558 			if ((ul.y < 30)) {
559 				element.bounds.moveTo(ul.x, 30);
560 			}
561 		});
562 		
563 		this._autoResizeLayout(shape);
564 	},

566 	/**
567 	 * Resizes the shape to the bounds of the child shapes 
568 	 */
569 	_autoResizeLayout: function(shape) {
570 		
571 		var elements = shape.getChildShapes(false);
572 		
573 		if (elements.length > 0) {
574 
575 		    elements = elements.sortBy(function(element){
576 				return element.bounds.lowerRight().x;
577 		    });
578 		    
579 			var rightBound = elements.last().bounds.lowerRight().x;
580                  
581 		    elements = elements.sortBy(function(element){
582 				return element.bounds.lowerRight().y;
583 		    });
584 		    
585 			var lowerBound = elements.last().bounds.lowerRight().y;
586 			
587 			var ul = shape.bounds.upperLeft();
588 			var lr = shape.bounds.lowerRight();
589 			
590 			//handle "flow" specially.
591 			if (shape.getStencil().id() ==shape.getStencil().namespace() + "flow"){
592 			 	
593 			 	if (lr.x < ul.x + rightBound + 30){
594 			 		shape.bounds.set(ul.x, ul.y, ul.x + rightBound + 30, lr.y);
595 			 		lr.x = ul.x + rightBound + 30;
596 			 		//shape._changed();
597 			 	};
598 			 	
599 			 	if (lr.y < ul.y + lowerBound + 30){
600 			 		shape.bounds.set(ul.x, ul.y, lr.x, ul.y + lowerBound + 30);
601 			 		//shape._changed();
602 			 	};			 	
603 			 } else {
604 			 	if (lr.x != ul.x + rightBound + 30 || lr.y != ul.y + lowerBound + 30){
605 			 		shape.bounds.set(ul.x, ul.y, ul.x + rightBound + 30, ul.y + lowerBound + 30);
606 			 		//shape._changed();
607 			 	};
608 			 };
609 		};
610 		
611 		return;
612 	},
613 	
614 	_resetBounds: function (shape) {
615 
616 		// all the shapes without children will be reseted
617 		var ul = shape.bounds.upperLeft();
618 		var lr = shape.bounds.lowerRight();
619 		
620 		if (shape.getStencil().id() == shape.getStencil().namespace() + "process"){
621 			if (shape.getStencil().namespace() == "http://b3mn.org/stencilset/bpel#"){
622 				if (lr.x != ul.x + 600 || lr.y != ul.y + 500){
623 					shape.bounds.set(ul.x, ul.y, ul.x + 600, ul.y + 500);
624 					//shape._changed();
625 				};
626 			} else if (shape.getStencil().namespace() == "http://b3mn.org/stencilset/bpel4chor#"){
627 				if (lr.x != ul.x + 690 || lr.y != ul.y + 200){
628 					shape.bounds.set(ul.x, ul.y, ul.x + 690, ul.y + 200);
629 					//shape._changed();
630 				};
631 			} else {
632 				return;
633 			}
634 		} else if (shape.getStencil().id() == shape.getStencil().namespace() + "flow"){
635 			if (lr.x != ul.x + 290 || lr.y != ul.y + 250){
636 				shape.bounds.set(ul.x, ul.y, ul.x + 290, ul.y + 250);
637 				//shape._changed();
638 			};
639 		} else if (this._isHandlers(shape)){
640 			if (lr.x != ul.x + 160 || lr.y != ul.y + 80){
641 				shape.bounds.set(ul.x, ul.y, ul.x + 160, ul.y + 80);
642 				//shape._changed();
643 			};
644 		} else {
645 			if (lr.x != ul.x + 100 || lr.y != ul.y + 80){

689 	/**
690 	 * find a element with the role "lastChild", that means, this shape should be
691 	 * the last child of their parent, e.g.: "else" in "if-block". then move these elements
692 	 * to the last position of the set.
693 	 * 
694 	 * 
695 	 * @param {} elements : the set of all elements
696 	 * @pre      all the elements in set are already once arranged, so we just put the 
697 	 *           "lastChild" after the current last one. 
698 	 * @return   if the arrangement of elements is changed.
699 	 */
700 	_moveSomeElementToLastPosition: function (elements){
701 		var lastChild = elements.find(function(node) {
702 			 	return (Array.indexOf(node.getStencil().roles(), node.getStencil().namespace() + "lastChild")>= 0);
703 			});	
704 		
705 		// if there are not such element or it's already the last child,
706 		// do nothing.	
707 		if (!lastChild || lastChild == elements.last()){
708 			return false;
709 		}
710 		
711 		// move it after the current last child
712 		ulOfCurrentLastChild = elements.last().bounds.upperLeft();
713 		lastChild.bounds.moveTo(ulOfCurrentLastChild.x + 1, ulOfCurrentLastChild.y + 1);
714 		
715 		return true;
716 	},
