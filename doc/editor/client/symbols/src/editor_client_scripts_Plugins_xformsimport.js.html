 27 ORYX.Plugins.XFormsImport = Clazz.extend({
 28 
 29 	facade: undefined,
 30 
 31 	construct: function(facade) {
 32 		this.facade = facade;
 33 
 34 		this.facade.offer({
 35 			'name':ORYX.I18N.XFormsSerialization.importXForms,
 36 			'functionality': this.importXForms.bind(this),
 37 			'group': ORYX.I18N.XFormsSerialization.group,
 38 			'icon': ORYX.PATH + "images/xforms_import.png",
 39 			'description': ORYX.I18N.XFormsSerialization.importXFormsDesc,
 40 			'index': 3,
 41 			'minShape': 0,
 42 			'maxShape': 0});
 43 	},
 44 
 45 	
 46 	/**
 47 	 * Imports a XForms+XHTML document
 48 	 * 
 49 	 */
 50 	importXForms: function(){
 51 		this._showImportDialog();
 52 	},		
 53 
 54 	
 55 	/**
 56 	 * 
 57 	 * 
 58 	 * @param {Object} url
 59 	 * @param {Object} params
 60 	 * @param {Object} successcallback
 61 	 */
 62 	sendRequest: function( url, params, successcallback, failedcallback ){
 63 
 64 		var suc = false;
 65 
 66 		new Ajax.Request(url, {
 67             method			: 'POST',
 68             asynchronous	: false,
 69             parameters		: params,
 70 			onSuccess		: function(request) {
 71 				
 72 				suc = true;
 73 				
 74 				if(successcallback){
 75 					successcallback( request )	
 76 				}
 77 				
 78 			}.bind(this),
 79 			
 80 			onFailure		: function(request) {
 81 
 82 				if(failedcallback){
 83 					
 84 					failedcallback();
 85 					
 86 				} else {
 87 					Ext.Msg.alert(ORYX.I18N.Oryx.title, ORYX.I18N.XFormsSerialization.impFailed);
 88 					ORYX.log.warn("Import XForms failed: " + transport.responseText);	
 89 				}
 90 				
 91 			}.bind(this)		
 92 		});
 93 		
 94 		
 95 		return suc;
 96 							
 97 	},
 98 
 99 	
100 	throwWarning: function( text ){
101 		Ext.MessageBox.show({
102 					title: 		ORYX.I18N.Oryx.title,
103  					msg: 		text,
104 					buttons: 	Ext.MessageBox.OK,
105 					icon: 		Ext.MessageBox.WARNING
106 				});
107 	},
108 	
109 	
110 	/**
111 	 * Opens an upload dialog.
112 	 * 
113 	 */
114 	_showImportDialog: function( successCallback ){
115 	
116 	    var form = new Ext.form.FormPanel({
117 			baseCls: 		'x-plain',
118 	        labelWidth: 	50,
119 	        defaultType: 	'textfield',
120 	        items: [{
121 	            text : 		ORYX.I18N.XFormsSerialization.selectFile, 
122 				style : 	'font-size:12px;margin-bottom:10px;display:block;',
123 	            anchor:		'100%',
124 				xtype : 	'label' 
125 	        },{
126 	            fieldLabel: ORYX.I18N.XFormsSerialization.file,
127 	            name: 		'subject',
128 				inputType : 'file',
129 				style : 	'margin-bottom:10px;display:block;',
130 				itemCls :	'ext_specific_window_overflow'
131 	        }, {
132 	            xtype: 'textarea',
133 	            hideLabel: true,
134 	            name: 'msg',
135 	            anchor: '100% -63'  
136 	        }]
137 	    });
138 
139 
140 
141 		// Create the panel
142 		var dialog = new Ext.Window({ 
143 			autoCreate: true, 
144 			layout: 	'fit',
145 			plain:		true,
146 			bodyStyle: 	'padding:5px;',
147 			title: 		ORYX.I18N.XFormsSerialization.impTitle, 
148 			height: 	350, 
149 			width:		500,
150 			modal:		true,
151 			fixedcenter:true, 
152 			shadow:		true, 
153 			proxyDrag: 	true,
154 			resizable:	true,
155 			items: 		[form],
156 			buttons:[
157 				{
158 					text:ORYX.I18N.XFormsSerialization.impButton,
159 					handler:function(){
160 						
161 						var loadMask = new Ext.LoadMask(Ext.getBody(), {msg:ORYX.I18N.XFormsSerialization.impProgress});
162 						loadMask.show();
163 						
164 						window.setTimeout(function(){
165 					
166 							var xhtmlString =  form.items.items[2].getValue();
167 							
168 							var params = { resource: location.href, data: xhtmlString };
169 							this.sendRequest(
170 								ORYX.CONFIG.XFORMS_IMPORT_URL, 
171 								params, 
172 								function(request) {
173 									this.facade.importJSON(request.responseText);
174 									loadMask.hide();
175 									dialog.hide();
176 									//this.loadERDF(request.responseText, function(){loadMask.hide();dialog.hide()}.bind(this), function(){loadMask.hide();}.bind(this))
177 								}.bind(this) );
178 														
179 							
180 						}.bind(this), 100);
181 			
182 					}.bind(this)
183 				},{
184 					text:ORYX.I18N.XFormsSerialization.close,
185 					handler:function(){
186 						
187 						dialog.hide();
188 					
189 					}.bind(this)
190 				}
191 			]
192 		});
193 		
194 		// Destroy the panel when hiding
195 		dialog.on('hide', function(){
196 			dialog.destroy(true);
197 			delete dialog;
198 		});
199 
200 
201 		// Show the panel
202 		dialog.show();
203 		
204 				
205 		// Adds the change event handler to 
206 		form.items.items[1].getEl().dom.addEventListener('change',function(evt){
207 				var text = evt.target.files[0].getAsBinary();
208 				form.items.items[2].setValue( text );
209 			}, true)
210 
211 	}
212 	
213 });
