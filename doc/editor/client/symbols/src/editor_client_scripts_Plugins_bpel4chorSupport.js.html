 27 ORYX.Plugins.BPEL4ChorSupport = ORYX.Plugins.AbstractPlugin.extend({
 28 
 29 	/**
 30 	 * Offers the plugin functionality:
 31 	 */
 32     construct: function(){
 33         // Call super class constructor
 34         arguments.callee.$.construct.apply(this, arguments);
 35 
 36 		this.dialogSupport = new ORYX.Plugins.TransformationDownloadDialog();
 37 		
 38 	    this.facade.offer({
 39 			'name':ORYX.I18N.BPEL4ChorSupport.exp,
 40 			'functionality': this.exportProcess.bind(this),
 41 			'group': ORYX.I18N.JSONSupport.exp.group,
 42 			'icon':  ORYX.PATH + "images/bpel4chor_export_icon.png",
 43 			'description': ORYX.I18N.BPEL4ChorSupport.expDesc,
 44 			'index': 0,
 45 			'minShape': 0,
 46 			'maxShape': 0
 47 		});
 48 			
 49         this.facade.offer({
 50 			'name':ORYX.I18N.BPEL4ChorSupport.imp,
 51 			'functionality': this.importProcess.bind(this),
 52 			'group': ORYX.I18N.JSONSupport.imp.group,
 53 			'icon':  ORYX.PATH + "images/bpel4chor_import_icon.png",
 54 			'description': ORYX.I18N.BPEL4ChorSupport.impDesc,
 55 			'index': 1,
 56 			'minShape': 0,
 57 			'maxShape': 0,
 58 			'isEnabled': function(){ return false}
 59 		});
 60 	},
 61 	
 62 	/***************************** export **********************************/	
 63 		
 64 	exportProcess: function(){
 65 	
 66 		// raise loading enable event
 67         this.facade.raiseEvent({
 68             type: ORYX.CONFIG.EVENT_LOADING_ENABLE
 69         });
 70             
 71 		// asynchronously ...
 72         window.setTimeout((function(){
 73 			
 74 			// ... save synchronously
 75             this.exportSynchronously();
 76 			
 77 			// raise loading disable event.
 78             this.facade.raiseEvent({
 79                 type: ORYX.CONFIG.EVENT_LOADING_DISABLE
 80             });
 81 			
 82         }).bind(this), 10);
 83 
 84 		return true;
 85     },
 86     
 87     exportSynchronously: function() {
 88 
 89         var resource = location.href;
 90 		
 91 		
 92 		try {
 93 			var serialized_rdf = this.getRDFFromDOM();
 94 			if (!serialized_rdf.startsWith("<?xml")) {
 95 				serialized_rdf = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>" + serialized_rdf;
 96 			}
 97 			  
 98 			// Send the request to the server.
 99 			new Ajax.Request(ORYX.CONFIG.BPEL4CHOR_EXPORT_URL, {
100 				method: 'POST',
101 				asynchronous: false,
102 				parameters: {
103 					resource: resource,
104 					data: serialized_rdf
105 				},
106                 onSuccess: function(response){
107                 	this.displayResult(response.responseText);
108 				}.bind(this)
109 			});
110                 	
111 			
112 		} catch (error){
113 			this.facade.raiseEvent({type:ORYX.CONFIG.EVENT_LOADING_DISABLE});
114 			Ext.Msg.alert(ORYX.I18N.Oryx.title, error);
115 	 	}
116     
117 	},
118 	
119 	
120 	/**
121 	 * Builds up the data that will be shown in the result dialog of
122 	 * the BPEL4Chor transformation.
123 	 * For this purpose the process names are determined and
124 	 * it is checked if the topology and process were generated
125 	 * successfully.
126 	 * 
127 	 * @param {String}   topology     The generated topology 
128 	 * @param {String}   grounding    The generated grounding 
129 	 * @param {String[]} processes    The generated processes
130 	 */
131 	buildTransData: function(topology, grounding, processes) {
132 		var data = [["topology", topology, this.dialogSupport.getResultInfo(topology)]];
133 		
134 		var counter = 1;
135 		
136 		if (grounding.indexOf("operation")>0){
137 			data[1]= ["grounding", grounding, this.dialogSupport.getResultInfo(grounding)];
138 			counter++;
139 		};
140 		
141 		for (var i = 0; i < processes.length; i++) {
142 			var name = this.dialogSupport.getProcessName(processes[i]);
143 			if (name == undefined) {
144 				name = "Process " + (i+1);
145 			}
146 			data[i+counter] = [name, processes[i], this.dialogSupport.getResultInfo(processes[i])];
147 		}	
148 		
149 		return data;
150 	},
151 	
152 	
153 	/**
154 	 * Analyzes the result of the servlet call.
155 	 * 
156 	 * If an fault occured or the answer is undefined, the error is shown
157 	 * using a message dialog.
158 	 * 
159 	 * If the first result starts with "ParserError" the error is shown using an 
160 	 * error dialog. Otherwise the result is shown using the result dialog.
161 	 * 
162 	 * @param {Object} result - the result of the transformation servlet (JSON)
163 	 */
164 	displayResult: function(result) {
165 		this.facade.raiseEvent({type:ORYX.CONFIG.EVENT_LOADING_DISABLE});
166 
167 		var resultString = '(' + result + ')';
168 		
169 		//alert (resultString);
170 		
171 		var resultObject;
172 		
173 		try {
174 			resultObject = eval(resultString);
175 		} catch (e1) {
176 			alert("Error during evaluation of result: " + e1 + "\r\n" + resultString);
177 		}
178 		
179 		if ((!resultObject.res) || (resultObject.res.length == 0)) {
180 			this.dialogSupport.openMessageDialog(ORYX.I18N.TransformationDownloadDialog.error,ORYX.I18N.TransformationDownloadDialog.noResult);
181 		} else if (resultObject.res[0].content.indexOf("Parser Error")>0) {
182 			this.dialogSupport.openErrorDialog(resultObject.res[0].content);
183 		} else {
184 			var topology = resultObject.res[0].content;
185 			var grounding = resultObject.res[1].content;
186 			var processes = new Array();
187 			for (var i = 2; i < resultObject.res.length; i++) {
188 				processes[i-2] = resultObject.res[i].content;
189 			}
190 			var data = this.buildTransData(topology,grounding,processes);
191 			
192 			this.dialogSupport.openResultDialog(data);
193 		}
194 	},
195

197

199 		this.openUploadDialog ();
200 	},
201

203 	 * Opens a upload dialog.
204 	 */
205 	openUploadDialog: function(){
206 		
207 	},

211 		var parser = new DOMParser();			
212 		var doc    = parser.parseFromString(erdfString ,"text/xml");
213 		
214 		alert(erdfString);
215 		this.facade.importERDF( doc );
216 
217 	}
