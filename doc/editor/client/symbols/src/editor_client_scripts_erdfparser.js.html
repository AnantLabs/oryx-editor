  5 var ERDF = {
  6 
  7 	LITERAL: 0x01,
  8 	RESOURCE: 0x02,
  9 	DELIMITERS: ['.', '-'],
 10 	HASH: '#',
 11 	HYPHEN: "-",

 13 	schemas: [],
 14 	callback: undefined,
 15 	log: undefined,

 17 	init: function(callback) {

 33 	run: function() {
 34 
 35 		//if(ERDF.log.isTraceEnabled())
 36 		//	ERDF.log.trace("ERDF Parser is running.");
 37 
 38 		// do the work.
 39 		return ERDF._checkProfile() && ERDF.parse();
 40 	},

 42 	parse: function() {
 43 		
 44 		//(ERDF.log.isDebugEnabled())
 45 		//	ERDF.log.debug("Begin parsing document metadata.");
 46 		
 47 		// time measuring
 48 		ERDF.__startTime = new Date();
 49 
 50 		var bodies = document.getElementsByTagNameNS(XMLNS.XHTML, 'body');
 51 		var subject = {type: ERDF.RESOURCE, value: ''};
 52 
 53 		var result = ERDF._parseDocumentMetadata() &&
 54 			ERDF._parseFromTag(bodies[0], subject);
 55 			
 56 		// time measuring
 57 		ERDF.__stopTime = new Date();
 58 
 59 		var duration = (ERDF.__stopTime - ERDF.__startTime)/1000.;
 60 		//alert('ERDF parsing took ' + duration + ' s.');
 61 		
 62 		return result;
 63 	},

 65 	_parseDocumentMetadata: function() {
 66 
 67 		// get links from head element.
 68 		var heads = document.getElementsByTagNameNS(XMLNS.XHTML, 'head');
 69 		var links = heads[0].getElementsByTagNameNS(XMLNS.XHTML, 'link');
 70 		var metas = heads[0].getElementsByTagNameNS(XMLNS.XHTML, 'meta');
 71 
 72 		// process links first, since they could contain schema definitions.
 73 		$A(links).each(function(link) {
 74 			var properties = link.getAttribute('rel');
 75 			var reversedProperties = link.getAttribute('rev');
 76 			var value = link.getAttribute('href');
 77 			
 78 			ERDF._parseTriplesFrom(
 79 				ERDF.RESOURCE, '',
 80 				properties,
 81 				ERDF.RESOURCE, value);
 82 				
 83 			ERDF._parseTriplesFrom(
 84 				ERDF.RESOURCE, value,
 85 				reversedProperties,
 86 				ERDF.RESOURCE, '');
 87 		});
 88 
 89 		// continue with metas.
 90 		$A(metas).each(function(meta) {
 91 			var property = meta.getAttribute('name');
 92 			var value = meta.getAttribute('content');
 93 			
 94 			ERDF._parseTriplesFrom(
 95 				ERDF.RESOURCE, '',
 96 				property,
 97 				ERDF.LITERAL, value);
 98 		});
 99 
100 		return true;
101 	},

103 	_parseFromTag: function(node, subject, depth) {
104 		
105 		// avoid parsing non-xhtml content.
106 		if(node.namespaceURI != XMLNS.XHTML) { return; }
107 		
108 		// housekeeping.
109 		if(!depth) depth=0;
110 		var id = node.getAttribute('id');
111 
112 		// some logging.
113 		//if(ERDF.log.isTraceEnabled())
114 		//	ERDF.log.trace(">".times(depth) + " Parsing " + node.nodeName + " ("+node.nodeType+") for data on " +
115 		//		((subject.type == ERDF.RESOURCE) ? ('<' + subject.value + '>') : '') +
116 		//		((subject.type == ERDF.LITERAL) ? '"' + subject.value + '"' : ''));
117 		
118 		/* triple finding! */
119 		
120 		// in a-tags...
121 		if(node.nodeName.endsWith(':a') || node.nodeName == 'a') {
122 			var properties = node.getAttribute('rel');
123 			var reversedProperties = node.getAttribute('rev');
124 			var value = node.getAttribute('href');
125 			var title = node.getAttribute('title');
126 			var content = node.textContent;
127 
128 			// rel triples
129 			ERDF._parseTriplesFrom(
130 				subject.type, subject.value,
131 				properties,
132 				ERDF.RESOURCE, value,
133 				function(triple) {
134 					var label = title? title : content;
135 					
136 					// label triples
137 					ERDF._parseTriplesFrom(
138 						triple.object.type, triple.object.value,
139 						'rdfs.label',
140 						ERDF.LITERAL, label);
141 				});
142 
143 			// rev triples
144 			ERDF._parseTriplesFrom(
145 				subject.type, subject.value,
146 				reversedProperties,
147 				ERDF.RESOURCE, '');
148 				
149 			// type triples
150 			ERDF._parseTypeTriplesFrom(
151 				subject.type, subject.value,
152 				properties);
153 
154 		// in img-tags...
155 		} else if(node.nodeName.endsWith(':img') || node.nodeName == 'img') {
156 			var properties = node.getAttribute('class');
157 			var value = node.getAttribute('src');
158 			var alt = node.getAttribute('alt');
159 
160 			ERDF._parseTriplesFrom(
161 				subject.type, subject.value,
162 				properties,
163 				ERDF.RESOURCE, value,
164 				function(triple) {
165 					var label = alt;
166 					
167 					// label triples
168 					ERDF._parseTriplesFrom(
169 						triple.object.type, triple.object.value,
170 						'rdfs.label',
171 						ERDF.LITERAL, label);
172 				});
173 
174 		}
175 		
176 		// in every tag
177 		var properties = node.getAttribute('class');
178 		var title = node.getAttribute('title');
179 		var content = node.textContent;
180 		var label = title ? title : content;
181 		
182 		// regular triples
183 		ERDF._parseTriplesFrom(
184 			subject.type, subject.value,
185 			properties,
186 			ERDF.LITERAL, label);
187 
188 		if(id) subject = {type: ERDF.RESOURCE, value: ERDF.HASH+id};
189 		
190 		// type triples
191 		ERDF._parseTypeTriplesFrom(
192 			subject.type, subject.value,
193 			properties);
194 
195 		// parse all children that are element nodes.
196 		var children = node.childNodes;
197 		if(children) $A(children).each(function(_node) {
198 			if(_node.nodeType == _node.ELEMENT_NODE)
199 				ERDF._parseFromTag(_node, subject, depth+1); });
200 	},

202 	_parseTriplesFrom: function(subjectType, subject, properties,
203 		objectType, object, callback) {
204 		
205 		if(!properties) return;
206 		properties.toLowerCase().split(' ').each( function(property) {
207 			
208 			//if(ERDF.log.isTraceEnabled())
209 			//	ERDF.log.trace("Going for property " + property);
210 
211 			var schema = ERDF.schemas.find( function(schema) {
212 				return false || ERDF.DELIMITERS.find( function(delimiter) {
213 					return property.startsWith(schema.prefix + delimiter);
214 				});
215 			});
216 			
217 			if(schema && object) {
218 				property = property.substring(
219 					schema.prefix.length+1, property.length);
220 				var triple = ERDF.registerTriple(
221 					new ERDF.Resource(subject),
222 					{prefix: schema.prefix, name: property},
223 					(objectType == ERDF.RESOURCE) ?
224 						new ERDF.Resource(object) :
225 						new ERDF.Literal(object));
226 						
227 				if(callback) callback(triple);
228 			}
229 		});
230 	},
231 	
232 	_parseTypeTriplesFrom: function(subjectType, subject, properties, callback) {
233 		
234 		if(!properties) return;
235 		properties.toLowerCase().split(' ').each( function(property) {
236 			
237 			//if(ERDF.log.isTraceEnabled())
238 			//	ERDF.log.trace("Going for property " + property);
239 				
240 			var schema = ERDF.schemas.find( function(schema) {
241 				return false || ERDF.DELIMITERS.find( function(delimiter) {
242 					return property.startsWith(ERDF.HYPHEN + schema.prefix + delimiter);
243 				});
244 			});
245 			
246 			if(schema && subject) {
247 				property = property.substring(schema.prefix.length+2, property.length);
248 				var triple = ERDF.registerTriple(
249 					(subjectType == ERDF.RESOURCE) ?
250 						new ERDF.Resource(subject) :
251 						new ERDF.Literal(subject),
252 					{prefix: 'rdf', name: 'type'},
253 					new ERDF.Resource(schema.namespace+property));
254 				if(callback) callback(triple);
255 			}
256 		});
257 	},

259 	/**
260 	 * Checks for ERDF profile declaration in head of document.
261 	 */
262 	_checkProfile: function() {
263 
264 		// get profiles from head element.
265 		var heads = document.getElementsByTagNameNS(XMLNS.XHTML, 'head');
266 		var profiles = heads[0].getAttribute("profile");
267 		var found = false;
268 
269 		// if erdf profile is contained.
270 		if(profiles && profiles.split(" ").member(XMLNS.ERDF)) {
271 
272 			// pass check.
273 			//if(ERDF.log.isTraceEnabled())
274 			//	ERDF.log.trace("Found ERDF profile " + XMLNS.ERDF);
275 			return true;
276 			
277 		} else {
278 		
279 			// otherwise fail check.
280 			//if(ERDF.log.isFatalEnabled())
281 			//	ERDF.log.fatal("No ERDF profile found.");
282 			return false;
283 		}
284 	},

286 	__stripHashes: function(s) {
287 		return (s && s.substring(0, 1)=='#') ? s.substring(1, s.length) : s;
288 	},

290 	registerSchema: function(prefix, namespace) {
291 		
292 		// TODO check whether already registered, if so, complain.
293 		ERDF.schemas.push({
294 			prefix: prefix,
295 			namespace: namespace
296 		});
297 		
298 		//if(ERDF.log.isDebugEnabled())
299 		//	ERDF.log.debug("Prefix '"+prefix+"' for '"+namespace+"' registered.");
300 	},

302 	registerTriple: function(subject, predicate, object) {
303 		
304 		// if prefix is schema, this is a schema definition.
305 		if(predicate.prefix.toLowerCase() == 'schema')
306 			this.registerSchema(predicate.name, object.value);
307 			
308 		var triple = new ERDF.Triple(subject, predicate, object);
309 		ERDF.callback(triple);
310 		
311 		//if(ERDF.log.isInfoEnabled())
312 		//	ERDF.log.info(triple)
313 		
314 		// return the registered triple.
315 		return triple;
316 	},

318 	__enhanceObject: function() {
319 		
320 		/* Resource state querying methods */
321 		this.isResource = function() {
322 			return this.type == ERDF.RESOURCE };
323 		this.isLocal = function() {
324 			return this.isResource() && this.value.startsWith('#') };
325 		this.isCurrentDocument = function() {
326 			return this.isResource() && (this.value == '') };
327 		
328 		/* Resource getter methods.*/
329 		this.getId = function() {
330 			return this.isLocal() ? ERDF.__stripHashes(this.value) : false; };
331 
332 		/* Liiteral state querying methods  */
333 		this.isLiteral = function() {
334 			return this.type == ERDF.LIITERAL };
335 	},

337 	serialize: function(literal) {
338 		
339 		if(!literal){
340 			return "";
341 		}else if(literal.constructor == String) {
342 			return literal;
343 		} else if(literal.constructor == Boolean) {
344 			return literal? 'true':'false';
345 		} else {
346 			return literal.toString();
347 		}
348 	}
349 };

352 ERDF.Triple = function(subject, predicate, object) {
353 	
354 	this.subject = subject;
355 	this.predicate = predicate;
356 	this.object = object;
357 	
358 	this.toString = function() {
359 		
360 		return "[ERDF.Triple] " +
361 			this.subject.toString() + ' ' +
362 			this.predicate.prefix + ':' + this.predicate.name + ' ' +
363 			this.object.toString();
364 		};
365 };
366 
367 ERDF.Resource = function(uri) {
368 	
369 	this.type = ERDF.RESOURCE;
370 	this.value = uri;
371 	ERDF.__enhanceObject.apply(this);
372 	
373 	this.toString = function() {
374 		return '<' + this.value + '>';
375 	}
376 	
377 };
378 
379 ERDF.Literal = function(literal) {
380 	
381 	this.type = ERDF.LITERAL;
382 	this.value = ERDF.serialize(literal);
383 	ERDF.__enhanceObject.apply(this);
384 
385 	this.toString = function() {
386 		return '"' + this.value + '"';
387 	}
