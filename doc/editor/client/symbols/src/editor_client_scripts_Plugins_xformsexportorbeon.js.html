 24 if(!ORYX.Plugins)
 25 	ORYX.Plugins = new Object();
 26 
 27 ORYX.Plugins.XFormsExportOrbeon = ORYX.Plugins.AbstractPlugin.extend({
 28 	
 29 	CSS_URL: "/oryx/css/xforms_default.css",
 30 
 31 	facade: undefined,
 32 
 33 	construct: function(facade) {
 34 		this.facade = facade;
 35 
 36 		this.facade.offer({
 37 			'name': "Run XForm with Orbeon",
 38 			'functionality': this.exportIt.bind(this),
 39 			'group': ORYX.I18N.XFormsSerialization.group,
 40 			'icon': ORYX.PATH + "images/xforms_orbeon_export.png",
 41 			'description': 'XForms export for Orbeon',
 42 			'index': 1,
 43 			'minShape': 0,
 44 			'maxShape': 0});
 45 	},
 46 
 47 	exportIt: function(){
 48 
 49 		// raise loading enable event
 50         this.facade.raiseEvent({
 51             type: ORYX.CONFIG.EVENT_LOADING_ENABLE
 52         });
 53 		
 54 		//this.checkClientXFormsSupport();
 55             
 56 		// asynchronously ...
 57         window.setTimeout((function(){
 58 			
 59 			// ... save synchronously
 60             this.exportSynchronously();
 61 			
 62 			// raise loading disable event.
 63             this.facade.raiseEvent({
 64                 type: ORYX.CONFIG.EVENT_LOADING_DISABLE
 65             });
 66 			
 67         }).bind(this), 10);
 68 
 69 		return true;
 70     },
 71 
 72     exportSynchronously: function() {
 73 
 74         var resource = location.href;
 75 		
 76 		
 77 		try {
 78 			var serialized_rdf 	= this.getRDFFromDOM();
 79 			serialized_rdf 		= serialized_rdf.startsWith("<?xml") ? serialized_rdf : "<?xml version=\"1.0\" encoding=\"UTF-8\"?>" + serialized_rdf;
 80 
 81 			// Send the request to the server.
 82 			new Ajax.Request(ORYX.CONFIG.XFORMS_EXPORT_ORBEON_URL, {
 83 				method: 'POST',
 84 				asynchronous: false,
 85 				parameters: {
 86 					resource: resource,
 87 					data: serialized_rdf,
 88 					css: this.CSS_URL
 89 				},
 90 				onSuccess: function(request){
 91 					
 92 						var win = window.open("");
 93 						win.document.write(request.responseText);
 94 						
 95 				},
 96 				onFailure: function(request){
 97 					var win = window.open("");
 98 					win.document.write(request.responseText);
 99 				}
100 			});
101 			
102 		} catch (error){
103 			this.facade.raiseEvent({type:ORYX.CONFIG.EVENT_LOADING_DISABLE});
104 			Ext.Msg.alert(ORYX.I18N.Oryx.title, error);
105 	 	}
106 	},
107 	
108 	checkClientXFormsSupport: function() {
109 		if(!clientSupportsXForms) {
110 			
111 			var output = ORYX.I18N.XFormsSerialization.noClientXFormsSupportDesc;
112 
113 			var win = new Ext.Window({
114                             width: 320,
115                             height: 240,
116                             resizable: false,
117                             minimizable: false,
118                             modal: true,
119                             autoScroll: true,
120                             title: ORYX.I18N.XFormsSerialization.noClientXFormsSupport,
121                             html: output,
122                             buttons: [{
123                                 text: ORYX.I18N.XFormsSerialization.ok,
124                                 handler: function(){
125                                     win.hide();
126                                 }
127                             }]
128                         });
129         	win.show();
130 			
131 		}
132 	}
133 
134 });
