 51     defineSoD: function(){
 52     	//variable declaration
 53     	var selectedElements = this.facade.getSelection(); //all selected elements
 54     	var taskElements = []; //all selected tasks
 55     	var selectedTasks; //new tasks to be separated
 56     	var i = 0; //counter
 57     	var len = selectedElements.length; //number of selected elements
 58     	
 59     	//get only selected tasks
 60     	for (var index = 0; index < len; index++) {
 61     		var item = selectedElements[index];
 62     		if(item.properties["oryx-activitytype"] == "Task") {
 63     			taskElements[i] = item;
 64 				if(taskElements[i].properties["oryx-id"] == "") {
 65 					taskElements[i].setProperty("oryx-id", taskElements[i].id);
 66 				}
 67     			if(i == 0) { //get ids of tasks to be separated
 68 					selectedTasks = taskElements[i].properties["oryx-id"];
 69     			} else {
 70     				selectedTasks = selectedTasks + "; " + taskElements[i].properties["oryx-id"];
 71     			}
 72     			i++;
 73     		}
 74     	}
 75 		
 76     	if(taskElements.length > 1) {
 77 			for (var taskCounter = 0; taskCounter < taskElements.length; taskCounter++) {
 78     		//variable declaration
 79         	var savedSeparationsData = []; //already saved separations
 80         	var savedBindingsData = false; //already saved bindings
 81         	
 82         	//check whether other separations are defined for current task
 83         	if(taskElements[taskCounter].properties["oryx-separationofduties"] != "") {
 84         		savedSeparationsData = this.separationsCheck(taskElements[taskCounter], selectedTasks);
 85         		if (savedSeparationsData[3] == true) {
 86 	        		break;
 87 	        	}
 88         	} else {
 89 				savedSeparationsData[1] = 0; //total number of separation entries is 0 as field is empty
 90 				savedSeparationsData[2] = 0; //highest id used is 0 as no id is used
 91 			}
 92         	
 93         	//check whether a contradicting binding is defined for current task
 94         	if(taskElements[taskCounter].properties["oryx-bindingsofduties"] != ("")) {
 95         		savedBindingsData = this.bindingsCheck(taskElements[taskCounter], selectedTasks);
 96 	        	if (savedBindingsData == true) {
 97 	        		break;
 98 	        	}
 99         	}
100         	
101         	//write entry
102 	    	this.writeEntry(savedSeparationsData, selectedTasks, taskElements[taskCounter]);
103     		}
104 		} else {
105 			alert("Please select at least two tasks to define constraints for.");
106 		}

109     separationsCheck: function(taskElement, selectedTasks) { //check for existing separations with same tasks
110 		var returnValue = [];
111 		//get previous separation of duties assignments
112 		var elementIds = []; //collection of used ids for saved separations
113 		var taskIds = []; //collection of saved task ids
114 		var totalCount = 0; //number of defined separations for a specific task
115     	var highestId = 1; //next element id for separations of a specific task
116     	var j = 0; //counter
117     	var m = 0; //counter
118     	var n = 0; //counter
119 		var separations = taskElement.properties["oryx-separationofduties"]; //get saved separations
120     	separations = separations.toString();
121     	//store saved separations including task ids
122     	var savedSeparations = separations.substring(0,separations.indexOf(']'));
123     	//extract number of entries and element ids of these entries
124     	while(separations.indexOf(',') > -1) {
125     		if(j%2 == 1) { //get element ids as integer
126     			elementIds[m] = parseInt(separations.substring((separations.indexOf('"')) + 1));
127     			m++;
128 		    	separations = separations.substring((separations.indexOf(',')) + 1);
129     		} else if(j == 0) { //get total count as integer
130     			totalCount = parseInt(separations.substring((separations.indexOf(':')) + 1));
131     			separations = separations.substring((separations.indexOf(',')) + 1);
132     		} else { //get task ids of separations
133     			separations = separations.substring((separations.indexOf('"')) + 1);
134     			taskIds[n] = separations.substring(0,separations.indexOf('"'));
135     			n++;
136     			separations = separations.substring((separations.indexOf(',')) + 1);
137     		}
138 	        j++;
139     	}
140     	if(separations.indexOf('"') > -1) {
141     		taskIds[n] = separations.substring((separations.indexOf('"')) + 1);
142     	} else {
143     		taskIds[n] = separations; //get last task ids entry
144     	}
145     	taskIds[n]= taskIds[n].substring(0,taskIds[n].indexOf('"'));
146     	
147         //check whether entry does already exist
148         var separatedTasksCopy = selectedTasks; //working copy
149         var singleTaskIdsInsertedSeparation = []; //task ids of new separations
150         var singleTaskIdsSavedSeparations = []; //task ids of saved separations
151 		
152         var y = 0; //counter
153         
154 		var equal = false; //duplicate existing?
155         
156         //extract task ids of newly entered separation tasks
157     	while(separatedTasksCopy.indexOf(';') > -1) {
158     		singleTaskIdsInsertedSeparation[y] = separatedTasksCopy.substring(0, separatedTasksCopy.indexOf(';'));
159     		separatedTasksCopy = separatedTasksCopy.substring((separatedTasksCopy.indexOf(';'))+2);
160 	        y++;
161         }
162     	singleTaskIdsInsertedSeparation[y] = separatedTasksCopy;
163         
164         //extract task ids of stored separation tasks
165         for(var index = 0; index < taskIds.length; index++) {
166         	var z = 0; //counter
167 			var equalCounter = 0;
168 			while(taskIds[index].indexOf(';') > -1) {
169         		singleTaskIdsSavedSeparations[z] = taskIds[index].substring(0, taskIds[index].indexOf(';'));
170         		taskIds[index] = taskIds[index].substring((taskIds[index].indexOf(';'))+2);
171 		        z++;
172 	        }
173         	singleTaskIdsSavedSeparations[z] = taskIds[index];
174         	z++;
175         	
176 			//compare all collected tasks/ task ids
177         	if(singleTaskIdsInsertedSeparation.length == singleTaskIdsSavedSeparations.length) {
178         		for(var x = 0; x < singleTaskIdsSavedSeparations.length; x++) {
179         			for(var w = 0; w < singleTaskIdsInsertedSeparation.length; w++) {
180         				if(singleTaskIdsInsertedSeparation[w] == singleTaskIdsSavedSeparations[x]) {
181         					equal = true;
182         					break;
183         				}
184         				equal = false;
185         			}
186         			if(equal == false) { //one item of set 1 not found in set 2
187     					break;
188     				}
189         		}
190         	} else if(singleTaskIdsInsertedSeparation.length > singleTaskIdsSavedSeparations.length) {
191         		for(var x = 0; x < singleTaskIdsSavedSeparations.length; x++) {
192         			for(var w = 0; w < singleTaskIdsInsertedSeparation.length; w++) {
193 						if (singleTaskIdsInsertedSeparation[w] == singleTaskIdsSavedSeparations[x]) {
194 							equalCounter++;
195 							break;
196 						}
197 					}
198 					if(equalCounter == singleTaskIdsSavedSeparations.length) { //all saved tasks are found in new entries
199 						var result = this.deleteSeparationEntry(taskElement, (index+1)); //delete existing entry to write new entry later and following updating the entry
200 						savedSeparations = result[0];
201 						totalCount = result[1];
202 						equalCounter = 0;
203 						break;
204 					}
205 				}
206 				equal = false;
207         	} else {
208         		for(var x = 0; x < singleTaskIdsInsertedSeparation.length; x++) {
209         			for(var w = 0; w < singleTaskIdsSavedSeparations.length; w++) {
210         				if(singleTaskIdsSavedSeparations[w] == singleTaskIdsInsertedSeparation[x]) {
211         					equal = true;
212         					break;
213         				}
214         				equal = false;
215         			}
216         			if(equal == false) { //one item of smaller set not found in bigger set
217     					break;
218     				}
219         		}
220         	}
221 			singleTaskIdsSavedSeparations.splice(0,singleTaskIdsSavedSeparations.length);
222         	//entry does exist, that's why no new one needs to be done
223 	        if(equal == true) {
224 	        	break;
225 	        }
226         }
227 		
228     	//get highest element id
229     	for(var index = 0; index < elementIds.length; index++) {
230     		if(elementIds[index] > highestId) {
231     			highestId = elementIds[index];
232     		}
233     	}
234     	returnValue[0] = savedSeparations;
235     	returnValue[1] = totalCount;
236     	returnValue[2] = highestId;
237     	returnValue[3] = equal;
238     	return returnValue;
239     },

241     bindingsCheck: function(taskElement, selectedTasks) { //check for existing bindings with same tasks
242     	var returnValue;
243 		//get previous binding of duties assignments
244 		var boundTaskIds = []; //collection of saved task ids
245     	var l = 0; //counter
246     	var q = 0; //counter
247 		var bindings = taskElement.properties["oryx-bindingsofduties"]; //get saved bindings
248     	bindings = bindings.toString();
249     	//extract number of entries and element ids of these entries
250     	while(bindings.indexOf(',') > -1) {
251     		if(l%2 == 1) { //ignore element ids
252 		    	bindings = bindings.substring((bindings.indexOf(',')) + 1);
253     		} else if(l == 0) { //ignore totalCount
254     			bindings = bindings.substring((bindings.indexOf(',')) + 1);
255     		} else {
256     			bindings = bindings.substring((bindings.indexOf('"')) + 1);
257     			boundTaskIds[q] = bindings.substring(0,bindings.indexOf('"'));
258     			q++;
259     			bindings = bindings.substring((bindings.indexOf(',')) + 1);
260     		}
261 	        l++;
262     	}
263     	if(bindings.indexOf('"') > -1) {
264     		boundTaskIds[q] = bindings.substring((bindings.indexOf('"')) + 1);
265     	} else {
266     		boundTaskIds[q] = bindings; //never reached?
267     	}
268     	boundTaskIds[q]= boundTaskIds[q].substring(0,boundTaskIds[q].indexOf('"'));
269     	
270         //check whether entry does already exist
271         var boundTasksCopy = selectedTasks; //working copy
272         var singleTaskIdsInsertedBinding = []; ////task ids of new bindings
273         var singleTaskIdsSavedBinding = []; //task ids of saved bindings
274         var s = 0; //counter
275         var t = 0; //counter
276         var duplicate = false; //duplicate existing?
277         
278         //extract task ids of newly entered separation tasks
279     	while(boundTasksCopy.indexOf(';') > -1) {
280     		singleTaskIdsInsertedBinding[s] = boundTasksCopy.substring(0, boundTasksCopy.indexOf(';'));
281     		boundTasksCopy = boundTasksCopy.substring((boundTasksCopy.indexOf(';'))+2);
282 	        s++;
283         }
284     	singleTaskIdsInsertedBinding[s] = boundTasksCopy;
285         
286         //extract task ids of stored bound tasks
287         for(var index = 0; index < boundTaskIds.length; index++) {
288         	while(boundTaskIds[index].indexOf(';') > -1) {
289         		singleTaskIdsSavedBinding[t] = boundTaskIds[index].substring(0, boundTaskIds[index].indexOf(';'));
290         		boundTaskIds[index] = boundTaskIds[index].substring((boundTaskIds[index].indexOf(';'))+2);
291 		        t++;
292 	        }
293         	singleTaskIdsSavedBinding[t] = boundTaskIds[index];
294         	
295 	        var check = 0;
296         	if(singleTaskIdsInsertedBinding.length >= singleTaskIdsSavedBinding.length) {
297         		for(var x = 0; x < singleTaskIdsSavedBinding.length; x++) {
298         			for(var w = 0; w < singleTaskIdsInsertedBinding.length; w++) {
299         				if(singleTaskIdsInsertedBinding[w] == singleTaskIdsSavedBinding[x]) {
300         					check++;
301         					if(check == 2) {
302         						duplicate = true;
303         					}
304         					break;
305         				}
306         				duplicate = false;
307         			}
308         			if(duplicate == true) { //two items of set 1 found in set 2 --> contradicting
309     					break;
310     				}
311         		}
312         	} else {
313         		for(var x = 0; x < singleTaskIdsInsertedBinding.length; x++) {
314         			for(var w = 0; w < singleTaskIdsSavedBinding.length; w++) {
315         				if(singleTaskIdsSavedBinding[w] == singleTaskIdsInsertedBinding[x]) {
316         					check++;
317         					if(check == 2) {
318         						duplicate = true;
319         					}
320         					break;
321         				}
322         				duplicate = false;
323         			}
324         			if(duplicate == true) { //two items of set 1 found in set 2 --> contradicting
325     					break;
326     				}
327         		}
328         	}
329 
330 	        if(duplicate == true) { //entry does exist, delete?
331 				var bodDelete = confirm("Do you want to delete the existing contradicting binding for allowing this separation constraint to be entered? Otherwise, this separation constraint will be discarded.");
332 				if(bodDelete == true) {
333 					this.findBindingEntry(selectedTasks);
334 					duplicate = false;
335 				}
336 	        	break;
337 	        }
338         }//end check whether binding entry does already exist
339     	
340         returnValue = duplicate;
341         return returnValue;
342     },

344     writeEntry: function(savedSeparationsData, selectedTasks, taskElement) { //create JSON object and write to task attributes
345     	if(savedSeparationsData[1] > 0) { //saved entries do exist
346     		savedSeparationsData[1]++; //number of entries
347     		savedSeparationsData[2]++; //highestId needs to be higher than all found ids 
348     		savedSeparationsData[0] = savedSeparationsData[0].substring((savedSeparationsData[0].indexOf(':'))+2);
349     		savedSeparationsData[0] = "{'totalCount':" + savedSeparationsData[1] + savedSeparationsData[0] + ", {sodId:\"" + savedSeparationsData[2] + "\", SeparatedTasks:\"" + selectedTasks + "\"}]}";
350     	} else { //first entry to be done
351     		savedSeparationsData[1] = 1;
352     		savedSeparationsData[2]++; //highestId needs to be higher than all found ids
353     		savedSeparationsData[0] = "{'totalCount':" + savedSeparationsData[1] + ", 'items':[{sodId:\"" + savedSeparationsData[2] + "\", SeparatedTasks:\"" + selectedTasks + "\"}]}";
354     	}
355     	taskElement.setProperty("oryx-separationofduties", savedSeparationsData[0]);
356     },

358 	deleteSeparationEntry: function(taskElement, elementId) { //delete given entry from given task from separation contraints
359 		var returnValue = [];
360 		var separations = taskElement.properties["oryx-separationofduties"]; //get saved separations
361 		var keptSeparations;
362 		var temp;
363 		var totalCount = parseInt(separations.substring((separations.indexOf(':')) + 1));
364 		totalCount = totalCount - 1;
365 		if(totalCount > 0)  {
366 			separations = separations.substring(separations.indexOf('[')); //{'totalCount':3, 'items':
367 			keptSeparations = separations.substring(0,separations.indexOf('{')); //[
368 			separations = separations.substring(separations.indexOf('{')); //[
369 			temp = separations.substring(0,(separations.indexOf('"')) + 1); //{sodId:"
370 			separations = separations.substring((separations.indexOf('"')) + 1); //{sodId:"
371 			while(parseInt(separations) != elementId) {
372 				keptSeparations = keptSeparations + temp + separations.substring(0,(separations.indexOf('{')) - 2); //'id'", SeparatedTasks:"task_a; task_b"}
373 				separations = separations.substring((separations.indexOf('{')) - 2); //'id'", SeparatedTasks:"task_a; task_b"}
374 				temp = separations.substring(0,((separations.indexOf('"')) + 1)); //, {sodId:"
375 				separations = separations.substring((separations.indexOf('"')) + 1); //, {sodId:"
376 			}
377 			separations = separations.substring((separations.indexOf('}'))+3); //'sodId'", SeparatedTasks:"task_c; task_d; task_e"} or //{sodId:"'id'", SeparatedTasks:"task_c; task_d; task_e"},_
378 			var entry = "{'totalCount':" + totalCount + ", 'items':" + keptSeparations + separations + "]}";
379 			taskElement.setProperty("oryx-separationofduties", entry);
380 			
381 			entry = entry.toString();
382 			entry = entry.substring(0,entry.indexOf(']'));
383 		
384 		} else {
385 			var entry = "";
386 			taskElement.setProperty("oryx-separationofduties", entry);
387 		}
388 		
389 		returnValue[0] = entry;
390 		returnValue[1] = totalCount;
391 		return returnValue;
392 	},

394 	findBindingEntry: function(taskIds) { //find binding entry with given task ids to be deleted at all affected tasks
395 		var taskIdsCopy = taskIds;
396 		var singleTaskIds = [];
397 		var s = 0;
398 		
399 		//extract task ids of affected tasks
400     	while(taskIdsCopy.indexOf(';') > -1) {
401     		singleTaskIds[s] = taskIdsCopy.substring(0, taskIdsCopy.indexOf(';'));
402     		taskIdsCopy = taskIdsCopy.substring((taskIdsCopy.indexOf(';'))+2);
403 	        s++;
404         }
405     	singleTaskIds[s] = taskIdsCopy;
406 		
407 		taskIdsCopy = taskIds + ";";
408 		for (var m = 0; m < singleTaskIds.length; m++) { //go through all affected tasks
409 			var currentTask = this.getTaskById(singleTaskIds[m]);
410 			var bindings = currentTask.properties["oryx-bindingsofduties"]; //get saved bindings
411 			var boundTaskIds = [];
412 			var totalCount = 0;
413 			var i = 0; //counter
414 			var j = 0; //counter
415 			while(bindings.indexOf(',') > -1) {
416 	    		if(i%2 == 1) { //get element ids as integer
417 			    	bindings = bindings.substring((bindings.indexOf(',')) + 1);
418 	    		} else if(i == 0) { //get totalCount
419 	    			totalCount = parseInt(bindings.substring((bindings.indexOf(':')) + 1));
420 					bindings = bindings.substring((bindings.indexOf(',')) + 1);
421 	    		} else { //get task ids
422 	    			bindings = bindings.substring((bindings.indexOf('"')) + 1);
423 	    			boundTaskIds[j] = bindings.substring(0,bindings.indexOf('"'));
424 	    			j++;
425 	    			bindings = bindings.substring((bindings.indexOf(',')) + 1);
426 	    		}
427 		        i++;
428 	    	}
429 			//transform id string into array - one constraint set equals one array entry
430 	    	if(bindings.indexOf('"') > -1) {
431 	    		boundTaskIds[j] = bindings.substring((bindings.indexOf('"')) + 1);
432 	    	} else {
433 	    		boundTaskIds[j] = bindings; //never reached??
434 	    	}
435 	    	boundTaskIds[j]= boundTaskIds[j].substring(0,boundTaskIds[j].indexOf('"'));
436 						
437 			var singleTaskIdsStoredBinding = [];
438 			
439 			//check each binding entry
440 			for (var index = 0; index < boundTaskIds.length; index++) {
441 				var boundTasksCopy = boundTaskIds[index];
442 				var t = 0; //counter
443 				//extract task ids of stored elements
444 				while (boundTasksCopy.indexOf(';') > -1) {
445 					singleTaskIdsStoredBinding[t] = boundTasksCopy.substring(0, boundTasksCopy.indexOf(';'));
446 					boundTasksCopy = boundTasksCopy.substring((boundTasksCopy.indexOf(';')) + 2);
447 					t++;
448 				}
449 				singleTaskIdsStoredBinding[t] = boundTasksCopy;
450 				
451 				if (singleTaskIds.length >= singleTaskIdsStoredBinding.length) {
452 					var equalCounter = 0;
453 					for (var x = 0; x < singleTaskIdsStoredBinding.length; x++) {
454 						for (var y = 0; y < singleTaskIds.length; y++) {
455 							if (singleTaskIdsStoredBinding[x] == singleTaskIds[y]) {
456 								equalCounter++;
457 								break;
458 							}
459 						}
460 						if (equalCounter == singleTaskIdsStoredBinding.length) { //entry found
461 							this.deleteBindingEntry(currentTask, (index+1)); //delete entry
462 							break;
463 						}
464 					}
465 				}
466 				singleTaskIdsStoredBinding.splice(0,singleTaskIdsStoredBinding.length);
467 			}
468 		}
469 	},

499 		}

502 	getTaskById: function(taskId){
503 		var shapes = this.facade.getCanvas().getChildShapes(true);
504 		var task;
505 		for (var index = 0; index < shapes.length; index++) {
506     		if (shapes[index].properties["oryx-activitytype"] == "Task") {
507 				if (shapes[index].properties["oryx-id"] == taskId) {
508 					task = shapes[index];
509 					break;
510 				}
511 			}
512     	}
513 		return task;
514 	}
