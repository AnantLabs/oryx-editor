 22 if (!ORYX.Plugins) 
 23     ORYX.Plugins = new Object();
 24 
 25 ORYX.Plugins.ResourcesBoDShow = Clazz.extend({
 26 
 27     facade: undefined,
 28     
 29     construct: function(facade){
 30 		
 31         this.facade = facade;
 32         
 33 		this.active 		= false;
 34 		this.raisedOverlayEventIds = [];
 35 		this.raisedHighlightEventIds = [];
 36 		
 37         this.facade.offer({
 38             'name': ORYX.I18N.ResourcesBoDShow.name,
 39             'functionality': this.showBoD.bind(this),
 40             'group': ORYX.I18N.ResourcesBoDShow.group,
 41 			'dropDownGroupIcon': ORYX.PATH + "images/bod.png",
 42             'icon': ORYX.PATH + "images/bod_view.png",
 43             'description': ORYX.I18N.ResourcesBoDShow.desc,
 44             'index': 4,
 45             'toggle': true,
 46             'minShape': 1,
 47             'maxShape': 1
 48         });
 49 		
 50     },
 51     
 52 	showBoD: function(){
 53 		this.removeHighlightsAndOverlays();
 54 		var selectedElements = this.facade.getSelection();
 55  		if(selectedElements[0].properties["oryx-activitytype"] == "Task") {
 56 			this.highlightSelectedTask(selectedElements[0]);
 57 			this.prepareOverlays(selectedElements[0]);
 58 		} else {
 59 			alert("Please select a task to show the related Binding of Duties constraints.");
 60 		}
 61 	},
 62 	
 63 	highlightSelectedTask: function(task){ //edge marking of selected/ given task
 64 		if(!(task instanceof ORYX.Core.Shape)) return;
 65 		this.facade.raiseEvent({
 66 			type:			ORYX.CONFIG.EVENT_HIGHLIGHT_SHOW, 
 67 			highlightId:	task.properties["oryx-id"],
 68 			elements:		[task],
 69 			color:			'#28BB25'
 70 		});
 71 		this.raisedHighlightEventIds.push(task.id);
 72 	},
 73 	
 74 	showOverlaysForSeparations: function(task, elementId) { //colour given task green
 75 		if(!(task instanceof ORYX.Core.Shape)) return;
 76 		//set overlay attributes
 77 		var attr = {fill: "#28BB25", stroke:"white", "stroke-width": 1};
 78 		this.facade.raiseEvent({
 79 			type: 			ORYX.CONFIG.EVENT_OVERLAY_SHOW,
 80 			id: 			task.properties["oryx-id"],
 81 			shapes: 		[task],
 82 			attributes: 	attr
 83 		});
 84 		//calculate node position
 85 		var appearanceCounter = 0;
 86 		var nodePosition;
 87 		var x; //x coordinate for text
 88 		var y; //y coordinate for text
 89 		var cx; //x coordinate for circle center
 90 		var cy; //y coordinate for circle center
 91 		for(index = 0; index < this.raisedOverlayEventIds.length; index++) {
 92 			if (task.properties["oryx-id"] == this.raisedOverlayEventIds[index]) {
 93 				appearanceCounter++;
 94 			} 
 95 		}
 96 		switch(appearanceCounter) { //avoid overlappings of dependency markers as long as not more than 8 exist
 97 			case 0:		nodePosition = 'NW';
 98 						x = 9;
 99 						y = 17;
100 						cx = 13;
101 						cy = 13;
102 						break;
103 			case 1:		nodePosition = 'NE';
104 						x = -17;
105 						y = 17;
106 						cx = -13;
107 						cy = 13;
108 						break;
109 			case 2:		nodePosition = 'SW';
110 						x = 9;
111 						y = -9;
112 						cx = 13;
113 						cy = -13;
114 						break;
115 			case 3:		nodePosition = 'SE';
116 						x = -17;
117 						y = -9;
118 						cx = -13;
119 						cy = -13;
120 						break;
121 			case 4:		nodePosition = 'N';
122 						x = -4;
123 						y = 17;
124 						cx = 0;
125 						cy = 13;
126 						break;
127 			case 5:		nodePosition = 'S';
128 						x = -4;
129 						y = -9;
130 						cx = 0;
131 						cy = -13;
132 						break;
133 			case 6:		nodePosition = 'W';
134 						x = 9;
135 						y = 4;
136 						cx = 13;
137 						cy = 0;
138 						break;
139 			case 7:		nodePosition = 'E';
140 						x = -17;
141 						y = 4;
142 						cx = -13;
143 						cy = 0;
144 						break;
145 			case 8: 	alert("There exist more Binding of Duties constraints for task " + task.properties["oryx-id"] + ", but they will not be illustrated by a number in that task's rectangle.");
146 						break;
147 			default:	break;
148 		}
149 		if(elementId >= 10) { //keep marker centered based on single-digit or double-digit entry
150 			x = x - 4;
151 		}
152 		if(appearanceCounter < 8) { //skip number representation, if task occurs too often
153 			//create circle of marker
154 			var circle = ORYX.Editor.graft("http://www.w3.org/2000/svg", null, 
155 				['circle', {"cx":cx, "cy":cy, "r":"10", "stroke":"white", "fill":"white", "stroke-width":"2"}]
156 			);
157 			this.facade.raiseEvent({
158 				type: 			ORYX.CONFIG.EVENT_OVERLAY_SHOW,
159 				id: 			task.properties["oryx-id"],
160 				shapes: 		[task],
161 				node:			circle,
162 				nodePosition:	nodePosition
163 			});
164 			//create text/ number of marker
165 			var text = ORYX.Editor.graft("http://www.w3.org/2000/svg", null, 
166 				['text', {"x":x, "y":y, "style": "font-size: 12px;"}, elementId]
167 			);
168 			this.facade.raiseEvent({
169 				type: 			ORYX.CONFIG.EVENT_OVERLAY_SHOW,
170 				id: 			task.properties["oryx-id"],
171 				shapes: 		[task],
172 				node:			text,
173 				nodePosition:	nodePosition
174 			});
175 		}
176 		this.raisedOverlayEventIds.push(task.properties["oryx-id"]);
177 	},
178 	
179 	prepareOverlays: function(task) { //identify tasks to be coloured as a constraint exist with given and initiate colouring
180 		if(task.properties["oryx-bindingsofduties"] != "") {
181 			var jsonObject = task.properties["oryx-bindingsofduties"].evalJSON();
182 			var items = jsonObject.items.toArray();
183 			var overlayTask;
184 			for(var index = 0; index < items.length; index++) {
185 				var taskIds = items[index].BoundTasks;
186 				while(taskIds.indexOf(';') > -1) {
187 //					overlayTask = this.facade.getCanvas().getChildById(taskIds.substring(0,taskIds.indexOf(';')));
188 					overlayTask = this.getTaskById(taskIds.substring(0,taskIds.indexOf(';')));
189 					taskIds = taskIds.substring((taskIds.indexOf(';')) + 2);
190 					this.showOverlaysForSeparations(overlayTask, index+1);
191 				}
192 				overlayTask = this.getTaskById(taskIds);
193 				//initiate colouring
194 				this.showOverlaysForSeparations(overlayTask, index+1);
195 			}
196 		} else {
197 			alert("No Binding of Duties Constraints are defined for this task");
198 		}
199 	},
200 	
201 	removeHighlightsAndOverlays: function(){
202 		var allShapes = this.facade.getCanvas().getChildShapes(true);
203 		var allShapeIds = [];
204 		var i = 0;
205 		//get all tasks of canvas
206 		for (var index = 0; index < allShapes.length; index++) {
207     		if (allShapes[index].properties["oryx-activitytype"] == "Task") {
208 				allShapeIds[i] = allShapes[index].properties["oryx-id"];
209 				i++;
210 			}
211     	}
212 		
213 		//remove highlights
214 		allShapeIds.each(function(id){
215 			this.facade.raiseEvent({
216 					type: 	ORYX.CONFIG.EVENT_HIGHLIGHT_HIDE,
217 					highlightId: 	id
218 				});
219 		}.bind(this))
220 		this.raisedHighlightEventIds = [];
221 		
222 		//remove overlays 
223 		allShapeIds.each(function(id){
224 			this.facade.raiseEvent({
225 					type: 	ORYX.CONFIG.EVENT_OVERLAY_HIDE,
226 					id: 	id
227 				});
228 		}.bind(this))
229 		this.raisedOverlayEventIds = [];
230 	},
231 	
232 	getTaskById: function(taskId){
233 		var shapes = this.facade.getCanvas().getChildShapes(true);
234 		var task;
235 		for (var index = 0; index < shapes.length; index++) {
236     		if (shapes[index].properties["oryx-activitytype"] == "Task") {
237 				if (shapes[index].properties["oryx-id"] == taskId) {
238 					task = shapes[index];
239 					break;
240 				}
241 			}
242     	}
243 		return task;
244 	}
245 });
