1 if(!ORYX.Plugins)
  2 	ORYX.Plugins = new Object();
  3 
  4 ORYX.Plugins.CpnSupport = Clazz.extend({
  5 
  6 	/**
  7 	 *	Constructor
  8 	 *	@param {Object} Facade: The Facade of the Editor
  9 	 */
 10 	construct: function(facade) 
 11 	{
 12 		this.facade = facade;
 13 		
 14 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_RESIZE_END, this.resetTokenPosition.bind(this));
 15 	},
 16 	
 17 	resetTokenPosition: function()
 18 	{
 19 		// Get selected places		
 20 		var allplaces = this.facade.getSelection().findAll(function(selectedItem) {
 21 			return (selectedItem.getStencil().id() === "http://b3mn.org/stencilset/coloredpetrinet#Place");
 22 		});
 23 		
 24 		if (allplaces.length > 0)
 25 		{
 26 			allplaces.each(function(place) {
 27 				
 28 				var placeBounds = place.absoluteBounds();
 29 				var placeCenter = placeBounds.center();
 30 				
 31 				// Calculate radius in order to check if a token is in the place
 32 				var radiusY = placeCenter.y - placeBounds.upperLeft().y;
 33 				var radiusX = placeCenter.x - placeBounds.upperLeft().x;
 34 				var radius = Math.min(radiusY,radiusX);
 35 				var c = radius / 2;
 36 				
 37 				// Get all tokens inside the place 
 38 				var alltokens = place.getChildNodes(false).findAll(function(child) {
 39 					return (child.getStencil().id() === "http://b3mn.org/stencilset/coloredpetrinet#Token");
 40 				});
 41 				
 42 				if (alltokens.length > 0)
 43 				{
 44 					var i = 0;
 45 					var x = 0;
 46 					var y = 0;
 47 					
 48 					alltokens.each(function(token) {
 49 						var tokenBounds = token.absoluteBounds();
 50 						var tokenCenter = tokenBounds.center();
 51 						
 52 						// Calculate the distance between token and center of the place
 53 						var diffX = placeCenter.x - tokenCenter.x;
 54 						var diffY = placeCenter.y - tokenCenter.y;
 55 						var distanceToPlaceCenter= diffX*diffX + diffY*diffY; // take care it's squared
 56 						
 57 						// Check if the token is in the place
 58 						if (radius*radius <= distanceToPlaceCenter)
 59 						{	// if the token is out of the place, calculate the position for the token
 60 							// the token are positioned in circle which is in the place
 61 							y = Math.round(Math.sin((Math.PI / 6) * i) * c);
 62 							x = Math.round(Math.cos((Math.PI / 6) * i) * c);
 63 							// take care centerMoveTo is referred to the position in the selected place (not absolute) 
 64 							token.bounds.centerMoveTo(place.bounds.width() / 2  + x, place.bounds.height() / 2 + y);
 65 							token.update();
 66 							i = i + 1;
 67 						}
 68 					});					
 69 				}
 70 			});
 71 
 72 		}			
 73 		this.facade.getCanvas().update();
 74 	}	
 75 });
