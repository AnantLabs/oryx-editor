 24 if(!ORYX.Plugins)
 25 	ORYX.Plugins = new Object();
 26 
 27 /**
 28  * Transforms a BPEL process into its BPMN representation.
 29  * 
 30  */
 31 ORYX.Plugins.BPEL2BPMN = Clazz.extend({
 32 
 33 	facade: undefined,
 34 
 35 	/**
 36 	 * Offers the plugin functionality:
 37 	 * 
 38 	 */
 39 	construct: function(facade) {
 40 		this.facade = facade;
 41 		
 42 		this.facade.offer({
 43 			'name':"Transform BPEL into BPMN",
 44 			'functionality': this.transform.bind(this),
 45 			'group': 			'Export',
 46             dropDownGroupIcon: ORYX.PATH + "images/import.png",
 47 			'description': "Transform a BPEL process into its BPMN representation",
 48 			'index': 1,
 49 			'minShape': 0,
 50 			'maxShape': 0});
 51 			
 52 	},
 53 
 54 	
 55 	/**
 56 	 * Trigger the actual BPEL 2 BPMN transformation.
 57 	 */
 58 	transform: function(){
 59 		this.openUploadDialog();
 60 	},
 61 	
 62 	/**
 63 	 * Opens a upload dialog.
 64 	 * (adapted from the epcSupport plugin)
 65 	 * 
 66 	 */
 67 	openUploadDialog: function(){
 68 		
 69 		var form = new Ext.form.FormPanel({
 70 			frame : false,
 71 			defaultType : 'textfield',
 72 		 	waitMsgTarget : true,
 73 		  	labelAlign : 'left',
 74 		  	buttonAlign: 'right',
 75 		  	fileUpload : true,
 76 		  	enctype : 'multipart/form-data',
 77 		  	style: 'font-size:12px;',
 78 		  	items : [
 79 		  	{
 80 		    	fieldLabel : 'File',
 81 		    	inputType : 'file',
 82 			  	style: 'font-size:12px;',
 83 				allowBlank: false
 84 		  	}]
 85 		});
 86 		
 87 		var errorMsg = new Ext.Panel({style: 'font-size:12px;', autoScroll: true});
 88 		
 89 		var dialog;
 90 
 91 		dialog = new Ext.Window({ 
 92 			autoCreate: true, 
 93 			title: 'Upload BPEL File', 
 94 			height: 240, 
 95 			width: 400, 
 96 			modal:true,
 97 			collapsible:false,
 98 			fixedcenter: true, 
 99 			shadow:true, 
100 		  	style: 'font-size:12px;',
101 			proxyDrag: true,
102 			resizable:true,
103 			items: [new Ext.form.Label({text: "Select a BPEL (.bpel) file and transform it to BPMN.", style: 'font-size:12px;'}),form, errorMsg],
104 			buttons:[{
105 				text:"Submit",
106 				handler: function()
107 				{
108 					form.form.submit({
109 			      		clientValidation: false,
110 						url: ORYX.PATH + '/bpel2bpmn',
111 			      		waitMsg: "Transforming...",
112 			      		success: function(f,a){
113 							/*
114 							 * The XML that comes from the server is escaped. Therefore we need to replace the escape symbols once again.
115 							 */
116 							var myString = a.response.responseText.replace(/</g,'<').replace(/>/g,'>');
117 							
118 							if (myString) {
119 							
120 								var resultObject = myString.evalJSON();
121 								
122 								var eRDF = resultObject.content;
123 								var successfulValidation = resultObject.successValidation;
124 								var validationError = resultObject.validationError;
125 								
126 								/*
127 								 * In case the BPEL file complied to the schema the dialog is hidden.
128 								 */
129 								if (successfulValidation) {
130 									dialog.hide();
131 								}
132 								else {
133 									/*
134 									 * In case the BPEL file did not comply to the schema the dialog remains open.
135 									 */
136 									errorMsg.body.dom.innerHTML = '<p style="background-color: transparent;">Your BPEL file does not comply with the XML schema definition. <br /> <br />Error message: ' + validationError + '</p>';
137 								}
138 								
139 								/*
140 								 * In all cases we try to import the resulting eRDF.
141 								 */
142 								eRDF = '<?xml version="1.0" encoding="utf-8"?><html xmlns="http://www.w3.org/1999/xhtml" xmlns:b3mn="http://b3mn.org/2007/b3mn" xmlns:ext="http://b3mn.org/2007/ext" xmlns:rdf="http://www.w3.org/1999/02/22-rdf-syntax-ns#" xmlns:atom="http://b3mn.org/2007/atom+xhtml"><head profile="http://purl.org/NET/erdf/profile"><link rel="schema.dc" href="http://purl.org/dc/elements/1.1/" /><link rel="schema.dcTerms" href="http://purl.org/dc/terms/ " /><link rel="schema.b3mn" href="http://b3mn.org" /><link rel="schema.oryx" href="http://oryx-editor.org/" /><link rel="schema.raziel" href="http://raziel.org/" /><base href="http://localhost:8080/backend/poem/new" /></head><body>'
143 									+eRDF+'</body></html>';
144 								var parser	= new DOMParser();			
145 								this.facade.importERDF(parser.parseFromString(eRDF ,"text/xml"));
146 							}
147 							else {
148 								/*
149 								 * Something went totally wrong. No chance to recover.
150 								 */
151 								Ext.MessageBox.show({
152 		           					title: 'Error',
153 		          	 				msg: 'The BPEL file could not be imported.',
154 		           					buttons: Ext.MessageBox.OK,
155 		           					icon: Ext.MessageBox.ERROR
156 		       					});
157 							}							
158 			      		}.bind(this),
159 						failure: function(f,a){
160 							dialog.hide();
161 							Ext.MessageBox.show({
162 	           					title: 'Error',
163 	          	 				msg: a.response.responseText.substring(a.response.responseText.indexOf("content:'")+9, a.response.responseText.indexOf("'}")),
164 	           					buttons: Ext.MessageBox.OK,
165 	           					icon: Ext.MessageBox.ERROR
166 	       					});
167 			      		}.bind(this)
168 			  		});
169 			  	}.bind(this)
170 			}]
171 		});
172 		dialog.on('hide', function(){
173 			dialog.destroy(true);
174 			delete dialog;
175 		});
176 		dialog.show();
177 	},
178 });
