 24 if(!ORYX.Plugins)
 25 	ORYX.Plugins = new Object();
 26 
 27 ORYX.Plugins.XFormsExport = ORYX.Plugins.AbstractPlugin.extend({
 28 
 29 	facade: undefined,
 30 
 31 	construct: function(facade) {
 32 		this.facade = facade;
 33 
 34 		this.facade.offer({
 35 			'name':ORYX.I18N.XFormsSerialization.exportXForms,
 36 			'functionality': this.exportXForms.bind(this),
 37 			'group': ORYX.I18N.XFormsSerialization.group,
 38 			'icon': ORYX.PATH + "images/xforms_export.png",
 39 			'description': ORYX.I18N.XFormsSerialization.exportXFormsDesc,
 40 			'index': 2,
 41 			'minShape': 0,
 42 			'maxShape': 0});
 43 	},
 44 	
 45 	exportXForms: function(){
 46 		this._showCssDialog();
 47 	},
 48 
 49 	exportIt: function(cssUrl){
 50 
 51 		// raise loading enable event
 52         this.facade.raiseEvent({
 53             type: ORYX.CONFIG.EVENT_LOADING_ENABLE
 54         });
 55 		
 56 		//this.checkClientXFormsSupport();
 57             
 58 		// asynchronously ...
 59         window.setTimeout((function(){
 60 			
 61 			// ... save synchronously
 62             this.exportSynchronously(cssUrl);
 63 			
 64 			// raise loading disable event.
 65             this.facade.raiseEvent({
 66                 type: ORYX.CONFIG.EVENT_LOADING_DISABLE
 67             });
 68 			
 69         }).bind(this), 10);
 70 
 71 		return true;
 72     },
 73 
 74     exportSynchronously: function(cssUrl) {
 75 		 var resource = location.href;
 76 		try {
 77 			var serialized_rdf 	= this.getRDFFromDOM();
 78 			serialized_rdf 		= serialized_rdf.startsWith("<?xml") ? serialized_rdf : "<?xml version=\"1.0\" encoding=\"UTF-8\"?>" + serialized_rdf;
 79 
 80 			// Send the request to the server.
 81 			new Ajax.Request(ORYX.CONFIG.XFORMS_EXPORT_URL, {
 82 				method: 'POST',
 83 				asynchronous: false,
 84 				parameters: {
 85 					resource: resource,
 86 					data: serialized_rdf,
 87 					css: cssUrl
 88 				},
 89 				onSuccess: function(request){
 90 					
 91 						var win = window.open("data:text/xml," +
 92 								request.responseText, 
 93 								"_blank", 
 94 								"resizable=yes,width=640,height=480,toolbar=0,scrollbars=yes");
 95 						
 96 				}
 97 			});
 98 			
 99 		} catch (error){
100 			this.facade.raiseEvent({type:ORYX.CONFIG.EVENT_LOADING_DISABLE});
101 			Ext.Msg.alert(ORYX.I18N.Oryx.title, error);
102 	 	}
103 	},
104 	
105 	checkClientXFormsSupport: function() {
106 		if(!clientSupportsXForms) {
107 			
108 			var output = ORYX.I18N.XFormsSerialization.noClientXFormsSupportDesc;
109 
110 			var win = new Ext.Window({
111                             width: 320,
112                             height: 240,
113                             resizable: false,
114                             minimizable: false,
115                             modal: true,
116                             autoScroll: true,
117                             title: ORYX.I18N.XFormsSerialization.noClientXFormsSupport,
118                             html: output,
119                             buttons: [{
120                                 text: ORYX.I18N.XFormsSerialization.ok,
121                                 handler: function(){
122                                     win.hide();
123                                 }
124                             }]
125                         });
126         	win.show();
127 			
128 		}
129 	},
130 	
131 	/**
132 	 * Opens dialog
133 	 * 
134 	 */
135 	_showCssDialog: function( successCallback ){
136 	
137 	    var form = new Ext.form.FormPanel({
138 			baseCls: 		'x-plain',
139 	        labelWidth: 	50,
140 	        defaultType: 	'textfield',
141 	        items: [{
142 	            text : 		ORYX.I18N.XFormsSerialization.selectCss, 
143 				style : 	'font-size:12px;margin-bottom:10px;display:block;',
144 	            anchor:		'100%',
145 				xtype : 	'label' 
146 	        },{
147 	            xtype: 'textarea',
148 	            hideLabel: true,
149 	            name: 'msg',
150 	            anchor: '100% -30' 
151 	        }]
152 	    });
153 
154 
155 
156 		// Create the panel
157 		var dialog = new Ext.Window({ 
158 			autoCreate: true, 
159 			layout: 	'fit',
160 			plain:		true,
161 			bodyStyle: 	'padding:5px;',
162 			title: 		ORYX.I18N.XFormsSerialization.expTitle, 
163 			height: 	150, 
164 			width:		500,
165 			modal:		true,
166 			fixedcenter:true, 
167 			shadow:		true, 
168 			proxyDrag: 	true,
169 			resizable:	true,
170 			items: 		[form],
171 			buttons:[
172 				{
173 					text:ORYX.I18N.XFormsSerialization.ok,
174 					handler:function(){		
175 						this.exportIt(form.items.items[1].getValue());
176 						dialog.hide();
177 					}.bind(this)
178 				},{
179 					text:ORYX.I18N.XFormsSerialization.close,
180 					handler:function(){
181 						dialog.hide();
182 					}.bind(this)
183 				}
184 			]
185 		});
186 		
187 		// Destroy the panel when hiding
188 		dialog.on('hide', function(){
189 			dialog.destroy(true);
190 			delete dialog;
191 		});
192 
193 
194 		// Show the panel
195 		dialog.show();
196 
197 	}
198 
199 });
