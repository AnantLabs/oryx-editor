390 	/**
391 	 * This method serializes a single div into a string that satisfies the
392 	 * client/server communication protocol. It ingnores all elements that have
393 	 * an attribute named class that includes 'transient'.
394 	 * @param {Object} node the element to serialize.
395 	 * @param {Object} preserveNamespace whether to preserve the parent's
396 	 *                 namespace. If you are not sure about namespaces, provide
397 	 *                 just the element to be serialized.
398 	 */
399 	serialize: function(node, preserveNamespace) {
400 
401 		if (node.nodeType == node.ELEMENT_NODE) {
402 			// serialize an element node.
403 			
404 			var children = $A(node.childNodes);
405 			var attributes = $A(node.attributes);
406 			var clazz = new String(node.getAttribute('class'));
407 			var ignore = clazz.split(' ').member('transient');
408 
409 			// ignore transients.
410 
411 			if(ignore)
412 				return '';
413 
414 			// start serialization.
415 			
416 			var result = '<' + node.nodeName;
417 			
418 			// preserve namespace?
419 			if(!preserveNamespace) 
420 				result += ' xmlns="' + (node.namespaceURI ? node.namespaceURI : XMLNS.XHTML) + '" xmlns:oryx="http://oryx-editor.org"';
421 			
422 			// add all attributes.
423 			
424 			attributes.each(function(attribute) {
425 				result += ' ' + attribute.nodeName + '="' +
426 					attribute.nodeValue + '"';});
427 			
428 			// close if no children.
429 			
430 			if(children.length == 0)
431 				result += '/>';
432 				
433 			else {
434 				
435 				// serialize all children.
436 				
437 				result += '>';
438 				children.each(function(_node) {
439 					result += DataManager.serialize(_node, true)});
440 				result += '</' + node.nodeName + '>'
441 			}
442 
443 			return result;
444 			
445 		} else if (node.nodeType == node.TEXT_NODE) {
446 			
447 			// serialize a text node.
448 			return  node.nodeValue;
449 		}
450 		
451 		//TODO serialize cdata areas also.
452 		//TODO work on namespace awareness.
453 	},

455 	addTriple: function(triple) {
456 
457 		// assert the subject is a resource
458 		
459 		if(!triple.subject.type == ERDF.LITERAL)
460 			throw 'Cannot add the triple ' + triple.toString() +
461 				' because the subject is not a resource.'
462 		
463 		// get the element which represents this triple's subject.
464 		var elementId = ERDF.__stripHashes(triple.subject.value);
465 		var element = $(elementId);
466 				
467 		// assert the subject is inside this document.
468 		if(!element)
469 			throw 'Cannot add the triple ' + triple.toString() +
470 				' because the subject "'+elementId+'" is not in the document.';
471 
472 		if(triple.object.type == ERDF.LITERAL)
473 
474 			// object is literal
475 			DataManager.graft(XMLNS.XHTML, element, [
476 				'span', {'class': (triple.predicate.prefix + "-" +
477 					triple.predicate.name)}, triple.object.value.escapeHTML()
478 			]);
479 			
480 		else {
481 
482 			// object is resource
483 			DataManager.graft(XMLNS.XHTML, element, [
484 				'a', {'rel': (triple.predicate.prefix + "-" +
485 					triple.predicate.name), 'href': triple.object.value}
486 			]);
487 			
488 		}
489 
490 		return true;
491 	},

576 	/**
577 	 * graft() function
578 	 * Originally by Sean M. Burke from interglacial.com, altered for usage with
579 	 * SVG and namespace (xmlns) support. Be sure you understand xmlns before
580 	 * using this funtion, as it creates all grafted elements in the xmlns
581 	 * provided by you and all element's attribures in default xmlns. If you
582 	 * need to graft elements in a certain xmlns and wish to assign attributes
583 	 * in both that and another xmlns, you will need to do stepwise grafting,
584 	 * adding non-default attributes yourself or you'll have to enhance this
585 	 * function. Latter, I would appreciate: martin�apfelfabrik.de
586 	 * @param {Object} namespace The namespace in which
587 	 * 					elements should be grafted.
588 	 * @param {Object} parent The element that should contain the grafted
589 	 * 					structure after the function returned.
590 	 * @param {Object} t the crafting structure.
591 	 * @param {Object} doc the document in which grafting is performed.
592 	 */
593 	graft: function(namespace, parent, t, doc) {
594 		
595 	    doc = (doc || (parent && parent.ownerDocument) || document);
596 	    var e;
597 	    if(t === undefined) {
598 	        echo( "Can't graft an undefined value");
599 	    } else if(t.constructor == String) {
600 	        e = doc.createTextNode( t );
601 	    } else {
602 	        for(var i = 0; i < t.length; i++) {
603 	            if( i === 0 && t[i].constructor == String ) {
604 					var snared = t[i].match( /^([a-z][a-z0-9]*)\.([^\s\.]+)$/i );
605 	                if( snared ) {
606 	                    e = doc.createElementNS(namespace, snared[1]);
607 	                    e.setAttributeNS(null, 'class', snared[2] );
608 	                    continue;
609 	                }
610 	                snared = t[i].match( /^([a-z][a-z0-9]*)$/i );
611 	                if( snared ) {
612 	                    e = doc.createElementNS(namespace, snared[1]);  // but no class
613 	                    continue;
614 	                }
615 	
616 	                // Otherwise:
617 	                e = doc.createElementNS(namespace, "span");
618 	                e.setAttribute(null, "class", "namelessFromLOL" );
619 	            }
620 	
621 	            if( t[i] === undefined ) {
622 	                echo("Can't graft an undefined value in a list!");
623 	            } else if( t[i].constructor == String || t[i].constructor == Array) {
624 	                this.graft(namespace, e, t[i], doc );
625 	            } else if(  t[i].constructor == Number ) {
626 	                this.graft(namespace, e, t[i].toString(), doc );
627 	            } else if(  t[i].constructor == Object ) {
628 	                // hash's properties => element's attributes
629 	                for(var k in t[i]) { e.setAttributeNS(null, k, t[i][k] ); }
630 	            } else if(  t[i].constructor == Boolean ) {
631 	                this.graft(namespace, e, t[i] ? 'true' : 'false', doc );
632 				} else
633 					throw "Object " + t[i] + " is inscrutable as an graft arglet.";
634 	        }
635 	    }
636 		
637 		if(parent) parent.appendChild(e);
638 	
639 	    return Element.extend(e); // return the topmost created node
640 	},

663 	query: function(subject, predicate, object) {
664 
665 		/*
666 		 * Typical triple.
667 		 *	{value: subject, type: subjectType},
668 		 *	{prefix: schema.prefix, name: property},
669 		 *	{value: object, type: objectType});
670 		 */	
671 		 	
672 		return DataManager._triples.select(function(triple) {
673 			
674 			var select = ((subject) ?
675 				(triple.subject.type == subject.type) &&
676 				(triple.subject.value == subject.value) : true);
677 			if(predicate) {
678 				select = select && ((predicate.prefix) ?
679 					(triple.predicate.prefix == predicate.prefix) : true);
680 				select = select && ((predicate.name) ?
