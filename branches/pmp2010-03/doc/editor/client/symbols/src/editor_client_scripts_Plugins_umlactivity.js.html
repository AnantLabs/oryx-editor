 31 /**
 32  * The UMLActivity plugin provides layout methods referring to the UMLActivity stencilset. 
 33  * 
 34  * @class ORYX.Plugins.UMLActivity
 35  * @extends Clazz
 36  * @param {Object} facade The facade of the editor
 37  */
 38  ORYX.Plugins.UMLActivity = 
 39 /** @lends ORYX.Plugins.UMLActivity.prototype */
 40 {
 41 	/**
 42 	 * Creates a new instance of the UMLActivity plugin and registers it on the
 43 	 * layout events listed in the UMLActivity stencil set.
 44 	 * 
 45 	 * @constructor
 46 	 * @param {Object} facade The facade of the editor
 47 	 */
 48 	construct: function(facade) {
 49 		this.facade = facade;
 50 		this.facade.registerOnEvent('layout.uml.activityPartition', this.handleLayoutActivityPartition.bind(this));
 51 	},
 52 	
 53 	/**
 54 	  * Global variable which contains a hashmap of the bounds of the activityPartions in the diagram
 55 	  * @private
 56 	  */
 57 	hashedBounds : {}, 
 58 	
 59 	/**
 60 	* Helper method, which returns true, if the received shape is an Activiy Partition. 
 61 	*@private
 62 	*@param {Object} shape The shape that is checked for being an Activiy Partition.
 63 	*@return {boolean} The result is true, if the shape is an Activiy Partition.
 64 	*/
 65 	isActivityPartitionNode : function (shape) {
 66 		return "http://b3mn.org/stencilset/umlactivity#activitypartition" == shape.getStencil().id().toLowerCase();
 67 	},
 68 	
 69 	/**
 70 	 * Handler for layouting event 'layout.uml.activityPartition'
 71 	 * Initiates also the layouting and hashing of the bounds of all activityPartitions
 72 	 * @param {Object} event This event is fired when an activityPartition Node is moved/resized/edited
 73 	 */
 74 	handleLayoutActivityPartition: function(event){
 75 		
 76 		var rootPartition = event.shape;
 77 		var directChildPartitions = this.getChildActivityPartitions(rootPartition, false);
 78 		
 79 		var currentShape = this.facade.getSelection().first(); 
 80 		currentShape = currentShape || rootPartition;
 81 		
 82 		//Guards
 83 		if (this.isActivityPartitionNode(rootPartition.parent)) {return;}
 84 		if(!this.isActivityPartitionNode(rootPartition)) {return;}
 85 		if (!this.isActivityPartitionNode(currentShape)) {return;}
 86 		if (directChildPartitions.length <= 0) {return;}
 87 		
 88 		this.currentRootPartition = rootPartition;
 89 		
 90 		if (!this.hashedBounds[rootPartition.resourceId]) {
 91 			this.hashedBounds[rootPartition.resourceId] = {};
 92 		}
 93 				
 94 		var allChildPartitions = this.getChildActivityPartitions(rootPartition, true);
 95 		var addedPartitions = this.getAllAddedPartitions(allChildPartitions);
 96 		var deletedPartitions = this.getAllDeletedPartitions(allChildPartitions);
 97 				
 98 		if (addedPartitions.length > 0){
 99 			currentShape = addedPartitions.first();
100 		}
101 		
102 		// layouting 
103 		if (deletedPartitions.length > 0 || addedPartitions.length > 0) {
104 			this.resizeAfterAddorDeleteOfPartition(rootPartition, directChildPartitions);
105 		}
106 		else if (rootPartition == currentShape) {
107 			this.resizeAfterRootPartitionChangend ( rootPartition, directChildPartitions);
108 		}	
109 		else {
110 			this.resizeAfterChildPartitionChanged ( rootPartition, directChildPartitions, currentShape);
111 		}
112 				
113 		this.cachePositions(rootPartition, allChildPartitions);
114 	},
115 	/**
116 	 * Caches the bounds of an activityPartiton in a two dimensional array
117 	 * The in the first dimension the rootPartition is saved, in the second dimension allChildPartitions are saved
118 	 * @private
119 	 * @param {ORYX.Core.Node} rootPartition This parameter a an activityPartition, whose parent is not an activityPartition
120 	 * @param {ORYX.Core.Node[]} allChildPartitions This parameter is an array of all child partitions of the rootPartition
121 	 */
122 	cachePositions : function(rootPartition, allChildPartitions) {
123 		this.hashedBounds[rootPartition.resourceId] = {};
124 		allChildPartitions.each(function(partition){
125 				this.hashedBounds[rootPartition.resourceId][partition.resourceId] = partition.absoluteBounds();
126 				this.forceToUpdateActivityPartition(partition);
127 			}.bind(this));
128 	},
129 	
130 	/**
131 	 * Resizes all activityPartitions which belong to the rootPartition
132 	 * after a partition is added or deleted
133 	 * @private
134 	 * @param {ORYX.Core.Node} rootPartition This parameter a an activityPartition, whose parent is not an activityPartition
135 	 * @param {ORYX.Core.Node[]} directChildPartitions This parameter is an array of all direct child partitions of the rootPartition
136 	 */
137 	resizeAfterAddorDeleteOfPartition : function (rootPartition, directChildPartitions){
138 		var width, height;
139 		width = this.updateActivityPartitionWidth(rootPartition);
140 		height = this.adjustActivityPartitionHeight(directChildPartitions, rootPartition.bounds.height());	
141 		rootPartition.update();
142 		this.setActivityPartitionDimensions(rootPartition, width, height);
143 	},
144 	
145 	/**
146 	 * Resizes all activityPartitions which belong to the rootPartition
147 	 * after the rootPartition is resized/moved
148 	 * @private
149 	 * @param {ORYX.Core.Node} rootPartition This parameter a an activityPartition, whose parent is not an activityPartition
150 	 * @param {ORYX.Core.Node[]} directChildPartitions This parameter is an array of all direct child partitions of the rootPartition
151 	 */
152 	resizeAfterRootPartitionChangend : function (rootPartition, directChildPartitions){
153 		var width, height;
154 		width = this.adjustActivityPartitionWidth(directChildPartitions, undefined, rootPartition.bounds.width());
155 		height = this.adjustActivityPartitionHeight(directChildPartitions, rootPartition.bounds.height());
156 		this.setActivityPartitionDimensions(rootPartition, width, height);
157 	},
158 	
159 	/**
160 	 * Resizes all activityPartitions which belong to the rootPartition
161 	 * after the rootPartition is resized/moved
162 	 * @private
163 	 * @param {ORYX.Core.Node} rootPartition This parameter a an activityPartition, whose parent is not an activityPartition
164 	 * @param {ORYX.Core.Node[]} directChildPartitions This parameter is an array of all direct child partitions of the rootPartition
165 	 */
166 	resizeAfterChildPartitionChanged : function ( rootPartition, directChildPartitions, currentShape){
167 		var width, height;
168 		width = this.adjustActivityPartitionWidth(directChildPartitions, currentShape);
169 		height = this.adjustActivityPartitionHeight(directChildPartitions, currentShape.bounds.height()+(this.getDepth(currentShape,rootPartition)*30));
170 		this.setActivityPartitionDimensions(rootPartition, width, height);
171 	},
172 	
173 	/**
174 	 * Finds all Partitions, which are added to the currentRootPartition
175 	 * At the time a layouting event is fired the lanes are already children 
176 	 * of the currentRootPartition, but they are not added to the hashedBounds
177 	 * @private
178 	 * @param {ORYX.Core.Node[]} allChildPartitions This parameter is an array of all child partitions of the rootPartition
179 	 * @return {ORYX.Core.Node[]} The result is an array of all added partitions
180 	 */
181 	getAllAddedPartitions : function (allChildPartitions){
182 			return allChildPartitions.findAll(function(shape) {
183 					if (!this.hashedBounds[this.currentRootPartition.resourceId][shape.resourceId]) {return shape;}
184 					}.bind(this));
185 	},
186 	
187 	/**
188 	 * Finds all Partitions, which are deleted from the currentRootPartition
189 	 * At the time a layouting event is fired the lanes are not children of the 
190 	 * currentRootPartition any more, but they are still referenced in the hashedBounds 
191 	 * @private
192 	 * @param {ORYX.Core.Node[]} allChildPartitions This parameter is an array of all child partitions of the rootPartition
193 	 * @return {ORYX.Core.Node[]} The result is an array of all deleted partitions
194 	 */
195 	getAllDeletedPartitions : function (allChildPartitions){
196 	
197 		var resourceIds = $H(this.hashedBounds[this.currentRootPartition.resourceId]).keys();
198 		
199 		var deletedResourceIds = resourceIds.reject(function (resourceId){
200 			return allChildPartitions.any(function(partition){ return partition.resourceId == resourceId;});
201 			}); 
202 		
203 		var deletedPartitions = deletedResourceIds.findAll(function(resourceId) {
204 			return this.hashedBounds[this.currentRootPartition.resourceId][resourceId];
205 		}.bind(this));
206 		
207 		return deletedPartitions;
208 		
209 	},
210 	
211 	/**
212 	 * Forces an activityPartition to update itself, 
213 	 * if the width saved in the stencil, that belongs to 
214 	 * the shape is different, to avoid inconsistencies
215 	 * @private
216 	 * @param {ORYX.Core.Node} partition This parameter is an Partition in an incosistent layouting state
217 	 */
218 	forceToUpdateActivityPartition: function(partition){
219 		
220 		if (partition.bounds.width() !== partition._svgShapes[0].width) {	
221 			partition.isChanged = true;
222 			partition.isResized = true;
223 			partition._update();
224 		}
225 	},
226 	
227 	/**
228 	 * Forces an activityPartition to update itself, 
229 	 * if the width saved in the stencil, that belongs to 
230 	 * the shape is different, to avoid inconsistencies
231 	 * @private
232 	 * @param {ORYX.Core.Node} partition This parameter is an Partition in an incosistent layouting state
233 	 */
234 	getDepth: function(child, parent){
235 		
236 		var i=0;
237 		while(child && child.parent && child !== parent){
238 			child = child.parent;
239 			++i;
240 		}
241 		return i;
242 	},
243 	
244 	/**
245 	 * Sets the bounds and therefore the size of an ActivityPartition
246 	 * @private
247 	 * @param {ORYX.Core.Node} shape This parameter is an Partition which gets new bounds
248 	 * @param {number} width The new width of the Partition
249 	 * @param {number} height The new height of the Partition
250 	 */
251 	setActivityPartitionDimensions: function (shape, width, height){
252 		var isChildActivityPartition = this.isActivityPartitionNode(shape);
253 		isChildActivityPartition = (isChildActivityPartition && this.isActivityPartitionNode(shape.parent));
254 		shape.bounds.set(
255 			shape.bounds.a.x,
256 			isChildActivityPartition ? 30 : shape.bounds.a.y,
257 			width ? shape.bounds.a.x + width : shape.bounds.b.x,
258 			height ? shape.bounds.a.y + height - (isChildActivityPartition ? 30:0) : shape.bounds.b.y
259 		);
260 	},
261 	
262 	/**
263 	 * Lowers an ActivityPartition, so that the head of the parent partition is visible
264 	 * @private
265 	 * @param {ORYX.Core.Node} shape This parameter is an Partition which gets new bounds
266 	 * @param {number} x The new relative Offset to the left edge of parent partition 
267 	 */
268 	setActivityPartitionPosition: function(shape, x){
269 		shape.bounds.moveTo(x, 30);
270 	},
271 	
272 	/**
273 	 * Sets recursively the new height of the child partitions
274 	 * @private
275 	 * @param {ORYX.Core.Node[]} partitions Array of all direct child nodes
276 	 * @param {number} height The new height for the partitions 
277 	 */
278 	adjustActivityPartitionHeight: function(partitions, height) {
279 		(partitions||[]).each(function(partition){
280 			this.setActivityPartitionDimensions(partition, null, height);
281 			this.adjustActivityPartitionHeight(this.getChildActivityPartitions(partition), height-30);
282 		}.bind(this));
283 		return height;
284 	},
285 	
286 	/**
287 	 * Sets recursively the new width of the child partitions 
288 	 * regarding to the partition which was changed
289 	 * @private
290 	 * @param {ORYX.Core.Node[]} activityPartitions Array of all direct childPartitions
291 	 * @param {ORYX.Core.Node} activityPartition The activityPartition which is not changed
292 	 * @param {number} propagateWidth The width 
293 	 * @return {number} The new width for the parent activityPartition
294 	 */
295 	adjustActivityPartitionWidth: function(activityPartitions, changedActivityPartition, propagateWidth){
296 		
297 		// you cannot place oldWidth in the else if part even though the condition is the same, 
298 		// because down there you are inside the inject part at var width
299 		var oldWidth = 0;	
300 		if(!changedActivityPartition && propagateWidth){
301 			oldWidth = activityPartitions.inject(0, function(widthAcc, partition) {
302 				return widthAcc + partition.bounds.width();});
303 		} 
304 		
305 		var width = activityPartitions.inject(0, function (widthAcc, partition) {
306 				
307 			if (partition === changedActivityPartition) {
308 				this.propagateWidthToChildren(partition, widthAcc); //recursive
309 			} else if (!changedActivityPartition && propagateWidth) {
310 				this.setChildWidthProportionallyToTheOldWidth(partition, widthAcc, propagateWidth, oldWidth); //recursive
311 			} else {
312 				this.getWidthFromChildPartitions(partition, widthAcc, propagateWidth, changedActivityPartition); //recursive
313 			}				
314 				return widthAcc + partition.bounds.width();
315 			}.bind(this));
316 		return width;
317 	},
318 	
319 	/**
320 	 * Sets the size of the changed partition and informs the children of the partition
321 	 * that the partition was horizontally resized
322 	 * @private
323 	 * @param {ORYX.Core.Node} partition The partition which needs resizing
324 	 * @param {number} currentWidth The width the partition will receive
325 	 */
326 	propagateWidthToChildren: function(partition, currentWidth){
327 		this.adjustActivityPartitionWidth(this.getChildActivityPartitions(partition), undefined, partition.bounds.width()); //recursion
328 		partition.bounds.set({y:30, x:currentWidth}, {y:partition.bounds.height()+30, x:partition.bounds.width()+currentWidth});
329 	},
330 	
331 	/**
332 	 * Sets the width of the partition (which was actually not resized by the user) 
333 	 * proportionally to the width of its sibling and informs the children
334 	 * of the partition that the partition was horizontally resized
335 	 * @private
336 	 * @param {ORYX.Core.Node} partition The partition which is resized
337 	 * @param {number} offset The horizontal offset this partition receives, since it may have siblings to its left
338 	 * @param {number} propagateWidth The new width the parent partition has
339 	 * @param {number} oldWidth The width the parent partition had before
340 	 */
341 	setChildWidthProportionallyToTheOldWidth: function(partition, offset, propagateWidth, oldWidth){
342 				
343 		var newWidth = (partition.bounds.width() * propagateWidth) / oldWidth;
344 				this.adjustActivityPartitionWidth(this.getChildActivityPartitions(partition), undefined, newWidth); //recursion
345 				this.setActivityPartitionDimensions(partition,newWidth, null);
346 				this.setActivityPartitionPosition(partition, offset);
347 	},
348 	
349 	/**
350 	 * Sets the width of the partition (which was actually not resized by the user) 
351 	 * proportionally to the width of its sibling and informs the children
352 	 * of the partition that the partition was horizontally resized
353 	 * @private
354 	 * @param {ORYX.Core.Node} partition The partition which is resized
355 	 * @param {number} offset The horizontal offset this partition receives, since it may have siblings to its left
356 	 * @param {number} propagateWidth The new recommended width of the partition
357 	 * @param {ORYX.Core.Node} changedActivityPartition The partition the user changed
358 	 */
359 	getWidthFromChildPartitions: function(partition, offset, propagateWidth, changedActivityPartition){
360 				var newWidth = this.adjustActivityPartitionWidth(this.getChildActivityPartitions(partition), changedActivityPartition, propagateWidth);//recursion
361 				
362 				if (!newWidth) {
363 					newWidth = partition.bounds.width();
364 				}
365 				
366 				this.setActivityPartitionDimensions(partition, newWidth, null);
367 				this.setActivityPartitionPosition(partition, offset);
368 	},
369 	
370 	/**
371 	 * Updates the width of all partitions when partitions were added or deleted
372 	 * @private
373 	 * @param {ORYX.Core.Node} rootPartition The partition which is resized 
374 	 */
375 	updateActivityPartitionWidth: function(rootPartition){
376 		var activityPartitions = this.getChildActivityPartitions(rootPartition);
377 		if (activityPartitions.length === 0){
378 			return rootPartition.bounds.width();
379 		}
380 		
381 		var width = activityPartitions.inject(0,function (widthAcc, partition){
382 			this.setActivityPartitionPosition(partition, widthAcc);
383 			return widthAcc + this.updateActivityPartitionWidth(partition);
384 		}.bind(this));
385 		
386 		
387 		this.setActivityPartitionDimensions(rootPartition, width, null);
388 		
389 		return width;
390 	},
391 	
392 	/**
393 	 * Returns the hashed bounds of the shape if they are available
394 	 * otherwise a copy of the shape's bounds.
395 	 * @private
396 	 * @param {ORYX.Core.Node} shape The shape for which there are hashed bounds if it is an activityPartition
397 	 * @return {ORYX.Core.Bounds} The hashed bounds or a copy of the shape's bounds
398 	 */
399 	getHashedBounds: function(shape){
400 		return this.currentRootPartition && this.hashedBounds[this.currentRootPartition.resourceId][shape.resourceId] ? this.hashedBounds[this.currentRootPartition.resourceId][shape.resourceId] : shape.bounds.clone();
401 	},
402 	
403 	/**
404 	 * Returns a sorted set on all child activityPartitions for the given Shape. If recursive is TRUE, also indirect children will be returned (default is FALSE)
405 	 * The set is sorted with the first child the lowest x-coordinate and the last one the highest.
406 	 * @param {ORYX.Core.Node} shape The partition the direct/all children were asked for
407 	 * @param {boolean} recursive Specifies, whether the you receive the direct or all children (default is FALSE = direct children)
408 	 * @return {ORYX.Core.Node[]} An ordered set of the child nodes with the first child the lowest x-coordinate and the last one the highest
409 	 */
410 	getChildActivityPartitions: function(shape, recursive){
411 		var activityPartitions = shape.getChildNodes(recursive||false).findAll(function(node) { 
412 			if (this.isActivityPartitionNode(node)) {
413 				return node;}
414 			}.bind(this));
415 		activityPartitions = activityPartitions.sort(function(a, b){
416 					// Get x coordinate
417 					var ax = Math.round(a.bounds.upperLeft().x);
418 					var bx = Math.round(b.bounds.upperLeft().x);
419 					
420 					// If equal, than use the old one
421 					if (ax == bx) {
422 						ax = Math.round(this.getHashedBounds(a).upperLeft().x);
423 						bx = Math.round(this.getHashedBounds(b).upperLeft().x);
424 					}
425 					return  ax < bx ? -1 : (ax > bx ? 1 : 0);
426 				}.bind(this));
427 		return activityPartitions;
428 	}
