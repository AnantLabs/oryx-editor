 32 /**
 33  * This plugin offers the layout callbacks used in the BPMNplus stencil set.
 34  * 
 35  * @class ORYX.Plugins.BPMNPlus
 36  * @extends Clazz
 37  * @param {Object} facade The editor facade for plugins.
 38  */
 39 ORYX.Plugins.BPMNPlusLayout = {
 40 	/** @lends ORYX.Plugins.BPMNPlusLayout.prototype */
 41 	
 42 	/**
 43 	 *	Constructor
 44 	 *	@param {Object} Facade: The Facade of the Editor
 45 	 */
 46 	construct: function(facade) {
 47 		this.facade = facade;
 48 
 49 		this.facade.registerOnEvent('layout.bpmnplus.pool', this.handleLayoutPool.bind(this));
 50 		this.facade.registerOnEvent('layout.bpmnplus.poolset', this.handleLayoutPoolSet.bind(this));
 51 	},
 52 	
 53 	/**
 54 	 * 'layout.bpmnplus.pool' event handler
 55 	 * 
 56 	 * @param {Object} event
 57 	 * 		The layout event.
 58 	 */
 59 	handleLayoutPool: function(event) {
 60 		var shape = event.shape;
 61 		var lanes = shape.getChildNodes(false).findAll(function(node) {
 62 			return (node.getStencil().id() === "http://b3mn.org/stencilset/bpmnplus#Lane");
 63 		});
 64 		
 65 		if(lanes.length > 0) {
 66 			lanes = lanes.sortBy(function(lane) {
 67 				return lane.bounds.upperLeft().y;
 68 			});
 69 			
 70 			var shapeWidth = shape.bounds.width();
 71 			var shapeHeight = 0;
 72 			lanes.each(function(lane) {
 73 				var ul = lane.bounds.upperLeft();
 74 				var lr = lane.bounds.lowerRight();
 75 				ul.y = shapeHeight;
 76 				lr.y = ul.y + lane.bounds.height();
 77 				shapeHeight += lane.bounds.height();
 78 				ul.x = 30;
 79 				lr.x = shapeWidth;
 80 				lane.bounds.set(ul, lr);
 81 			});
 82 			
 83 			var upl = shape.bounds.upperLeft();
 84 			shape.bounds.set(upl.x, upl.y, shape.bounds.lowerRight().x, upl.y + shapeHeight);
 85 			
 86 			// set label at the middle
 87 			shape.getLabels().each(function(label) {
 88 				label.y = shapeHeight / 2;
 89 				label.x = 12;
 90 			});
 91 		}		
 92 	},
 93 	
 94 	/**
 95 	 * The event handler for the 'layout.bpmnplus.poolset' event
 96 	 * 
 97 	 * @param {Object} event
 98 	 * 		The layout event to handle
 99 	 */
100 	handleLayoutPoolSet: function(event) {
101 		var shape = event.shape;
102 		
103 		var lanes = shape.getChildNodes(false).findAll(function(node) {
104 			return (node.getStencil().id() === "http://b3mn.org/stencilset/bpmnplus#Lane");
105 		});
106 		
107 		if(lanes.length > 0) {
108 		
109 			lanes = lanes.sortBy(function(lane) {
110 				return lane.bounds.upperLeft().y;
111 			});
112 			
113 			var shapeWidth = shape.bounds.width();
114 			var laneWidth = shapeWidth * 500 / 515;
115 			
116 			// calculate height based on the contained lanes
117 			var oldShapeHeight = 0;
118 			lanes.each(function(lane) {
119 				oldShapeHeight += lane.bounds.height();
120 			});
121 			
122 			// calculate new height based on the pool set shade
123 			var newShapeHeight = oldShapeHeight * 315 / 300;					
124 			var diff = newShapeHeight - oldShapeHeight;
125 			
126 			// calculate lane bounds
127 			var shapeHeight = diff;
128 			lanes.each(function(lane) {
129 				var ul = lane.bounds.upperLeft();
130 				var lr = lane.bounds.lowerRight();
131 				ul.y = shapeHeight;
132 				lr.y = ul.y + lane.bounds.height();
133 				shapeHeight += lane.bounds.height();
134 				ul.x = 30;
135 				lr.x = laneWidth;
136 				lane.bounds.set(ul, lr);
137 			});
138 			
139 			// calculate shape bounds
140 			var upl = shape.bounds.upperLeft();
141 			shape.bounds.set(upl.x, upl.y, shape.bounds.lowerRight().x, upl.y + shapeHeight);
142 			
143 			// set labels at the middle
144 			shape.getLabels().each(function(label) {
145 				label.y = shapeHeight / 2;
146 				label.x = 12;
147 			});
148 		}
149 	}
150 	
151 };
152 
153 ORYX.Plugins.BPMNPlusLayout = Clazz.extend(ORYX.Plugins.BPMNPlusLayout);
