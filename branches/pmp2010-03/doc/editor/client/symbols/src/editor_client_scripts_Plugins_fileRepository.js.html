 32 ORYX.Config.SignavioFileRepositoryModelHandler = "/signavio/p/model/"; 
 33 
 34 ORYX.Plugins.FileRepositorySave = Clazz.extend({
 35 	
 36     facade: undefined,
 37 	modelUri: window.location.hash.substring(1).strip() || undefined,
 38 	
 39     construct: function(facade){
 40 		this.facade = facade;
 41 
 42 		this.facade.offer({
 43 			'name': ORYX.I18N.Save.save,
 44 			'functionality': this.save.bind(this, false),
 45 			'group': ORYX.I18N.Save.group,
 46 			'icon': ORYX.PATH + "images/disk.png",
 47 			'description': ORYX.I18N.Save.saveDesc,
 48 			'index': 1,
 49 			'minShape': 0,
 50 			'maxShape': 0
 51 		});
 52 		
 53 		this.facade.offer({
 54 			'name': ORYX.I18N.Save.saveAs,
 55 			'functionality': this.save.bind(this, true),
 56 			'group': ORYX.I18N.Save.group,
 57 			'icon': ORYX.PATH + "images/disk_multi.png",
 58 			'description': ORYX.I18N.Save.saveAsDesc,
 59 			'index': 2,
 60 			'minShape': 0,
 61 			'maxShape': 0
 62 		});
 63 		
 64 
 65 		// ask before closing the window
 66 		this.changeDifference = 0;		
 67 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_UNDO_EXECUTE, function(){ this.changeDifference++; });
 68 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_EXECUTE_COMMANDS, function(){this.changeDifference++; });
 69 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_UNDO_ROLLBACK, function(){this.changeDifference--; });
 70 		
 71 		window.onbeforeunload = function(){
 72 			if (this.changeDifference > 0){
 73 				return ORYX.I18N.Save.unsavedData;
 74 			}
 75 		}.bind(this);
 76 	},
 77 
 78 	saveAs: function(){
 79 		return this.save(true);
 80 	},
 81 	
 82 	save: function(_saveAs) {
 83 		// saveAs requires to save it as a new file
 84 		// save first time requires to save it as a new file
 85 		_saveAs = _saveAs || !this.modelUri;
 86 
 87 		var svgDOM = DataManager.serialize(this.facade.getCanvas().getSVGRepresentation(true));
 88 		var serializedDOM = Ext.encode(this.facade.getJSON());
 89 		
 90 		var modelData = {
 91 			name: "New Process",
 92 			description: "",
 93 			comment: "",
 94 			namespace: this.facade.getStencilSets().values()[0].namespace(),
 95 			json_xml: serializedDOM,
 96 			parent: undefined,
 97 			svg_xml: svgDOM,
 98 			type: this.facade.getStencilSets().values()[0].namespace()
 99 		};
100 		
101 		// try to load available information
102 		if (this.modelUri) {
103 			var request = new Ajax.Request(ORYX.Config.SignavioFileRepositoryModelHandler + this.modelUri.replace(/^\/?/,"") + "/info", {
104 				method: "GET",
105 				asynchronous: false
106 			});
107 			
108 			if (!request.transport || !request.transport.status == 200) {
109 				Ext.Msg.show({
110 				   title: "Unable to load model data",
111 				   msg: "The model does not seem to exist anymore or the model storage is unavailable. Remove the model-id (everything behind #) and try again.",
112 				   buttons: Ext.MessageBox.OK
113 				});
114 				
115 				return false;
116 			}
117 			
118 			var data = request.transport.responseText.evalJSON();
119 			["name", "description", "comment", "parent", "type"].each(function(key) {
120 				modelData[key] = data[key] || modelData[key];
121 			});
122 		}
123 		else {
124 			// model has no URI, so it should at least have parent assigned in the editor URL
125 			var parent = Ext.urlDecode(location.search.substring(1))["directory"];
126 			if (!parent) {
127 				Ext.Msg.show({
128 				   title: "Parent Directory Missing",
129 				   msg: "The model does not semm to be saved yet, however I can't find a parent diractory. Thus the model cannot be saved at all. My bad. :( \n PS: Your best shot is to try to export the model, create a new one, import andh hope for the best.",
130 				   buttons: Ext.MessageBox.OK
131 				});
132 				
133 				return false;
134 			}
135 			modelData.parent = parent;
136 		}
137 		
138 			
139 		if (!_saveAs) {
140 			this.submit(modelData, false);
141 			return true;
142 		}
143 			
144 		var form = new Ext.XTemplate(		
145 			'<form class="oryx_repository_edit_model" action="#" id="edit_model" onsubmit="return false;">',
146 				'<fieldset>',
147 					'<p class="description">' + ORYX.I18N.Save.dialogDesciption + '</p>',
148 					'<input type="hidden" name="namespace" value="{namespace}" />',
149 					'<p><label for="edit_model_title">' + ORYX.I18N.Save.dialogLabelTitle + '</label><input type="text" class="text" name="title" value="{name}" id="edit_model_title" onfocus="this.className = \'text activated\'" onblur="this.className = \'text\'"/></p>',
150 					'<p><label for="edit_model_summary">' + ORYX.I18N.Save.dialogLabelDesc + '</label><textarea rows="5" name="summary" id="edit_model_summary" onfocus="this.className = \'activated\'" onblur="this.className = \'\'">{description}</textarea></p>',
151 					'<p><label for="edit_model_type">' + ORYX.I18N.Save.dialogLabelType + '</label><input type="text" name="type" class="text disabled" value="{type}" disabled="disabled" id="edit_model_type" /></p>',
152 				'</fieldset>',
153 			'</form>'
154 		);
155 
156 		var win = new Ext.Window({
157 			// id:		'Propertie_Window',
158 	        width:	'auto',
159 	        height:	'auto',
160 		    title:	(_saveAs ? ORYX.I18N.Save.saveAsTitle : ORYX.I18N.Save.save),
161 	        modal:	true,
162 			bodyStyle: 'background:#FFFFFF',
163 	        html: 	form.apply(modelData),
164 	        
165 			buttons:[
166 			    {
167 			    	text: ORYX.I18N.Save.saveBtn,
168 			    	handler: function(){
169 			    	
170 			    		modelData.name = $("edit_model_title").value.strip(); 
171 			    		modelData.description = $("edit_model_summary").value.strip()
172 
173 			    		this.submit(modelData, true);
174 						win.close();
175 						
176 					}.bind(this)
177 			    },{
178 			    	text: ORYX.I18N.Save.close,
179 			    	handler: function(){
180 			    		win.close();
181             		}
182 			    }
183 			],
184 			listeners: {
185 				close: function(){					
186                 	win.close();
187 				}
188 			}
189 	    });
190 	
191 		win.show();
192 		
193 		return true;
194 	
195 	},
196 	
197 	submit: function(params, _saveAs) {
198 		
199 		this.facade.raiseEvent({
200             type: ORYX.CONFIG.EVENT_LOADING_ENABLE,
201 			text: ORYX.I18N.Save.saving
202         });
203 		
204 		if (_saveAs) {
205 			new Ajax.Request(ORYX.Config.SignavioFileRepositoryModelHandler, {
206 				method: "POST",
207 				parameters: params,
208 				asynchronous: true,
209 				onSuccess: function success(transport){
210 					this.modelUri = transport.responseText.evalJSON()["href"].replace(/^\/?(model)?\/?/,"");
211 					location.hash =  this.modelUri;
212 					this.changeDifference = 0;
213 					
214 					this.facade.raiseEvent({
215 						type:ORYX.CONFIG.EVENT_LOADING_STATUS,
216 						text:ORYX.I18N.Save.saved
217 					});
218 					
219 					
220 					
221 				}.bind(this),
222 				onFailure: function failure(){
223 					alert("fail");
224 				}.bind(this),
225 				on403: function f403(){
226 					alert("403");
227 				}.bind(this)
228 			});
229 		}
230 		else {			
231 			new Ajax.Request(ORYX.Config.SignavioFileRepositoryModelHandler + this.modelUri, {
232 				method: "PUT",
233 				parameters: params,
234 				asynchronous: true,
235 				onSuccess: function success(transport){
236 					this.changeDifference = 0;					
237 					this.facade.raiseEvent({
238 						type:ORYX.CONFIG.EVENT_LOADING_STATUS,
239 						text:ORYX.I18N.Save.saved
240 					});
241 					
242 				}.bind(this),
243 				onFailure: function failure(){
244 					alert("fail");
245 				}.bind(this),
246 				on403: function f403(){
247 					alert("403");
248 				}.bind(this)
249 			});
250 		}
251 	}
252 });
253 
254 onOryxResourcesLoaded = function () {
255 	if (location.hash.slice(1).length == 0 || location.hash.slice(1).indexOf('new')!=-1) {
256 		var stencilset = ORYX.Utils.getParamFromUrl('stencilset') 
257 			? ORYX.Utils.getParamFromUrl('stencilset')
258 			: 'stencilsets/bpmn1.1/bpmn1.1.json';
259 
260 		new ORYX.Editor({
261 			id: 'oryx-canvas123',
262 			stencilset: {
263 				url: '/oryx/'+stencilset
264 			}
265 		});
266 	}
267 	else {
268 		ORYX.Editor.createByUrl(ORYX.Config.SignavioFileRepositoryModelHandler + location.hash.slice(1).replace(/^\/?/,"") + "/json", {
269 			id: 'oryx-canvas123'
270 		});
271 	};
272 };
