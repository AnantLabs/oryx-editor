187 	renderer: function(value, p, record) {
188 		
189 		this.tooltipRenderer(value, p, record);
190 				
191 		if(value instanceof Date) {
192 			// TODO: Date-Schema is not generic
193 			value = value.dateFormat(ORYX.I18N.PropertyWindow.dateFormat);
194 		} else if(String(value).search("<a href='") < 0) {
195 			// Shows the Value in the Grid in each Line
196 			value = String(value).gsub("<", "<");
197 			value = String(value).gsub(">", ">");
198 			value = String(value).gsub("%", "%");
199 			value = String(value).gsub("&", "&");
200 
201 			if(record.data.gridProperties.type == ORYX.CONFIG.TYPE_COLOR) {
202 				value = "<div class='prop-background-color' style='background-color:" + value + "' />";
203 			}			
204 
205 			record.data.icons.each(function(each) {
206 				if(each.name == value) {
207 					if(each.icon) {
208 						value = "<img src='" + each.icon + "' /> " + value;
209 					}
210 				}
211 			});
212 		}
213 
214 		return value;
215 	},

217 	beforeEdit: function(option) {
218 
219 		var editorGrid 		= this.dataSource.getAt(option.row).data.gridProperties.editor;
220 		var editorRenderer 	= this.dataSource.getAt(option.row).data.gridProperties.renderer;
221 
222 		if(editorGrid) {
223 			// Disable KeyDown
224 			this.facade.disableEvent(ORYX.CONFIG.EVENT_KEYDOWN);
225 
226 			option.grid.getColumnModel().setEditor(1, editorGrid);
227 			
228 			editorGrid.field.row = option.row;
229 			// Render the editor to the grid, therefore the editor is also available 
230 			// for the first and last row
231 			editorGrid.render(this.grid);
232 			
233 			//option.grid.getColumnModel().setRenderer(1, editorRenderer);
234 			editorGrid.setSize(option.grid.getColumnModel().getColumnWidth(1), editorGrid.height);
235 		} else {
236 			return false;
237 		}
238 		
239 		var key = this.dataSource.getAt(option.row).data.gridProperties.propId;
240 		
241 		this.oldValues = new Hash();
242 		this.shapeSelection.shapes.each(function(shape){
243 			this.oldValues[shape.getId()] = shape.properties[key];
244 		}.bind(this));

247 	afterEdit: function(option) {
248 		//Ext1.0: option.grid.getDataSource().commitChanges();
249 		option.grid.getStore().commitChanges();
250 
251 		var key 			 = option.record.data.gridProperties.propId;
252 		var selectedElements = this.shapeSelection.shapes;
253 		
254 		var oldValues 	= this.oldValues;	
255 		
256 		var newValue	= option.value;
257 		var facade		= this.facade;

260 		// Implement the specific command for property change
261 		var commandClass = ORYX.Core.Command.extend({
262 			construct: function(){
263 				this.key 		= key;
264 				this.selectedElements = selectedElements;
265 				this.oldValues = oldValues;
266 				this.newValue 	= newValue;
267 				this.facade		= facade;
268 			},			
269 			execute: function(){
270 				this.selectedElements.each(function(shape){
271 					if(!shape.getStencil().property(this.key).readonly()) {
272 						shape.setProperty(this.key, this.newValue);
273 					}
274 				}.bind(this));
275 				this.facade.setSelection(this.selectedElements);
276 				this.facade.getCanvas().update();
277 				this.facade.updateSelection();
278 			},
279 			rollback: function(){
280 				this.selectedElements.each(function(shape){
281 					shape.setProperty(this.key, this.oldValues[shape.getId()]);
282 				}.bind(this));
283 				this.facade.setSelection(this.selectedElements);
284 				this.facade.getCanvas().update();
285 				this.facade.updateSelection();
286 			}
287 		})		
288 		// Instanciated the class
289 		var command = new commandClass();
290 		
291 		// Execute the command
292 		this.facade.executeCommands([command]);

371 	/**
372 	 * Sets this.shapeSelection.commonPropertiesValues.
373 	 * If the value for a common property is not equal for each shape the value
374 	 * is left empty in the property window.
375 	 */
376 	setCommonPropertiesValues: function() {
377 		this.shapeSelection.commonPropertiesValues = new Hash();
378 		this.shapeSelection.commonProperties.each(function(property){
379 			var key = property.prefix() + "-" + property.id();
380 			var emptyValue = false;
381 			var firstShape = this.shapeSelection.shapes.first();
382 			
383 			this.shapeSelection.shapes.each(function(shape){
384 				if(firstShape.properties[key] != shape.properties[key]) {
385 					emptyValue = true;
386 				}
387 			}.bind(this));
388 			
389 			/* Set property value */
390 			if(!emptyValue) {
391 				this.shapeSelection.commonPropertiesValues[key]
392 					= firstShape.properties[key];
393 			}
394 		}.bind(this));
395 	},

397 	/**
398 	 * Returns the set of stencils used by the passed shapes.

400 	getStencilSetOfSelection: function() {
401 		var stencils = new Hash();
402 		
403 		this.shapeSelection.shapes.each(function(shape) {

405 		})
406 		return stencils;

409 	/**
410 	 * Identifies the common Properties of the selected shapes.
411 	 */
412 	identifyCommonProperties: function() {
413 		this.shapeSelection.commonProperties.clear();
414 		
415 		/* 
416 		 * A common property is a property, that is part of 
417 		 * the stencil definition of the first and all other stencils.
418 		 */
419 		var stencils = this.getStencilSetOfSelection();
420 		var firstStencil = stencils.values().first();
421 		var comparingStencils = stencils.values().without(firstStencil);
422 		
423 		
424 		if(comparingStencils.length == 0) {
425 			this.shapeSelection.commonProperties = firstStencil.properties();
426 		} else {
427 			var properties = new Hash();
428 			
429 			/* put all properties of on stencil in a Hash */
430 			firstStencil.properties().each(function(property){
431 				properties[property.namespace() + '-' + property.id() 
432 							+ '-' + property.type()] = property;
433 			});
434 			
435 			/* Calculate intersection of properties. */
436 			
437 			comparingStencils.each(function(stencil){
438 				var intersection = new Hash();
439 				stencil.properties().each(function(property){
440 					if(properties[property.namespace() + '-' + property.id()
441 									+ '-' + property.type()]){
442 						intersection[property.namespace() + '-' + property.id()
443 										+ '-' + property.type()] = property;
444 					}
445 				});
446 				properties = intersection;	
447 			});
448 			
449 			this.shapeSelection.commonProperties = properties.values();
450 		}

481 	 * selected shapes.
482 	 */
483 	createProperties: function() {
484 		this.properties = [];
485 		this.popularProperties = [];
486 
487 		if(this.shapeSelection.commonProperties) {
488 			
489 			// add new property lines
490 			this.shapeSelection.commonProperties.each((function(pair, index) {
491 
492 				var key = pair.prefix() + "-" + pair.id();
493 				
494 				// Get the property pair
495 				var name		= pair.title();
496 				var icons		= [];
497 				var attribute	= this.shapeSelection.commonPropertiesValues[key];
498 				
499 				var editorGrid = undefined;
500 				var editorRenderer = null;
501 				
502 				var refToViewFlag = false;
503 
504 				if(!pair.readonly()){
505 					switch(pair.type()) {
506 						case ORYX.CONFIG.TYPE_STRING:
507 							// If the Text is MultiLine
508 							if(pair.wrapLines()) {
509 								// Set the Editor as TextArea
510 								var editorTextArea = new Ext.form.TextArea({alignment: "tl-tl", allowBlank: pair.optional(),  msgTarget:'title', maxLength:pair.length()});
511 								editorTextArea.on('keyup', function(textArea, event) {
512 									this.editDirectly(key, textArea.getValue());
513 								}.bind(this));								
514 								
515 								editorGrid = new Ext.Editor(editorTextArea);
516 							} else {
517 								// If not, set the Editor as InputField
518 								var editorInput = new Ext.form.TextField({allowBlank: pair.optional(),  msgTarget:'title', maxLength:pair.length()});
519 								editorInput.on('keyup', function(input, event) {
520 									this.editDirectly(key, input.getValue());
521 								}.bind(this));
522 								
523 								// reverts the shape if the editor field is invalid
524 								editorInput.on('blur', function(input) {
525 									if(!input.isValid(false))
526 										this.updateAfterInvalid(key);
527 								}.bind(this));
528 								
529 								editorInput.on("specialkey", function(input, e) {
530 									if(!input.isValid(false))
531 										this.updateAfterInvalid(key);
532 								}.bind(this));
533 								
534 								editorGrid = new Ext.Editor(editorInput);
535 							}
536 							break;
537 						case ORYX.CONFIG.TYPE_BOOLEAN:
538 							// Set the Editor as a CheckBox
539 							var editorCheckbox = new Ext.form.Checkbox();
540 							editorCheckbox.on('check', function(c,checked) {
541 								this.editDirectly(key, checked);
542 							}.bind(this));
543 							
544 							editorGrid = new Ext.Editor(editorCheckbox);
545 							break;
546 						case ORYX.CONFIG.TYPE_INTEGER:
547 							// Set as an Editor for Integers
548 							var numberField = new Ext.form.NumberField({allowBlank: pair.optional(), allowDecimals:false, msgTarget:'title', minValue: pair.min(), maxValue: pair.max()});
549 							numberField.on('keyup', function(input, event) {
550 								this.editDirectly(key, input.getValue());
551 							}.bind(this));							
552 							
553 							editorGrid = new Ext.Editor(numberField);
554 							break;
555 						case ORYX.CONFIG.TYPE_FLOAT:
556 							// Set as an Editor for Float
557 							var numberField = new Ext.form.NumberField({ allowBlank: pair.optional(), allowDecimals:true, msgTarget:'title', minValue: pair.min(), maxValue: pair.max()});
558 							numberField.on('keyup', function(input, event) {
559 								this.editDirectly(key, input.getValue());
560 							}.bind(this));
561 							
562 							editorGrid = new Ext.Editor(numberField);
563 
564 							break;
565 						case ORYX.CONFIG.TYPE_COLOR:
566 							// Set as a ColorPicker
567 							// Ext1.0 editorGrid = new gEdit(new form.ColorField({ allowBlank: pair.optional(),  msgTarget:'title' }));
568 
569 							var editorPicker = new Ext.ux.ColorField({ allowBlank: pair.optional(),  msgTarget:'title', facade: this.facade });
570 							
571 							/*this.facade.registerOnEvent(ORYX.CONFIG.EVENT_COLOR_CHANGE, function(option) {
572 								this.editDirectly(key, option.value);
573 							}.bind(this));*/
574 							
575 							editorGrid = new Ext.Editor(editorPicker);
576 
577 							break;
578 						case ORYX.CONFIG.TYPE_CHOICE:
579 							var items = pair.items();
580 													
581 							var options = [];
582 							items.each(function(value) {
583 								if(value.value() == attribute)
584 									attribute = value.title();
585 									
586 								if(value.refToView()[0])
587 									refToViewFlag = true;
588 																
589 								options.push([value.icon(), value.title(), value.value()]);
590 															
591 								icons.push({
592 									name: value.title(),
593 									icon: value.icon()
594 								});
595 							});
596 							
597 							var store = new Ext.data.SimpleStore({
598 						        fields: [{name: 'icon'},
599 									{name: 'title'},
600 									{name: 'value'}	],
601 						        data : options // from states.js
602 						    });
603 							
604 							// Set the grid Editor
605 
606 						    var editorCombo = new Ext.form.ComboBox({
607 								tpl: '<tpl for="."><div class="x-combo-list-item">{[(values.icon) ? "<img src=\'" + values.icon + "\' />" : ""]} {title}</div></tpl>',
608 						        store: store,
609 						        displayField:'title',
610 								valueField: 'value',
611 						        typeAhead: true,
612 						        mode: 'local',
613 						        triggerAction: 'all',
614 						        selectOnFocus:true
615 						    });
616 								
617 							editorCombo.on('select', function(combo, record, index) {
618 								this.editDirectly(key, combo.getValue());
619 							}.bind(this))
620 							
621 							editorGrid = new Ext.Editor(editorCombo);
622 
623 							break;
624 						case ORYX.CONFIG.TYPE_DATE:
