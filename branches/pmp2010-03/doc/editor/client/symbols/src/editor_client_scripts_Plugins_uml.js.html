 31 /**
 32  * The UML plugin provides layout methods referring to the UML stencilset. 
 33  * 
 34  * @class ORYX.Plugins.UML
 35  * @extends Clazz
 36  * @param {Object} facade The facade of the editor
 37  */
 38 ORYX.Plugins.UML = 
 39 /** @lends ORYX.Plugins.UML.prototype */
 40 {
 41 	/**
 42 	 * Creates a new instance of the UML plugin and registers it on the
 43 	 * layout events listed in the UML stencil set.
 44 	 * 
 45 	 * @constructor
 46 	 * @param {Object} facade The facade of the editor
 47 	 */
 48 	construct: function(facade) {
 49 		this.facade = facade;
 50 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_PROPWINDOW_PROP_CHANGED, this.handlePropertyChanged.bind(this));
 51 		this.facade.registerOnEvent('layout.uml.class', this.handleLayoutClass.bind(this));
 52 		this.facade.registerOnEvent('layout.uml.list', this.handleLayoutList.bind(this));
 53 		this.facade.registerOnEvent('layout.uml.association', this.handleLayoutAssociation.bind(this));
 54 		this.facade.registerOnEvent('layout.uml.qualified_association', this.handleLayoutQualifiedAssociation.bind(this));
 55 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_LOADED, this.addReadingDirectionOnLoad.bind(this));
 56 
 57 	},
 58 	
 59 	/**
 60 	 * Add reading direction on load.
 61 	 *
 62 	 * Because the reading direction arrow is added to the
 63 	 * label of an edge, we have to iterate over all edges
 64 	 * and add the arrow on load for it to appear.
 65 	 *
 66 	 * @param {Object} event
 67 	 */
 68 	addReadingDirectionOnLoad : function(event) {
 69 		this.facade.getCanvas().edges.each(function(shape){
 70 			if (shape.properties["oryx-direction"] == "left" || shape.properties["oryx-direction"] == "right") {
 71 				this.addReadingDirection(shape);
 72 			}
 73 		}.bind(this));
 74 	},
 75 	
 76 	/**
 77 	 * calculates the height of a text, taking line breaks into consideration
 78 	 *
 79 	 * @param {Object} labelElement the label
 80 	 * @param {String} the label test
 81 	 */
 82 	
 83 	calculateLabelHeight : function (labelElement, labelValue) {
 84 		var fontSize = labelElement.getFontSize();
 85 		var newlineOccurences = 1;
 86 		
 87 		labelValue.scan('\n', function() { newlineOccurences += 1; });
 88 		
 89 		// 0.75 account for padding around the label
 90 		return newlineOccurences * fontSize + 0.75;
 91 	},
 92 	
 93 	/**
 94 	 * Add Reading Direction to the name label after it has been changed
 95 	 */
 96 	handlePropertyChanged : function(event) {
 97 		if (event["key"] == "oryx-name") {
 98 			this.addReadingDirection(event.elements[0]);
 99 		}
100 	},
101 	
102 	/**
103 	 * Layout class shapes.
104 	 *   - make text italic when abstract
105 	 *   - resize methods and attributes boxes according to their content
106 	 */
107 	handleLayoutClass : function(event) {
108 		var shape = event.shape;
109 		
110 		if (shape.propertiesChanged["oryx-abstract"] == true) {
111 			var className = event.shape.getLabels().find(
112 					function(label) { return label.id == (event.shape.id + "className") }
113 				);
114 			
115 			if (shape.properties["oryx-abstract"] == true) {
116 				className.node.setAttribute("font-style", "italic");
117 			} else {
118 				className.node.setAttribute("font-style", "none");
119 			}
120 		}
121 		
122 		if (shape.propertiesChanged["oryx-attributes"] == true || shape.propertiesChanged["oryx-methods"]) {
123 			var attributesValue = event.shape.properties["oryx-attributes"];
124 			var methodsValue = event.shape.properties["oryx-methods"];
125 			var attributes = event.shape.getLabels().find(
126 					function(label) { return label.id == (event.shape.id + "attributes") }
127 				);
128 			var methods = event.shape.getLabels().find(
129 					function(label) { return label.id == (event.shape.id + "methods") }
130 				);
131 			var separator = event.shape._svgShapes.find(
132 					function(element) { return element.element.id == (event.shape.id + "separator") }
133 				).element;
134 					
135 					
136 			var attributesHeight = this.calculateLabelHeight(attributes, attributesValue);
137 			var methodsHeight = this.calculateLabelHeight(methods, methodsValue);
138 			
139 			// 24px account for the class name
140 			var distanceTilSeparator = 24 + attributesHeight + 2;
141 			var distanceTilBottom = distanceTilSeparator + methodsHeight + 2;
142 			
143 			separator.setAttribute("y1", distanceTilSeparator);
144 			separator.setAttribute("y2", distanceTilSeparator);
145 			
146 			// realign methods label (so that oryx' internal references are correct)
147 			methods.y = distanceTilSeparator + 3;
148 			methods.node.setAttribute("y", distanceTilSeparator + 3);
149 			// realign the methods line by line (required for a visual change)
150 			for (var i = 0; i < methods.node.childElementCount; i++) {
151 				methods.node.childNodes[i].setAttribute("y", distanceTilSeparator + 2);
152 			}
153 			
154 			// resize shape
155 			shape.bounds.set(
156 				shape.bounds.a.x, 
157 				shape.bounds.a.y, 
158 				shape.bounds.b.x, 
159 				shape.bounds.a.y + distanceTilBottom + 5
160 			);
161 		}
162 	},
163 	
164 	/**
165 	 * Layout the interface and enumeration shape. Resize according to their content.
166 	 */
167 	handleLayoutList : function(event) {
168 		var shape = event.shape;
169 		
170 		if (shape.propertiesChanged["oryx-items"] == true) {
171 			var itemsValue = shape.properties["oryx-items"];
172 			var items = shape.getLabels().find(
173 					function(label) { return label.id == (event.shape.id + "items") }
174 				);
175 			
176 			var itemsHeight = this.calculateLabelHeight(items, itemsValue);
177 		
178 			var distanceTilBottom = 32 + itemsHeight + 2;
179 			
180 			// resize shape
181 			shape.bounds.set(
182 				shape.bounds.a.x, 
183 				shape.bounds.a.y, 
184 				shape.bounds.b.x, 
185 				shape.bounds.a.y + distanceTilBottom + 5
186 			);
187 		}
188 	},
189 	
190 	/**
191 	 * Draws the reading direction arrow when an association is changed.
192 	 */
193 	handleLayoutAssociation : function(event) {
194 		this.addReadingDirection(event.shape);
195 	},
196 	
197 	/**
198 	 * Adds the reading direction to the "name" label of an association.
199 	 */ 
200 	addReadingDirection : function(shape) {
201 		var name = shape.getLabels().find(
202 					function(label) { return label.id == (shape.id + "name") }
203 				);
204 		
205 		if (shape.properties["oryx-direction"] == "left") {
206 			name.text("◀ " + shape.properties["oryx-name"]);
207 		}
208 		else if (shape.properties["oryx-direction"] == "right") {
209 			name.text(shape.properties["oryx-name"] + " ▶");
210 		}
211 		else {
212 			name.text(shape.properties["oryx-name"]);
213 		}
214 		
215 		name.update();
216 	},
217 	
218 	
219 	/**
220 	 * Resizes the qualifier box of a qualified association according to its content.
221 	 */
222 	handleLayoutQualifiedAssociation : function(event) {
223 		var shape = event.shape;
224 		var qualifier = shape.getLabels().find(
225 					function(label) { return label.id == (event.shape.id + "qualifier") }
226 				);
227 		
228 		var size = qualifier._estimateTextWidth(shape.properties["oryx-qualifier"], 12);
229 		// enforce minimum size, looks bad otherwise
230 		if (size < 40) size = 40;
231 		shape._markers.values()[0].element.lastElementChild.setAttribute("width", size+5);
232 		shape._markers.values()[0].element.setAttributeNS(null, "markerWidth", size+5)
233 	},
234 };

236 ORYX.Plugins.UML = Clazz.extend(ORYX.Plugins.UML);
