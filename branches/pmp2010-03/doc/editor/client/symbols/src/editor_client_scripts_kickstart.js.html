 16 //TODO kann kickstart sich vielleicht auch um die erzeugung von paketen/
 17 // namespaces kï¿½mmern? z.b. requireNamespace("ORYX.Core.SVG");
 18 var Kickstart = {
 19  	started: false,
 20 	callbacks: [],
 21 	alreadyLoaded: [],
 22 	PATH: '',
 23 
 24 	load: function() { Kickstart.kick(); },
 25 
 26 	kick: function() {
 27 		//console.profile("loading");
 28 		if(!Kickstart.started) {
 29 			Kickstart.started = true;
 30 			Kickstart.callbacks.each(function(callback){
 31 				// call the registered callback asynchronously.
 32 				window.setTimeout(callback, 1);
 33 			});
 34 		}
 35 	},
 36 
 37 	register: function(callback) {
 38 		//TODO Add some mutual exclusion between kick and register calls.
 39 		with(Kickstart) {
 40 			if(started) window.setTimeout(callback, 1);
 41 			else Kickstart.callbacks.push(callback)
 42 		}
 43 	},
 44 
 45 	/**
 46 	 * Loads a js, assuring that it has only been downloaded once.
 47 	 * @param {String} url the script to load.
 48 	 */
 49 	require: function(url) {
 50 		// if not already loaded, include it.
 51 		if(Kickstart.alreadyLoaded.member(url))
 52 			return false;
 53 		return Kickstart.include(url);
 54 	},
 55 
 56 	/**
 57 	 * Loads a js, regardless of whether it has only been already downloaded.
 58 	 * @param {String} url the script to load.
 59 	 */
 60 	include: function(url) {
 61 
 62 		// prepare a script tag and place it in html head.
 63 		var head = document.getElementsByTagNameNS(XMLNS.XHTML, 'head')[0];
 64 		var s = document.createElementNS(XMLNS.XHTML, "script");
 65 		s.setAttributeNS(XMLNS.XHTML, 'type', 'text/javascript');
 66 	   	s.src = Kickstart.PATH + url;
 67 
 68 		//TODO macht es sinn, dass neue skript als letztes kind in den head
 69 		// einzubinden (stichwort reihenfolge der skript tags)?
 70 	   	head.appendChild(s);
 71 
 72 		// remember this url.
 73 		Kickstart.alreadyLoaded.push(url);
 74 
 75 		return true;
 76 	}
 77 }
 78 
 79 // register kickstart as the new onload event listener on current window.
 80 // previous listener(s) are triggered to launch with kickstart.
 81 Event.observe(window, 'load', Kickstart.load);
