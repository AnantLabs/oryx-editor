208 	issueQuery : function(options){
209 
210 		try {
211 			var serialized_rdf = this.getRDFFromDOM();
212 //			serialized_rdf = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>" + serialized_rdf;
213 
214 			this.facade.raiseEvent({
215 	            type: ORYX.CONFIG.EVENT_LOADING_ENABLE,
216 				text: "Processing query"  //ORYX.I18N.Save.saving
217 	        });
218 			// Send the request to the server.
219 			new Ajax.Request(ORYX.CONFIG.QUERYEVAL_URL, {
220 				method: 'POST',
221 				asynchronous: true,
222 				parameters: {
223 					resource	: location.href,
224 					command		: options.command,
225 					modelID		: options.modelID,
226 					stopAtFirstMatch: options.stopAtFirstMatch,
227 					data		: serialized_rdf
228 				},
229                 onSuccess: function(response){
230                     this.facade.raiseEvent({
231 						type:ORYX.CONFIG.EVENT_LOADING_DISABLE
232 					});
233 					
234 					var respXML = response.responseXML;
235                     var root = respXML.firstChild;
236                     var processList = new Array();
237                     var nodecnt, graph;
238                     var pchildren = root.getElementsByTagName("ProcessGraph");
239                     
240                     // puts all matching process models into array processList with model ID 
241                     // and model elements
242 					for (nodecnt = 0; nodecnt < pchildren.length; nodecnt++) {
243                         graph = pchildren.item(nodecnt);
244                         var graphID = graph.getAttributeNode("modelID").nodeValue;
245                         
246                         processList.push({
247 							id 			: graphID,
248 							elements 	: this.processResultGraph(graph),
249 							metadata    : '',
250 							description	: this.processMatchDescription(graph) //Ahmed Awad on 28.07.09
251 						});
252                         
253                     }
254 					try {
255 						this.processProcessList(processList);
256 					} catch (error) {
257 						Ext.Msg.alert(ORYX.I18N.Oryx.title, error);
258 					}
259                 }.bind(this),
260 				
261 				onFailure: function(response){
262 					this.facade.raiseEvent({
263 						type:ORYX.CONFIG.EVENT_LOADING_DISABLE
264 					});
265 					Ext.Msg.alert(ORYX.I18N.Oryx.title, "Server encountered an error (" + response.statusText + ").\n"
266 						+ response.responseText);
267 				}.bind(this)
268 			});
269 			
270 		} catch (error){
271 			this.facade.raiseEvent({type:ORYX.CONFIG.EVENT_LOADING_DISABLE});
272 			Ext.Msg.alert(ORYX.I18N.Oryx.title, error);
273 	 	}

277     processResultGraph: function(xmlNode){
278         var graphElements = new Array();
279 		
280 		for (var k = 0; k < xmlNode.childNodes.length; k++) {
281             var node = xmlNode.childNodes.item(k);
282             if (!(node instanceof Text)) {
283                 if (node.hasAttribute("id")) { // it is a node
284 					graphElements.push({
285 						nodeType : node.nodeName,
286 						nodeId : node.getAttributeNode("id").nodeValue
287 					});
288 					
289 				} else if ((node.hasAttribute("from"))
290 						&& node.hasAttribute("to")) { // it is an edge
291 					graphElements.push({
292 						edgeType : node.nodeName,
293 						from : node.getAttributeNode("from").nodeValue,
294 						to : node.getAttributeNode("to").nodeValue
295 					});
296 				}
297             }
298         }
299 		return graphElements;
300     },
301 	// Added by Ahmed Awad on 28.07.09 to extract the diagnosis and the match meta data
302 	processMatchDescription: function(xmlNode){
303         var metadata = new Array();
304 		
305 		for (var k = 0; k < xmlNode.childNodes.length; k++) {
306             var node = xmlNode.childNodes.item(k);
307             if ((node.nodeName === "diagnosis")) {
308                 
309 					metadata.push({
310 						diagnosis : node.textContent
311 					});
312 					
313 				} else if ((node.nodeName === "match")) { // it is an edge
314 					metadata.push({
315 						match : node.textContent
316 					});
317 				}
318             
319         }
320 		return metadata;
321     },

408 	getModelMetaData : function(processEntry) {
409 		var metaUri = processEntry.id.replace(/\/rdf$/, '/meta');
410 		new Ajax.Request(metaUri, 
411 			 {
412 				method			: "get",
413 				asynchronous 	: false,
414 				onSuccess		: function(transport) {
415 					processEntry.metadata = transport.responseText.evalJSON();
416 				}.bind(this),
417 				onFailure		: function() {
418 					Ext.MessageBox.alert(ORYX.I18N.Oryx.title, "Error loading model meta data.");
419 				}.bind(this)

424 	_onDblClick: function(dataGrid, index, node, e){
425 		
426 		// Select the new range
427 		dataGrid.selectRange(index, index);
428 
429 		// Get uri and matched element data from the clicked model
430 		var modelId 	= dataGrid.getRecord( node ).data.id;
431 		var matchedElements = dataGrid.getRecord( node ).data.elements;
432 		var description = dataGrid.getRecord( node ).data.description; // Added by Ahmed Awad on 30.07.09
433 		// convert object to JSOn representation
434 		var elementsAsJson = Ext.encode(matchedElements);
435 		var descriptionAsJson = Ext.encode(description); // Added by Ahmed Awad on 30.07.09
436 		// escape JSON string to become URI-compliant
437 		var encodedJson = encodeURIComponent(elementsAsJson);
438 		var encodedDescription = encodeURIComponent(descriptionAsJson);
439 		
440 		// remove the last URI segment, append editor's 'self' and json of model elements
441 		var slashPos = modelId.lastIndexOf("/");
442 		//var uri	= modelId.substr(0, slashPos) + "/self" + "?matches=" + encodedJson;
443  	    var uri	= modelId.substr(0, slashPos) + "/self" + "?matches=" + encodedJson+"&description="+encodedDescription;
444 		// Open the model in Editor
445 		var editor = window.open( uri );
446 		window.setTimeout(
447 	        function() {
448                 if(!editor || !editor.opener || editor.closed) {
449                         Ext.MessageBox.alert(ORYX.I18N.Oryx.title, ORYX.I18N.Oryx.editorOpenTimeout).setIcon(Ext.MessageBox.QUESTION);
450                 }
451 	        }, 5000);
