121 	/**
122 	 * Opens a dialog that presents the results of a transformation.
123 	 * The dialog shows a list containing the resulting XML files.
124 	 * Each file can be shown in a new window or downloaded.
125      *
126 	 * @param {Object} data The data to be shown in the dialog
127 	 * Format: array with three elements: 
128 	 *   * file - the file
129 	 *   * result - the content of file, may also be an error message.
130 	 *   * info - status of the result: "success" or "error"
131 	 */
132 	openResultDialog: function(data) {
133 
134 		var ds = new Ext.data.Store({
135 	        proxy: new Ext.data.MemoryProxy(data),
136 	        reader: new Ext.data.ArrayReader({}, [
137 	               {name: 'file', type: 'string'},
138 	               {name: 'result', type: 'string'},
139 	               {name: 'info', type: 'string'}
140 	        	])
141 		});
142 		
143 		ds.load();
144 
145 		// renderer
146 		var infoRenderer = function (val){
147             if(val == "success"){
148                 return '<span style="color:green;">' + val + '</span>';
149             }else if(val == "error"){
150                 return '<span style="color:red;">' + val + '</span>';
151             }
152             return val;
153         };
154 	
155 		var cm = new Ext.grid.ColumnModel([
156 		    {id:'file',header: "File", width: 200, sortable: false, dataIndex: 'file', resizable: false},
157 		    {header: "Info", width: 75, sortable: false, dataIndex: 'info', renderer: infoRenderer, resizable: false} 
158 		]);
159 				
160 		var grid = new Ext.grid.GridPanel({
161 			store:ds,
162 	        cm: cm,
163 	        sm: new Ext.grid.RowSelectionModel({ 	singleSelect:true }),
164 			autoWidth: true
165 	    });
166 		
167 	    var toolbar = new Ext.Toolbar();
168 		
169 		var dialog = new Ext.Window({ 
170 			autoCreate: true, 
171 			title: ORYX.I18N.TransformationDownloadDialog.transResult, 
172 			autoHeight: true, 
173 			width: 297, 
174 			modal:true,
175 			collapsible:false,
176 			fixedcenter: true, 
177 			shadow:true, 
178 			proxyDrag: true,
179 			resizable:false,
180 			items:[toolbar, grid]
181 		});
182 	
183 		dialog.on('hide', function(){
184 			dialog.destroy(true);
185 			grid.destroy(true);
186 			delete dialog;
187 			delete grid;
188 		});
189 		dialog.show();
190 		
191 		toolbar.add({
192 			icon: 'images/view.png', // icons can also be specified inline
193 	        cls: 'x-btn-icon',
194     	    tooltip: ORYX.I18N.TransformationDownloadDialog.showFile,
195 			handler: function() {
196 				var ds = grid.getStore();
197 				var selection = grid.getSelectionModel().getSelected();
198 				if (selection == undefined) {
199 					return;
200 				}
201 				var show = selection.get("result");
202 				if (selection.get("info") == "success") {
203 					this.openXMLWindow(show);
204 				} else {
205 					this.openErrorWindow(show);
206 				}
207 			}.bind(this)
208 		});
209 		toolbar.add({
210 			icon: 'images/disk.png', // icons can also be specified inline
211 	        cls: 'x-btn-icon',
212     	    tooltip: ORYX.I18N.TransformationDownloadDialog.downloadFile,
213 			handler: function() {
214 				var ds = grid.getStore();
215 				var selection = grid.getSelectionModel().getSelected();
216 				if (selection == undefined) {
217 					return;
218 				}
219 				this.openDownloadWindow(selection, false);
220 			}.bind(this)
221 		});
222 		toolbar.add({
223 			icon: 'images/disk_multi.png', // icons can also be specified inline
224 	        cls: 'x-btn-icon',
225     	    tooltip: ORYX.I18N.TransformationDownloadDialog.downloadAll,
226 			handler: function() {
227 				var ds = grid.getStore();				
228 				this.openDownloadWindow(ds.getRange(0, ds.getCount()), true);
229 			}.bind(this)
230 		});			
231 		
232 		// Select the first row
233 		grid.getSelectionModel().selectFirstRow();
234 	},

267 	/**
268 	 * Creates a hidden form element to communicate parameter values
269 	 * to a php file.
270 	 * 
271 	 * @param {Object} name  The name of the hidden field
272 	 * @param {Object} value The value of the hidden field
273 	 */
274 	createHiddenElement: function(name, value) {
275 		var newElement = document.createElement("input");
276 		newElement.name=name;
277 		newElement.type="hidden";
278 		newElement.value = value;
279 		return newElement
280 	},

283 	/**
284 	 * Adds a file extension to the given file name. If the file
285 	 * has the name "topology" or "XPDL4Chor" an .xml extension will
286 	 * be added. Otherwise a .bpel extension will be added
287 	 * 
288 	 * @param {Object} file The file name to add the extension to.
289 	 */
290 	addFileExtension: function(file) {
291 		if ((file.toLowerCase() == "topology") || (file == "XPDL4Chor")) {
292 			return file + ".xml";
293 		} else {
294 			return file + ".bpel";
295 		}
296 	},

299 	/**
300 	 * Opens a download window for downloading the given content.
301 	 * 
302 	 * Creates a submit form to send the contents to the 
303 	 * Oryx Legacy File Download Servlet (MultiDownloader).
304 	 * 
305 	 * @param {Object} content The content to be downloaded. If it is a zip 
306 	 *                         file, then this should be an array of contents.
307 	 * @param {Object} zip     True, if it is a zip file, false otherwise
308 	 */
309 	openDownloadWindow: function(content, zip) {
310 		var win = window.open("");
311 		if (win != null) {
312 			win.document.open();
313 			win.document.write("<html><body>");
314 			var submitForm = win.document.createElement("form");
315 			win.document.body.appendChild(submitForm);
316 			
317 			if (zip) {
318 				for (var i = 0; i < content.length; i++) {
319 					var file = this.addFileExtension(content[i].get("file"));
320 					submitForm.appendChild( this.createHiddenElement("download_" + i, content[i].get("result")));
321 					submitForm.appendChild( this.createHiddenElement("file_" + i, file));
322 				}
323 			} else {
324 				var file = this.addFileExtension(content.get("file"));
325 				submitForm.appendChild( this.createHiddenElement("download", content.get("result")));
326 				submitForm.appendChild( this.createHiddenElement("file", file));
327 			}
328 			
329 			submitForm.method = "POST";
330 			win.document.write("</body></html>");
331 			win.document.close();
332 			submitForm.action= "download";
333 			submitForm.submit();
334 		}		
335 	},

338 	/**
339 	 * Determines if the result is an XML file or not.
340 	 * For this purpose it is determined if the given
341 	 * result starts with "<?xml".
342 	 * 
343 	 * @param {Object} result The result to be checked.
344 	 * @return "success" if it is an XML file, "error" otherwise
345 	 */
346 	getResultInfo: function(result) {
347 		if (!result) {
348 			return "error";
349 		} else if (result.substr(0, 5) == "<?xml") {
350 			return "success";
351 		}
352 		
353 		return "error";
354 	},

357 	/**
358 	 * Determines the process name for a given process
359 	 * string. 
360 	 * 
361 	 * @param {String} process The BPEL4Chor process.
362 	 */
363 	getProcessName: function(process) {
364 		var parser	= new DOMParser();
365 		var doc		= parser.parseFromString(process,"text/xml");
366 		var name 	= doc.documentElement.getAttribute("name");
367 		return name;
368 	}
