 54     /**
 55        Show overlay on given shape.
 56        @methodOf ORYX.Plugins.AbstractPlugin.prototype
 57        @example
 58        showOverlay(
 59            myShape,
 60            { stroke: "green" },
 61            ORYX.Editor.graft("http://www.w3.org/2000/svg", null, ['path', {
 62                "title": "Click the element to execute it!",
 63                "stroke-width": 2.0,
 64                "stroke": "black",
 65                "d": "M0,-5 L5,0 L0,5 Z",
 66                "line-captions": "round"
 67            }])
 68        )
 69        @param {Oryx.XXX.Shape[]} shapes One shape or array of shapes the overlay should be put on
 70        @param {Oryx.XXX.Attributes} attributes some attributes...
 71        @param {Oryx.svg.node} svgNode The svg node which should be used as overlay
 72        @param {String} [svgNode="NW"] The svg node position where the overlay should be placed
 73     */
 74 	showOverlay: function(shapes, attributes, svgNode, svgNodePosition ){
 75 		
 76 		if( !(shapes instanceof Array) ){
 77 			shapes = [shapes]
 78 		}
 79 		
 80 		// Define Shapes
 81 		shapes = shapes.map(function(shape){
 82 			var el = shape;
 83 			if( typeof shape == "string" ){
 84 				el = this.facade.getCanvas().getChildShapeByResourceId( shape );
 85 				el = el || this.facade.getCanvas().getChildById( shape, true );
 86 			}
 87 			return el;
 88 		}.bind(this)).compact();
 89 		
 90 		// Define unified id
 91 		if( !this.overlayID ){
 92 			this.overlayID = this.type + ORYX.Editor.provideId();
 93 		}
 94 		
 95 		this.facade.raiseEvent({
 96 			type		: ORYX.CONFIG.EVENT_OVERLAY_SHOW,
 97 			id			: this.overlayID,
 98 			shapes		: shapes,
 99 			attributes 	: attributes,
100 			node		: svgNode,
101 			nodePosition: svgNodePosition || "NW"
102 		});
103 		
104 	},

117     /**
118        Does a transformation with the given xslt stylesheet.
119        @methodOf ORYX.Plugins.AbstractPlugin.prototype
120        @param {String} data The data (e.g. eRDF) which should be transformed
121        @param {String} stylesheet URL of a stylesheet which should be used for transforming data.
122     */
123 	doTransform: function( data, stylesheet ) {		
124 		
125 		if( !stylesheet || !data ){
126 			return ""
127 		}
128 
129         var parser 		= new DOMParser();
130         var parsedData 	= parser.parseFromString(data, "text/xml");
131 		source=stylesheet;
132 		new Ajax.Request(source, {
133 			asynchronous: false,
134 			method: 'get',
135 			onSuccess: function(transport){
136 				xsl = transport.responseText
137 			}.bind(this),
138 			onFailure: (function(transport){
139 				ORYX.Log.error("XSL load failed" + transport);
140 			}).bind(this)
141 		});
142         var xsltProcessor = new XSLTProcessor();
143 		var domParser = new DOMParser();
144 		var xslObject = domParser.parseFromString(xsl, "text/xml");
145         xsltProcessor.importStylesheet(xslObject);
146         
147         try {
148         	
149             var newData 		= xsltProcessor.transformToFragment(parsedData, document);
150             var serializedData 	= (new XMLSerializer()).serializeToString(newData);
151             
152            	/* Firefox 2 to 3 problem?! */
153             serializedData = !serializedData.startsWith("<?xml") ? "<?xml version=\"1.0\" encoding=\"UTF-8\"?>" + serializedData : serializedData;
154             
155             return serializedData;
156             
157         }catch (error) {
158             return -1;
159         }

179     /**
180      * Opens a download window for downloading the given content.
181      * @methodOf ORYX.Plugins.AbstractPlugin.prototype
182      * @param {String} filename The content's file name
183      * @param {String} content The content to download
184      */
185 	openDownloadWindow: function(filename, content) {
186 		var win = window.open("");
187 		if (win != null) {
188 			win.document.open();
189 			win.document.write("<html><body>");
190 			var submitForm = win.document.createElement("form");
191 			win.document.body.appendChild(submitForm);
192 			
193 			var createHiddenElement = function(name, value) {
194 				var newElement = document.createElement("input");
195 				newElement.name=name;
196 				newElement.type="hidden";
197 				newElement.value = value;
198 				return newElement
199 			}
200 			
201 			submitForm.appendChild( createHiddenElement("download", content) );
202 			submitForm.appendChild( createHiddenElement("file", filename) );
203 			
204 			
205 			submitForm.method = "POST";
206 			win.document.write("</body></html>");
207 			win.document.close();
208 			submitForm.action= ORYX.PATH + "/download";
209 			submitForm.submit();
210 		}

245     /**
246      * Sets the editor in read only mode: Edges/ dockers cannot be moved anymore,
247      * shapes cannot be selected anymore.
248      * @methodOf ORYX.Plugins.AbstractPlugin.prototype
249      */
250     enableReadOnlyMode: function(){
251         //Edges cannot be moved anymore
252         this.facade.disableEvent(ORYX.CONFIG.EVENT_MOUSEDOWN);
253         
254         // Stop the user from editing the diagram while the plugin is active
255         this._stopSelectionChange = function(){
256             if(this.facade.getSelection().length > 0) {
257                 this.facade.setSelection([]);
258             }
259         };
260         this.facade.registerOnEvent(ORYX.CONFIG.EVENT_SELECTION_CHANGED, this._stopSelectionChange.bind(this));
261     },
262     /**
263      * Disables read only mode, see @see
264      * @methodOf ORYX.Plugins.AbstractPlugin.prototype
265      * @see ORYX.Plugins.AbstractPlugin.prototype.enableReadOnlyMode
266      */
267     disableReadOnlyMode: function(){
268         // Edges can be moved now again
269         this.facade.enableEvent(ORYX.CONFIG.EVENT_MOUSEDOWN);
270         
271         if (this._stopSelectionChange) {
272             this.facade.unregisterOnEvent(ORYX.CONFIG.EVENT_SELECTION_CHANGED, this._stopSelectionChange.bind(this));
273             this._stopSelectionChange = undefined;
274         }

363 	/**
364 	 * Does a primitive layouting with the incoming/outgoing 
365 	 * edges (set the dockers to the right position) and if 
366 	 * necessary, it will be called the real layouting 
367 	 * @param {ORYX.Core.Node} node
368 	 * @param {Array} edges
369 	 */
370 	layoutEdges : function(node, allEdges, offset){		
371 		
372 		// Find all edges, which are related to the node and
373 		// have more than two dockers
374 		var edges = allEdges
375 			// Find all edges with more than two dockers
376 			.findAll(function(r){ return r.dockers.length > 2 }.bind(this))
377 			
378 		if (edges.length > 0) {
379 										
380 			// Get the new absolute center
381 			var center = node.absoluteXY();
382 			
383 			var ulo = {x: center.x - offset.x, y:center.y - offset.y}			
384 			
385 			center.x += node.bounds.width()/2;
386 			center.y += node.bounds.height()/2;
387 			
388 			// Get the old absolute center
389 			oldCenter = Object.clone(center);
390 			oldCenter.x -= offset ? offset.x : 0;
391 			oldCenter.y -= offset ? offset.y : 0;
392 			
393 			var ul = {x: center.x - (node.bounds.width() / 2), y: center.y - (node.bounds.height() / 2)}
394 			var lr = {x: center.x + (node.bounds.width() / 2), y: center.y + (node.bounds.height() / 2)}
395 			
396 			
397 			/**
398 			 * Align the bounds if the center is 
399 			 * the same than the old center
400 			 * @params {Object} bounds
401 			 * @params {Object} bounds2
402 			 */
403 			var align = function(bounds, bounds2){
404 				var xdif = bounds.center().x-bounds2.center().x;
405 				var ydif = bounds.center().y-bounds2.center().y;
406 				if (Math.abs(xdif) < 3){
407 					bounds.moveBy({x:(offset.xs?(((offset.xs*(bounds.center().x-ulo.x))+offset.x+ulo.x)-bounds.center().x):offset.x)-xdif, y:0});		
408 				} else if (Math.abs(ydif) < 3){
409 					bounds.moveBy({x:0, y:(offset.ys?(((offset.ys*(bounds.center().y-ulo.y))+offset.y+ulo.y)-bounds.center().y):offset.y)-ydif});		
410 				}
411 			};
412 									
413 			/**						
414 			 * Returns a TRUE if there are bend point which overlay the shape
415 			 */
416 			var isBendPointIncluded = function(edge){
417 				// Get absolute bounds
418 				var ab = edge.dockers.first().getDockedShape();
419 				var bb = edge.dockers.last().getDockedShape();
420 				
421 				if (ab) {
422 					ab = ab.absoluteBounds();
423 					ab.widen(5);
424 				}
425 				
426 				if (bb) {
427 					bb = bb.absoluteBounds();
428 					bb.widen(20); // Wide with 20 because of the arrow from the edge
429 				}
430 				
431 				return edge.dockers
432 						.any(function(docker, i){ 
433 							var c = docker.bounds.center();
434 									// Dont count first and last
435 							return 	i != 0 && i != edge.dockers.length-1 && 
436 									// Check if the point is included to the absolute bounds
437 									((ab && ab.isIncluded(c)) || (bb && bb.isIncluded(c)))
438 						})
439 			}
440 			// For every edge, check second and one before last docker
441 			// if there are horizontal/vertical on the same level
442 			// and if so, align the the bounds 
443 			edges.each(function(edge){
444 				if (edge.dockers.first().getDockedShape() === node){
445 					var second = edge.dockers[1];
446 					if (align(second.bounds, edge.dockers.first().bounds)){ second.update(); }
447 				} else if (edge.dockers.last().getDockedShape() === node) {
448 					var beforeLast = edge.dockers[edge.dockers.length-2];
449 					if (align(beforeLast.bounds, edge.dockers.last().bounds)){ beforeLast.update(); }									
450 				}
451 				edge._update(true);
452 				edge.removeUnusedDockers();
453 				if (isBendPointIncluded(edge)){
454 					this.doLayout(edge);
455 					return;
456 				}
457 			}.bind(this))
458 		}	
459 
460 		// Find all edges, which have only to dockers 
461 		// and is located horizontal/vertical.
462 		// Do layout with those edges
463 		allEdges.each(function(edge){
464 				// Find all edges with two dockers
465 				if (edge.dockers.length == 2){
466 					var p1 = edge.dockers.first().bounds.center();
467 					var p2 = edge.dockers.last().bounds.center();
468 					// Find all horizontal/vertical edges
469 					if (Math.abs(p1.x - p2.x) < 2 || Math.abs(p1.y - p2.y) < 2){
470 						edge.dockers.first().update();
471 						edge.dockers.last().update();
472 						this.doLayout(edge);
473 					}
474 				}
475 			}.bind(this));
476 	}
