 31 /**
 32  * The UMLUseCase plugin provides layout methods referring to the UMLUseCase stencilset. 
 33  * 
 34  * @class ORYX.Plugins.UMLUseCase
 35  * @extends Clazz
 36  * @param {Object} facade The facade of the editor
 37  */
 38 ORYX.Plugins.UMLUseCase = 
 39 /** @lends ORYX.Plugins.UMLUseCase.prototype */
 40 {
 41 	/**
 42 	 * Creates a new instance of the UML plugin and registers it on the
 43 	 * layout events listed in the UML UseCase stencil set.
 44 	 * 
 45 	 * @constructor
 46 	 * @param {Object} facade The facade of the editor
 47 	 */
 48 	construct: function(facade) {
 49 		this.facade = facade;
 50 		this.facade.registerOnEvent('layout.uml.useCaseExtended', this.handleUseCaseExtendedLayout.bind(this));
 51 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_LOADED, this.handleDiagramOnLoad.bind(this));
 52 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_PROPWINDOW_PROP_CHANGED, this.handlePropertyChanged.bind(this));
 53 	},
 54 	
 55 	/**
 56 	 * Handles the layouting after loading a diagram
 57 	 * This Function starts a recursion 
 58 	 * which layouts the stereotypes of all systems and
 59 	 * the UseCaseExtended.
 60 	 *
 61 	 * @param {Object} event The event, that is fired after a Use Case diagram is loaded
 62 	 */
 63 	handleDiagramOnLoad : function(event) {
 64 		this.layoutAllSystems(this.facade.getCanvas());
 65 	},
 66 	
 67 	/**
 68 	* Recursivly layouts the system, 
 69 	* all child systems and the UseCasesExtended the system includes
 70 	*
 71 	* @param {Object} shape A system shape
 72 	*/
 73 	layoutAllSystems : function (shape){
 74 		
 75 		this.layoutAllUseCaseExtended(shape);
 76 		
 77 		var systems = shape.getChildNodes().findAll( function(child) {
 78 			if (this.isSystemNode(child)) {
 79 				return child;}
 80 		}.bind(this));
 81 		
 82 		systems.each(function(systemNode){ 
 83 			this.layoutStereotype(systemNode);
 84 			this.layoutAllSystems(systemNode); //recursion
 85 		}.bind(this));
 86 	
 87 	},
 88 	
 89 	/**
 90 	* Layouts all UseCasesExtended contained in the system
 91 	*
 92 	* @param {Object} shape A UseCaseExtended shape
 93 	*/
 94 	layoutAllUseCaseExtended : function (shape){
 95 		
 96 		var useCasesExtended = shape.getChildNodes().findAll(function(child) {
 97 			if (this.isUseCaseExtendedNode(child)) {
 98 				return child;}
 99 		}.bind(this));
100 		
101 		useCasesExtended.each(function(useCaseExtendedNode){ 
102 			this.layoutUseCaseExtended(useCaseExtendedNode);
103 		}.bind(this));
104 	
105 	},
106 	
107 	/**
108 	* This method is called when a UseCaseExtended shape is modified.
109 	* It registers another handling method to the selection changed event, 
110 	* since that event is fired late enough to do the needed layouting.
111 	*
112 	* The chain of called methods, to do the layouting:
113 	*		handleUseCaseExtendedLayout(...) | layouting event fired
114 	*	->	handleUseCaseExtendedSelection(...) | selection event fired
115 	*	->	layoutUseCaseExtended(...) | simple method call
116 	*
117 	* @param {Object} event The event which is fired after the UseCaseExtended shape is modified
118 	*/
119 	handleUseCaseExtendedLayout : function(event) {
120 		
121 		var shape = event.shape;
122 		if (shape.isResized){
123 			this.SelectionEventFunction = this.handleUseCaseExtendedSelection.bind(this);
124 			this.facade.registerOnEvent(ORYX.CONFIG.EVENT_SELECTION_CHANGED, this.SelectionEventFunction);
125 		}		
126 	},
127 	
128 	/**
129 	* This method is called after a UseCaseExtended shape is modified.
130 	* It deregisters itself on the selection changed event, 
131 	* since this event is fired late enough to do the needed layouting.
132 	*
133 	* First the method checks whether the event has a reference to a UseCaseExtended shape, 
134 	* since there is no guarantee.
135 	* 
136 	* @param {Object} event The event which is fired after the UseCaseExtended is resized
137 	*/
138 	handleUseCaseExtendedSelection : function(event) {
139 		
140 		var shape = event.elements.first();
141 		if (!this.isUseCaseExtendedNode(shape)) {return;}
142 		
143 		this.facade.unregisterOnEvent(ORYX.CONFIG.EVENT_SELECTION_CHANGED, this.SelectionEventFunction);
144 		this.layoutUseCaseExtended(shape);
145 	},
146 	
147 	/**
148 	* This method layouts the UseCaseExtended shape
149 	* It places the textframe and text which describe the extensions directly underneath 
150 	* the text "Extension Points"
151 	* Neither the resize attribute nor the anchors attribute had the desired effect.
152 	* 
153 	* @param {Object} shape A UseCaseExtended shape
154 	*/
155 	layoutUseCaseExtended : function(shape){
156 	
157 		var extensionPointTextFrame = shape._svgShapes.find(
158 					function(element) { return element.element.id == (shape.id + "extension_point_text_frame"); }
159 				).element;
160 		var extensionPointText = shape.getLabels().find(
161 					function(label) { return label.id == (shape.id + "extensions"); }
162 				);
163 		var extensionText = shape.getLabels().find(
164 					function(label) { return label.id == (shape.id + "extensionpoint"); }
165 				);
166 		var verticalDifference = extensionPointTextFrame.y.baseVal.value - extensionText.y-16;
167 		extensionPointTextFrame.y.baseVal.value= extensionText.y+16;
168 		extensionPointText.y= extensionText.y+16;
169 		extensionPointTextFrame.height.baseVal.value += verticalDifference;	
170 	
171 	},	
172 
173 	
174 	/**
175 	* Initiates the Stereotype layout
176 	* This method is called whenever a property of a shape is changed. 
177 	* It filters the events, which belong to a System shape and are fired,
178 	* when the stereotype or name is changed.
179 	*
180 	*@param {Object} event The event that is fired after changing a property of a shape.
181 	*/
182 	handlePropertyChanged : function(event) {
183 		var shape = event.elements.first();
184 		if (!this.isSystemNode(shape)) {return;}
185 		if (!(this.isStereotypeKeyEvent(event) || this.isNameKeyEvent(event))) {return;}
186 		this.layoutStereotype(shape);			
187 	},
188 	
189 	/**
190 	* Layouts the stereotype and name properties of a System shape
191 	* It places the stereotype and property of a System properly and hides the stereotype when necessary
192 	*
193 	*@param {Object} shape The System shape the user modified the stereotype/name from.
194 	*/
195 	layoutStereotype: function(shape) {
196 	
197 		var stereotype = shape.getLabels().find(
198 				function(label) { return label.id == (shape.id + "stereotype"); }
199 		);
200 		var name = shape.getLabels().find(
201 				function(label) { return label.id == (shape.id + "text"); }
202 		);
203 		
204 		if (stereotype.text().empty()){
205 					name.y=stereotype.y;
206 					stereotype.hide();
207 				}else{
208 					name.y = stereotype.y+14;
209 					stereotype.show();
210 					stereotype.text("≪" + shape.properties["oryx-stereotype"] + "≫");
211 				}
212 		name.update();
213 		stereotype.update();
214 	},
215 	
216 	/**
217 	* Helper method, which returns true, if the received shape is an extended Use Case. 
218 	*@private
219 	*@param {Object} shape The shape that is checked for beeing an extended Use Case.
220 	*@return {boolean} The result is true, if the shape is an extended Use Case.
221 	*/
222 	isUseCaseExtendedNode : function (shape) {
223 		return "http://b3mn.org/stencilset/umlusecase#usecaseextended" == shape.getStencil().id().toLowerCase();
224 	},
225 	
226 	/**
227 	* Helper method, which returns true, if the received shape is a system. 
228 	*@private
229 	*@param {Object} shape The shape that is checked for beeing a system.
230 	*@return {boolean} The result is true, if the shape is a system
231 	*/
232 	isSystemNode : function (shape) {
233 		return "http://b3mn.org/stencilset/umlusecase#system" == shape.getStencil().id().toLowerCase();
234 	},
235 	
236 	/**
237 	* Helper method, which returns true, if the received event handles a stereotype.
238 	*@private 
239 	*@param {Object} event The event that is checked for handling an oryx-stereotype.
240 	*@return {boolean} The result is true, if event handles a stereotype
241 	*/
242 	isStereotypeKeyEvent : function (event) {
243 		return event["key"] == "oryx-stereotype";
244 	},
245 	
246 	/**
247 	* Helper method, which returns true, if the received event handles a name.
248 	*@private 
249 	*@param {Object} event The event that is checked for handling an oryx-name.
250 	*@return {boolean} The result is true, if event handles a name
251 	*/
252 	isNameKeyEvent : function (event) {
253 		return event["key"] == "oryx-name";
254 	}
255 };
256 
257 ORYX.Plugins.UMLUseCase = Clazz.extend(ORYX.Plugins.UMLUseCase);
