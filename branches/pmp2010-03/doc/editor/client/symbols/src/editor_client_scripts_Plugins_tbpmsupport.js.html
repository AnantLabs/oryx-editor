 27 /**
 28  * Enables exporting and importing current model in JSON.
 29  */
 30 ORYX.Plugins.TBPMSupport = ORYX.Plugins.AbstractPlugin.extend({
 31 
 32 	facade: undefined,
 33 	
 34 	tbpmImportServletURL: ORYX.CONFIG.TBPMIMPORT,
 35 	
 36 	canvasId: "ext-gen56",
 37 	
 38 	TMP_FOLDER: "/oryx/tmp/",
 39 	
 40     construct: function(){
 41         // Call super class constructor
 42         arguments.callee.$.construct.apply(this, arguments);
 43         
 44         this.facade.offer({
 45             'name': ORYX.I18N.TBPMSupport.imp.name,
 46             'functionality': this.showImportDialog.bind(this),
 47             'group': ORYX.I18N.TBPMSupport.imp.group,
 48             'toolbarGroup': ORYX.I18N.TBPMSupport.toolbarGroup,
 49             'dropDownGroupIcon': ORYX.PATH + "images/import.png",
 50 			'icon': ORYX.PATH + "images/page_white_picture.png",
 51             'description': ORYX.I18N.TBPMSupport.imp.desc,
 52             'index': 3,
 53             'minShape': 0,
 54             'maxShape': 0
 55         });
 56     },
 57     
 58     /**
 59      * Opens a upload dialog.
 60      *
 61      */
 62     showImportDialog: function(successCallback){
 63         this.form = new Ext.form.FormPanel({
 64             baseCls: 'x-plain',
 65             labelWidth: 50,
 66             defaultType: 'textfield',
 67             fileUpload : true,
 68 		  	enctype : 'multipart/form-data',
 69             items: [{
 70                 text: ORYX.I18N.TBPMSupport.imp.selectFile,
 71                 style: 'font-size:12px;margin-bottom:10px;display:block;',
 72                 anchor: '100%',
 73                 xtype: 'label'
 74             }, {
 75                 fieldLabel: ORYX.I18N.TBPMSupport.imp.file,
 76                 name: 'subject',
 77                 inputType: 'file',
 78                 style: 'margin-bottom:10px;display:block;width:95%',
 79                 itemCls: 'ext_specific_window_overflow'
 80             }]
 81         });
 82         
 83         // Create the panel
 84         this.dialog = new Ext.Window({
 85             autoCreate: true,
 86             layout: 'fit',
 87             plain: true,
 88             bodyStyle: 'padding:5px;',
 89             title: ORYX.I18N.TBPMSupport.imp.name,
 90             height: 150,
 91             width: 500,
 92             modal: true,
 93             fixedcenter: true,
 94             shadow: true,
 95             proxyDrag: true,
 96             resizable: true,
 97             items: [this.form],
 98             buttons: [{
 99                 text: ORYX.I18N.TBPMSupport.imp.btnImp,
100                 handler: this.uploadImage.bind(this)
101             }, {
102                 text: ORYX.I18N.TBPMSupport.imp.btnClose,
103                 handler: function(){
104                     this.dialog.close();
105                 }.bind(this)
106             }]
107         });
108         
109         this.dialog.on('hide', function(){
110 			this.dialog.destroy(true);
111 			delete this.dialog;
112 		}.bind(this));
113         
114         // Show the panel
115         this.dialog.show();        
116     },
117     
118     uploadImage: function(button, event) {    	
119     	  	
120     	var imageName = this.form.items.items[1].getValue().replace("png", "jpg");
121     	
122 		var loadMask = new Ext.LoadMask(Ext.getBody(), {
123 		    msg: ORYX.I18N.TBPMSupport.imp.progress
124 		});
125 		loadMask.show();
126     	
127 		this.form.form.submit({	
128     				url: this.tbpmImportServletURL, 
129     				clientValidation: true,
130     				waitMsg:'Saving Data...',
131     				method: "POST",
132     				
133     				success: function(form, action) {
134 						obj = Ext.util.JSON.decode(action.response.responseText);
135 						alert(object);
136     			    	this.dialog.hide();
137     					this.showConfirmDialog(imageName);
138     					// Hide the waiting panel
139     					loadMask.hide()  ;  					
140     			    }.bind(this),
141     			    
142     			    // invokes failure handler even i case of successful response (no idea why)
143     				failure: function(form, action){
144     			    	this.dialog.hide();
145     					this.showConfirmDialog(imageName, action.response.responseText);
146     					// Hide the waiting panel
147     					loadMask.hide()  ;
148     			    	
149     			    	//this.dialog.hide();
150     			    	//loadMask.hide();
151     					//Ext.Msg.alert(ORYX.I18N.Oryx.title, ORYX.I18N.TBPMSupport.imp.impFailed);
152     				}.bind(this)
153     	});
154         
155     },  
156     
157     /*
158      * show image with highlighted shapes
159      * import model and image layer if image confirmed
160      */
161     showConfirmDialog: function(imageName, json){
162     	
163     	var confirmDialog = new Ext.Window({
164     		autoCreate: true,
165     		layout: 'fit',
166     		width: 600,
167     	    height: 500,
168     	    bodyStyle: 'padding:5px;',
169     	    autoScroll: true,
170     	    modal: true,
171             fixedcenter: true,
172             shadow: true,
173             proxyDrag: true,
174             resizable: true,
175     	    title: ORYX.I18N.TBPMSupport.imp.confirm,
176     	    html: '<div style="width:100%;">' +
177             			'<img src="'+ this.TMP_FOLDER + imageName + '" style="width:550px;"></img>'+
178             		'</div>',
179     	    
180     	    buttons: [{
181                 text: ORYX.I18N.TBPMSupport.imp.btnImp,
182                 handler: function() {
183     	    		confirmDialog.close();
184     	    		this.processImport(imageName, json);
185     	    	}.bind(this)
186             }, {
187                 text: ORYX.I18N.TBPMSupport.imp.btnClose,
188                 handler: function(){
189                     confirmDialog.close();
190                 }.bind(this)
191             }]
192     	});
193     	
194     	confirmDialog.show();
195     },
196     
197     processImport: function(imageName, json){
198     	
199     	this.addImageLayer(imageName);
200     	this.importShapes(json);
201     	
202     	// update the canvas
203 		this.facade.getCanvas().update();
204     },
205     
206     /*
207      * show transparent Layer with image
208      * 
209      */
210     addImageLayer: function(imageName){  
211 		$(this.canvasId).style.background = "url(" + this.TMP_FOLDER + imageName + ") no-repeat scroll center center";
212     },
213     
214     /*
215      * generate detected shapes
216      */
217     importShapes: function(json){
218     	this.facade.importJSON(json, true);
219     }
220 
221    
222 });
