 25 /**
 26    This plugin is a generic syntax checker for different diagram types.
 27    Needs server communication.
 28    @class ORYX.Plugins.SyntaxChecker
 29    @constructor Creates a new instance
 30    @extends ORYX.Plugins.AbstractPlugin
 31 */
 32 ORYX.Plugins.SyntaxChecker = ORYX.Plugins.AbstractPlugin.extend({
 33     /**@private*/
 34     construct: function(){
 35         arguments.callee.$.construct.apply(this, arguments);
 36                 
 37         this.active = false;
 38         this.raisedEventIds = [];
 39         
 40         this.facade.offer({
 41             'name': ORYX.I18N.SyntaxChecker.name,
 42             'functionality': this.perform.bind(this),
 43             'group': ORYX.I18N.SyntaxChecker.group,
 44             'icon': ORYX.PATH + "images/checker_syntax.png",
 45             'description': ORYX.I18N.SyntaxChecker.desc,
 46             'index': 0,
 47             'toggle': true,
 48             'minShape': 0,
 49             'maxShape': 0
 50         });
 51         
 52         this.facade.registerOnEvent(ORYX.Plugins.SyntaxChecker.CHECK_FOR_ERRORS_EVENT, this.checkForErrors.bind(this));
 53         this.facade.registerOnEvent(ORYX.Plugins.SyntaxChecker.RESET_ERRORS_EVENT, this.resetErrors.bind(this));
 54         this.facade.registerOnEvent(ORYX.Plugins.SyntaxChecker.SHOW_ERRORS_EVENT, this.doShowErrors.bind(this));
 55     },
 56     
 57     perform: function(button, pressed){
 58         if (!pressed) {
 59             this.resetErrors();
 60         } else {
 61             this.checkForErrors({
 62                 onNoErrors: function(){
 63                     this.setActivated(button, false);
 64                     this.facade.raiseEvent({
 65             			type:ORYX.CONFIG.EVENT_LOADING_STATUS,
 66             			text:ORYX.I18N.SyntaxChecker.noErrors,
 67             			timeout:10000
 68             		});
 69                     //Ext.Msg.alert(ORYX.I18N.Oryx.title, ORYX.I18N.SyntaxChecker.noErrors);
 70                 }.bind(this),
 71                 onErrors: function(){
 72                     this.enableDeactivationHandler(button);
 73                 }.bind(this),
 74                 onFailure: function(){
 75                     this.setActivated(button, false);
 76                     Ext.Msg.alert(ORYX.I18N.Oryx.title, ORYX.I18N.SyntaxChecker.invalid);
 77                 }.bind(this)
 78             });      
 79         }
 80     },
 81     
 82     /**
 83      * Registers handler for deactivating syntax checker as soon as somewhere is clicked...
 84      * @param {Ext.Button} Toolbar button
 85      */
 86     enableDeactivationHandler: function(button){
 87         var deactivate = function(){
 88             this.setActivated(button, false);
 89             this.resetErrors();
 90             this.facade.unregisterOnEvent(ORYX.CONFIG.EVENT_MOUSEDOWN, deactivate);
 91         };
 92         
 93         this.facade.registerOnEvent(ORYX.CONFIG.EVENT_MOUSEDOWN, deactivate.bind(this));
 94     },
 95     
 96     /**
 97      * Sets the activated state of the plugin
 98      * @param {Ext.Button} Toolbar button
 99      * @param {Object} activated
100      */
101     setActivated: function(button, activated){
102         button.toggle(activated);
103         if(activated === undefined){
104             this.active = !this.active;
105         } else {
106             this.active = activated;
107         }
108     },
109     
110     /**
111      * Performs request to server to check for errors on current model.
112      * @methodOf ORYX.Plugins.SyntaxChecker.prototype
113      * @param {Object} options Configuration hash
114      * @param {String} context A context send to the syntax checker servlet
115      * @param {Function} [options.onNoErrors] Raised when model has no errors.
116      * @param {Function} [options.onErrors] Raised when model has errors.
117      * @param {Function} [options.onFailure] Raised when server communcation failed.
118      * @param {boolean} [options.showErrors=true] Display errors on nodes on canvas (by calling ORYX.Plugins.SyntaxChecker.prototype.showErrors)
119      */
120     checkForErrors: function(options){
121         Ext.applyIf(options || {}, {
122           showErrors: true,
123           onErrors: Ext.emptyFn,
124           onNoErrors: Ext.emptyFn,
125           onFailure: Ext.emptyFn
126         });
127             
128         Ext.Msg.wait(ORYX.I18N.SyntaxChecker.checkingMessage);
129 
130 		var ss = this.facade.getStencilSets();
131 		var data = null;
132 		var includesJson = false;
133 		
134 		if(ss.keys().include("http://b3mn.org/stencilset/bpmn2.0#") ||
135 				ss.keys().include("http://b3mn.org/stencilset/bpmn2.0conversation#")) {
136 			data = this.facade.getSerializedJSON();
137 			includesJson = true;
138 		} else {
139 			data = this.getRDFFromDOM();
140 		}
141         
142         // Send the request to the server.
143         new Ajax.Request(ORYX.CONFIG.SYNTAXCHECKER_URL, {
144             method: 'POST',
145             asynchronous: false,
146             parameters: {
147                 resource: location.href,
148                 data: data,
149                 context: options.context,
150 				isJson: includesJson
151             },
152             onSuccess: function(request){
153                 var resp = (request&&request.responseText?request.responseText:"{}").evalJSON();
154                 
155                 Ext.Msg.hide();
156                 
157                 if (resp instanceof Object) {
158                     resp = $H(resp)
159                     if (resp.size() > 0) {
160                         if(options.showErrors) this.showErrors(resp);
161                  
162                         options.onErrors();
163                     }
164                     else {
165                         options.onNoErrors();
166                     }
167                 }
168                 else {
169                     options.onFailure();
170                 }
171             }.bind(this),
172             onFailure: function(){
173                 Ext.Msg.hide();
174                 options.onFailure();
175             }
176         });
177     },
178     
179     /** Called on SHOW_ERRORS_EVENT.
180      * 
181      * @param {Object} event
182      * @param {Object} args
183      */
184     doShowErrors: function(event, args){
185         this.showErrors(event.errors);
186     },
187     
188     /**
189      * Shows overlays for each given error
190      * @methodOf ORYX.Plugins.SyntaxChecker.prototype
191      * @param {Hash|Object} errors
192      * @example
193      * showErrors({
194      *     myShape1: "This has an error!",
195      *     myShape2: "Another error!"
196      * })
197      */
198     showErrors: function(errors){
199         // If normal object is given, convert to hash
200         if(!(errors instanceof Hash)){
201             errors = new Hash(errors);
202         }
203         
204         // Get all Valid ResourceIDs and collect all shapes
205         errors.keys().each(function(value){
206             var sh = this.facade.getCanvas().getChildShapeByResourceId(value);
207             if (sh) {
208                 this.raiseOverlay(sh, this.parseCodeToMsg(errors[value]));
209             }
210         }.bind(this));
211         this.active = !this.active;
212         
213         //show a status message with a hint to the error messages in the tooltip
214         this.facade.raiseEvent({
215 			type:ORYX.CONFIG.EVENT_LOADING_STATUS,
216 			text:ORYX.I18N.SyntaxChecker.notice,
217 			timeout:10000
218 		});
219     },
220     parseCodeToMsg: function(code){
221     	var msg = code.replace(/: /g, "<br />").replace(/, /g, "<br />");
222     	var codes = msg.split("<br />");
223     	for (var i=0; i < codes.length; i++) {
224     		var singleCode = codes[i];
225     		var replacement = this.parseSingleCodeToMsg(singleCode);
226     		if (singleCode != replacement) {
227     			msg = msg.replace(singleCode, replacement);
228     		}
229     	}
230 		
231 		return msg;
232 	},
233 	
234 	parseSingleCodeToMsg: function(code){
235 		return ORYX.I18N.SyntaxChecker[code]||code;
236 	},
237     /**
238      * Resets all (displayed) errors
239      * @methodOf ORYX.Plugins.SyntaxChecker.prototype
240      */
241     resetErrors: function(){
242         this.raisedEventIds.each(function(id){
243             this.facade.raiseEvent({
244                 type: ORYX.CONFIG.EVENT_OVERLAY_HIDE,
245                 id: id
246             });
247         }.bind(this))
248         
249         this.raisedEventIds = [];
250         this.active = false;
251     },
252     
253     raiseOverlay: function(shape, errorMsg){
254         var id = "syntaxchecker." + this.raisedEventIds.length;
255         var crossId = ORYX.Editor.provideId();
256         var cross = ORYX.Editor.graft("http://www.w3.org/2000/svg", null, ['path', {
257             "id":crossId,
258             "title":"",
259             "stroke-width": 5.0,
260             "stroke": "red",
261             "d": "M20,-5 L5,-20 M5,-5 L20,-20",
262             "line-captions": "round"
263         }]);
264         
265         this.facade.raiseEvent({
266             type: ORYX.CONFIG.EVENT_OVERLAY_SHOW,
267             id: id,
268             shapes: [shape],
269             node: cross,
270             nodePosition: shape instanceof ORYX.Core.Edge ? "START" : "NW"
271         });
272         
273         var tooltip = new Ext.ToolTip({
274         	showDelay:50,
275         	html:errorMsg,
276         	target:crossId
277         });
278         
279         this.raisedEventIds.push(id);
280         
281         return cross;
282     }
283 });

285 ORYX.Plugins.SyntaxChecker.CHECK_FOR_ERRORS_EVENT = "checkForErrors";
286 ORYX.Plugins.SyntaxChecker.RESET_ERRORS_EVENT = "resetErrors";
287 ORYX.Plugins.SyntaxChecker.SHOW_ERRORS_EVENT = "showErrors";
