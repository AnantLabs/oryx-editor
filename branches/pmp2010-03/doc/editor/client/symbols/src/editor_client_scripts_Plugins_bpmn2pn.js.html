 26 ORYX.Plugins.BPMNImport = Clazz.extend({
 27     converterUrl: ORYX.CONFIG.ROOT_PATH + "bpmn2pn",
 28     
 29     // Offers the plugin functionality
 30     construct: function(facade){
 31     
 32         this.facade = facade;
 33        
 34         this.importBpmn();
 35     },
 36     
 37     /**
 38      * General helper method for parsing a param out of current location url
 39      * E.g. "http://oryx.org?param=value", getParamFromUrl("param") => "value"
 40      * @param {Object} name
 41      */
 42     getParamFromUrl: function(name){
 43         name = name.replace(/[\[]/, "\\\[").replace(/[\]]/, "\\\]");
 44         var regexS = "[\\?&]" + name + "=([^&#]*)";
 45         var regex = new RegExp(regexS);
 46         var results = regex.exec(window.location.href);
 47         if (results == null) {
 48             return null;
 49         }
 50         else {
 51             return results[1];
 52         }
 53     },
 54     
 55     /**
 56      * Posts rdf (BPNM) to server and loads erdf (Petri net) into canvas
 57      * @param {Object} bpmnRdf
 58      */
 59     bpmnToPn: function(bpmnRdf){
 60         Ext.Msg.updateProgress(0.66, ORYX.I18N.BPMN2PNConverter.progress.convertingModel);
 61         
 62        Ext.Ajax.request({
 63             url: this.converterUrl,
 64             method: 'POST',
 65             success: function(request){
 66     	   		try{
 67 	                var parser = new DOMParser();
 68 	                Ext.Msg.updateProgress(1.0, ORYX.I18N.BPMN2PNConverter.progress.renderingModel);
 69 	                var doc = parser.parseFromString(request.responseText, "text/xml");
 70 	                this.facade.importERDF(doc);
 71     	   		}catch(e){
 72     	   			Ext.Msg.alert("Rendering Failed :\n"+e);
 73     	   		}
 74                 Ext.Msg.hide();
 75             }.createDelegate(this),
 76             failure: function(){
 77                 Ext.Msg.alert(ORYX.I18N.BPMN2PNConverter.error, ORYX.I18N.BPMN2PNConverter.errors.server);
 78             },
 79             params: {
 80                 rdf: bpmnRdf
 81             }
 82         });
 83         
 84     },
 85     
 86     /**
 87      * Loads rdf of given bpmn url
 88      */
 89     importBpmn: function(){
 90         var importBPMNUrl = this.getParamFromUrl("importBPMN");
 91         
 92         if(!importBPMNUrl) return; //return if no model to import is given
 93         
 94         Ext.Msg.progress(ORYX.I18N.BPMN2PNConverter.progress.status, 
 95                          ORYX.I18N.BPMN2PNConverter.progress.importingModel
 96         );
 97         Ext.Msg.updateProgress(0.33, ORYX.I18N.BPMN2PNConverter.progress.fetchingModel);
 98         
 99         Ext.Ajax.request({
100             url: this.getRdfUrl(importBPMNUrl),
101             success: function(request){
102                 var bpmnRdf = request.responseText;
103                 this.bpmnToPn(bpmnRdf);
104             }.createDelegate(this),
105             failure: function(request){
106                 Ext.Msg.alert(ORYX.I18N.BPMN2PNConverter.error, ORYX.I18N.BPMN2PNConverter.errors.noRights)
107             },
108             method: "GET"
109         })
110     },
111     
112     /**
113      * getRdfUrl("http://localhost:8080/backend/poem/model/7/self") 
114      * => "http://localhost:8080/backend/poem/model/7/rdf"
115      * getRdfUrl("http://localhost:8080/backend/poem/model/7/rdf" => "http://localhost:8080/backend/poem/model/7/rdf")
116      * @param {String} url
117      */
118     getRdfUrl: function(url){
119         return url.replace(/\/self(\/)?$/, "/rdf")
120     }
121 });

123 ORYX.Plugins.PNExport = Clazz.extend({
124     // Offers the plugin functionality
125     construct: function(facade){
126     
127         this.facade = facade;
128        
129         this.facade.offer({
130             'name': ORYX.I18N.BPMN2PNConverter.name,
131             'functionality': this.exportIt.bind(this),
132             'group': ORYX.I18N.BPMN2PNConverter.group,
133             dropDownGroupIcon: ORYX.PATH + "images/export2.png",
134 			'icon': ORYX.PATH + "images/page_white_convert.png",
135             'description': ORYX.I18N.BPMN2PNConverter.desc,
136             'index': 3,
137             'minShape': 0,
138             'maxShape': 0
139         });
140     },
141     
142     exportIt: function(){
143         //Throw error if model hasn't been saved before
144     	var reqURI='';
145 		if(!location.hash.slice(1)){
146             Ext.Msg.alert(ORYX.I18N.BPMN2PNConverter.error, ORYX.I18N.BPMN2PNConverter.errors.notSaved);
147             return;
148 		}
149 		else{
150 			reqURI = '/backend/poem/'+(location.hash.slice(1).replace(/^\/?/,"").replace(/\/?$/,""))+"/rdf";
151 		}
152         
153         this.facade.raiseEvent({type: ORYX.Plugins.SyntaxChecker.RESET_ERRORS_EVENT});
154         this.facade.raiseEvent({
155             type: ORYX.Plugins.SyntaxChecker.CHECK_FOR_ERRORS_EVENT,
156             context: "bpmn2pn",
157             onNoErrors: function(){
158                 this.openPetriNetEditor(reqURI);
159             }.bind(this)
160         });
161     },
162     
163     /**
164      * Opens petri net editor with bpmn model import
165      * @methodOf: ORYX.Plugins.BPMNImport.prototype
166      */
167     openPetriNetEditor: function(importUrl){
168         window.open("/backend/poem/new?stencilset=/stencilsets/petrinets/petrinet.json&importBPMN=" + importUrl);
169     }
170 });
