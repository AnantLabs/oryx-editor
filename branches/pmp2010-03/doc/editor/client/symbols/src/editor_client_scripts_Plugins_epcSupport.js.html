 74 	/**
 75 	 * Exports the diagram into an AML or EPML file
 76 	 */
 77 	exportEPC: function(){
 78 
 79 		this.facade.raiseEvent({type:ORYX.CONFIG.EVENT_LOADING_ENABLE, text:ORYX.I18N.EPCSupport.progressExp});
 80 		var xmlSerializer = new XMLSerializer();
 81 
 82 		
 83 		// TODO: a Syntax Syntax-Check should be triggered, here.
 84 		 
 85 		// TODO: get process' name
 86 		var resource = "Oryx-EPC";
 87 		
 88 		// Force to set all resource IDs
 89 		var serializedDOM = DataManager.serializeDOM( this.facade );
 90 
 91 		//add namespaces
 92 		serializedDOM = '<?xml version="1.0" encoding="utf-8"?>' +
 93 		'<html xmlns="http://www.w3.org/1999/xhtml" ' +
 94 		'xmlns:b3mn="http://b3mn.org/2007/b3mn" ' +
 95 		'xmlns:ext="http://b3mn.org/2007/ext" ' +
 96 		'xmlns:rdf="http://www.w3.org/1999/02/22-rdf-syntax-ns#" ' +
 97 		'xmlns:atom="http://b3mn.org/2007/atom+xhtml">' +
 98 		'<head profile="http://purl.org/NET/erdf/profile">' +
 99 		'<link rel="schema.dc" href="http://purl.org/dc/elements/1.1/" />' +
100 		'<link rel="schema.dcTerms" href="http://purl.org/dc/terms/ " />' +
101 		'<link rel="schema.b3mn" href="http://b3mn.org" />' +
102 		'<link rel="schema.oryx" href="http://oryx-editor.org/" />' +
103 		'<link rel="schema.raziel" href="http://raziel.org/" />' +
104 		'<base href="' +
105 		location.href.split("?")[0] +
106 		'" />' +
107 		'</head><body>' +
108 		serializedDOM +
109 		'<div id="generatedProcessInfos"><span class="oryx-id">' + resource + '</span>' + 
110 		'<span class="oryx-name">' + resource + '</span></div>' +
111 		'</body></html>';
112 		
113 		/*
114 		 * Transform eRDF -> RDF
115 		 */
116 		var erdf2rdfXslt = ORYX.PATH + "/lib/extract-rdf.xsl";
117 
118 		var rdfResultString;
119 		rdfResult = this.transformString(serializedDOM, erdf2rdfXslt, true);
120 		if (rdfResult instanceof String) {
121 			rdfResultString = rdfResult;
122 			rdfResult = null;
123 		} else {
124 			rdfResultString = xmlSerializer.serializeToString(rdfResult);
125 			if (!rdfResultString.startsWith("<?xml")) {
126 				rdfResultString = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>" + rdfResultString;
127 			}
128 		}
129 		
130 		/*
131 		 * Transform RDF -> EPML
132 		 */
133 		var rdf2epmlXslt = ORYX.PATH + "/xslt/RDF2EPML.xslt";
134 		
135 		var epmlResult = this.transformDOM(rdfResult, rdf2epmlXslt, true);
136 		var epmlResultString;
137 		if (epmlResult instanceof String) {
138 			epmlResultString = epmlResult;
139 			epmlResult = null;
140 		} else {
141 			epmlResultString = xmlSerializer.serializeToString(epmlResult);
142 			if (!epmlResultString.startsWith("<?xml")) {
143 				epmlResultString = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>" + epmlResultString;
144 			}
145 		}
146 		
147 		this.facade.raiseEvent({type:ORYX.CONFIG.EVENT_LOADING_DISABLE});
148 		
149 		// At the moment, only EPML is going to be returned.
150 		this.openDownloadWindow(resource + ".epml", epmlResultString);
151     },

153 	/**
154 	 * Transforms the given string via xslt.
155 	 * 
156 	 * @param {String} string
157 	 * @param {String} xsltPath
158 	 * @param {Boolean} getDOM
159 	 */
160 	transformString: function(string_, xsltPath, getDOM){
161 		var parser = new DOMParser();
162 		var parsedDOM = parser.parseFromString(string_,"text/xml");	
163 		
164 		return this.transformDOM(parsedDOM, xsltPath, getDOM);
165 	},
166 	
167 	/**
168 	 * Transforms the given dom via xslt.
169 	 * 
170 	 * @param {Object} domContent
171 	 * @param {String} xsltPath
172 	 * @param {Boolean} getDOM
173 	 */
174 	transformDOM: function(domContent, xsltPath, getDOM){	
175 		if (domContent == null) {
176 			return new String("Parse Error: \nThe given dom content is null.");
177 		}
178 		var xsl = "";
179 		source=ORYX.PATH + "lib/extract-rdf.xsl";
180 		new Ajax.Request(source, {
181 			asynchronous: false,
182 			method: 'get',
183 			onSuccess: function(transport){
184 				xsl = transport.responseText
185 			}.bind(this),
186 			onFailure: (function(transport){
187 				ORYX.Log.error("XSL load failed" + transport);
188 			}).bind(this)
189 		});
190 		var result;
191 		var resultString;
192 		var xsltProcessor = new XSLTProcessor();
193 		var domParser = new DOMParser();
194 		var xslObject = domParser.parseFromString(xsl, "text/xml");
195 		xsltProcessor.importStylesheet(xslObject);
196 		try {
197 			result = xsltProcessor.transformToFragment(domContent, document);
198 		} catch (error){
199 			return new String("Parse Error: "+error.name + "\n" + error.message);
200 		}
201 		if (getDOM){
202 			return result;
203 		}
204 		resultString = (new XMLSerializer()).serializeToString(result);
205 		return resultString;
206 	},

208 	/**
209 	 * Opens a upload dialog.
210 	 * 
211 	 */
212 	openUploadDialog: function(){
213 		
214 		var form = new Ext.form.FormPanel({
215 			frame : 		true,
216 			bodyStyle:		'padding:5px;',
217 			defaultType : 	'textfield',
218 		  	labelAlign : 	'left',
219 		  	buttonAlign: 	'right',
220 		  	fileUpload : 	true,
221 		  	enctype : 		'multipart/form-data',
222 		  	items : [
223 		  	{
224 		    	text : 		ORYX.I18N.EPCSupport.selectFile, 
225 				style : 	'font-size:12px;margin-bottom:10px;display:block;',
226 				xtype : 	'label'
227 		  	},{
228 		    	fieldLabel : 	ORYX.I18N.EPCSupport.file,
229 		    	inputType : 	'file',
230 				labelStyle :	'width:50px;',
231 				itemCls :		'ext_specific_window_overflow'
232 		  	}]
233 		});
234 
235 
236 		var dialog = new Ext.Window({ 
237 			autoCreate: true, 
238 			title: 		ORYX.I18N.EPCSupport.impPanel, 
239 			height: 	'auto', 
240 			width: 		'auto', 
241 			modal:		true,
242 			collapsible:false,
243 			fixedcenter:true, 
244 			shadow:		true, 
245 			proxyDrag: 	true,
246 			resizable:	false,
247 			items: [form],
248 			buttons:[
249 				{
250 					text:ORYX.I18N.EPCSupport.impBtn,
251 					handler: function(){
252 						
253 							
254 						var loadMask = new Ext.LoadMask(Ext.getBody(), {msg:ORYX.I18N.EPCSupport.progressImp});
255 						loadMask.show();
256 												
257 						form.form.submit({
258 				      		url: ORYX.PATH + '/epc-upload',
259 				      		success: function(f,a){
260 								
261 								dialog.hide();
262 								
263 								// Get the erdf string					
264 								var erdf = a.result;
265 								erdf = erdf.startsWith('<?xml') ? erdf : '<?xml version="1.0" encoding="utf-8"?><div>'+erdf+'</div>';	
266 								// Load the content to the editor
267 								this.loadContent(erdf);
268 								// Hide the waiting panel
269 								loadMask.hide();
270 								
271 				      		}.bind(this),
272 							failure: function(f,a){
273 								dialog.hide();
274 								loadMask.hide();
275 								Ext.MessageBox.show({
276 		           					title: ORYX.I18N.EPCSupport.error,
277 		          	 				msg: a.response.responseText.substring(a.response.responseText.indexOf("content:'")+9, a.response.responseText.indexOf("'}")),
278 		           					buttons: Ext.MessageBox.OK,
279 		           					icon: Ext.MessageBox.ERROR
280 		       					});
281 				      		}
282 				  		});
283 					}.bind(this)
284 				},{
285 					text:ORYX.I18N.EPCSupport.close,
286 					handler:function(){
287 						dialog.hide();
288 					}.bind(this)
289 				}
290 			]
291 		});
292 
293 		dialog.on('hide', function(){
294 			dialog.destroy(true);
295 			delete dialog;
296 		});
297 		dialog.show();
298 	},

300 	/**
301 	 * Creates a hidden form element to communicate parameter values
302 	 * to a php file.
303 	 * 
304 	 * @param {Object} name  The name of the hidden field
305 	 * @param {Object} value The value of the hidden field
306 	 */
307 	createHiddenElement: function(name, value) {
308 		var newElement = document.createElement("input");
309 		newElement.name=name;
310 		newElement.type="hidden";
311 		newElement.value = value;
312 		return newElement
313 	},
314 	
315 	/**
316 	 * Returns the file name to the given result-entry.
317 	 * 
318 	 * @param {String} entry.
319 	 */
320 	getFileName: function(entry) {
321 		var l = entry.length;
322 		if (l > 5){
323 			if (entry.substr(l-5, 5) == "(AML)"){
324 				return entry.substr(0, l-6);
325 			}
326 		}
327 		return entry
328 	},

330 	/**
331 	 * Opens a download window for downloading the given content.
332 	 * 
333 	 * Creates a submit form to communicate the contents to the 
334 	 * download.php file.
335 	 * 
336 	 * @param {Object} content The content to be downloaded. If it is a zip 
337 	 *                         file, then this should be an array of contents.
338 	 * @param {Object} zip     True, if it is a zip file, false otherwise
339 	 */
340 	openDownloadWindow: function(file, content) {
341 		var win = window.open("");
342 		if (win != null) {
343 			win.document.open();
344 			win.document.write("<html><body>");
345 			var submitForm = win.document.createElement("form");
346 			win.document.body.appendChild(submitForm);
347 			
348 			var file = this.getFileName(file);
349 			submitForm.appendChild( this.createHiddenElement("download", content));
350 			submitForm.appendChild( this.createHiddenElement("file", file));
351 			
352 			
353 			submitForm.method = "POST";
354 			win.document.write("</body></html>");
355 			win.document.close();
356 			submitForm.action= ORYX.PATH + "/download";
357 			submitForm.submit();
358 		}		
359 	},
