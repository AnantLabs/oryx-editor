 29 ORYX.Plugins.BpmnLayouter = ORYX.Plugins.AbstractPlugin.extend({
 30 	facade: undefined,
 31 	construct: function(facade){
 32 		this.facade = facade;
 33 		this.facade.offer({
 34 			'name' : "Layout-BPMN",
 35 			'description' : "Layout BPMN Model",
 36 			'functionality' : this.layout.bind(this),
 37 			'group' : "Layout",
 38 			'icon' : ORYX.PATH + "images/auto_layout.png",
 39 			'index' : 1,
 40 			'minShape' : 0,
 41 			'maxShape' : 0
 42 		});
 43 	},
 44 	layout: function(){
 45 		
 46 		this.facade.raiseEvent({
 47             type: ORYX.CONFIG.EVENT_LOADING_ENABLE,
 48 			text: ORYX.I18N.Layouting.doing
 49         });
 50 		
 51 		
 52 		new Ajax.Request(ORYX.CONFIG.BPMN_LAYOUTER, {
 53 			method : 'POST',
 54 			asynchronous : false,
 55 			parameters : {
 56 				data: this.facade.getSerializedJSON(),
 57 				output: "coordinatesonly"
 58 			},
 59 			onFailure: function(request){
 60 				Ext.Msg.alert("Layouting Error", "Error while layouting:!\n" + request.responseText);
 61             	this.facade.raiseEvent({type:ORYX.CONFIG.EVENT_LOADING_DISABLE});
 62 			},
 63 			onSuccess: function(request){
 64 
 65 				/*Ext.Msg.alert("Oryx", "New Layout arrived:!\n" + request.responseText);*/
 66 				
 67 				var setLayoutCommandClass = ORYX.Core.Command.extend({
 68 					construct: function(layoutArray, plugin){
 69 						this.layoutArray = layoutArray;
 70 						this.plugin = plugin;
 71 						this.oldLayoutArray = [];
 72 					},
 73 					execute: function(){
 74 						this.layoutArray.each(function(elem){
 75 							/* get shape */
 76 							var shape = this.plugin.facade.getCanvas().getChildShapeByResourceId(elem.id);
 77 							
 78 							/* save old layout for undo*/
 79 							var oldLayout = {
 80 								id : elem.id,
 81 								bounds : shape.bounds.clone()
 82 							};
 83 							this.oldLayoutArray.push(oldLayout);
 84 							
 85 							/* set new bounds */
 86 							var bound = elem.bounds.split(" ");
 87 							shape.bounds.set(bound[0],bound[1],bound[2],bound[3]);
 88 							
 89 							/* set new dockers */
 90 							if(elem.dockers != null){
 91 								this.plugin.setDockersBad(shape,elem.dockers);
 92 							}
 93 							
 94 							shape.update();
 95 						}.bind(this));
 96 						
 97 						this.plugin.facade.getCanvas().update();
 98 						this.plugin.facade.updateSelection();					
 99 						
100 					},
101 					rollback: function(){
102 						this.oldLayoutArray.each(function(elem){
103 							var shape = this.plugin.facade.getCanvas().getChildShapeByResourceId(elem.id);
104 							shape.bounds.set(elem.bounds);
105 							shape.update();
106 						}.bind(this));
107 						
108 						this.plugin.facade.getCanvas().update();
109 						this.plugin.facade.updateSelection();	
110 					}
111 				});
112 				
113 				
114 				var resp = request.responseText.evalJSON();
115 				if (resp instanceof Array && resp.size() > 0) {
116 					/* create command */
117 					var command = new setLayoutCommandClass(resp, this);
118 					/* execute command */
119 					this.facade.executeCommands([command]);
120 				}
121             	this.facade.raiseEvent({type:ORYX.CONFIG.EVENT_LOADING_DISABLE});
122 			}.bind(this)
123 		})
124 	},
125 	setDockersBad: function(shape, dockers){
126 		var dockersString = "";
127 		dockers.each(function(p){
128 			dockersString += p.x + " " + p.y + " ";
129 		});
130 		dockersString += " # ";
131 		shape.deserialize([{
132 								prefix: 'oryx',
133 								name: 'dockers',
134 								value: dockersString
135 							}]);
136 	},
137 	setDockersGood: function(shape, dockers){
138 		if(elem.dockers.length == 1){
139 			/* docked event */
140 			
141 		}else{
142 			
143 			/* clear all except of the first and last dockers */
144 			var dockers = shape.getDockers().slice(1,-1);
145 			dockers.each(function(docker){
146 				shape.removeDocker(docker);
147 			});
148 			
149 			/* set first and last docker */
150 			var firstDocker = shape.getDockers()[0];
151 			if (firstDocker.getDockedShape()) {
152 				firstDocker.setReferencePoint(elem.dockers[0]);
153 			}
154 			else {
155 				firstDocker.bounds.moveTo(elem.dockers[0].x,elem.dockers[0].y);
156 			}
157 			firstDocker.refresh();
158 			
159 			var lastDocker = shape.getDockers()[1];
160 			if (lastDocker.getDockedShape()) {
161 				lastDocker.setReferencePoint(elem.dockers[elem.dockers.length - 1]);
162 			}
163 			else {
164 				lastDocker.bounds.moveTo(elem.dockers[elem.dockers.length - 1].x, elem.dockers[elem.dockers.length - 1].y);
165 			}
166 			lastDocker.refresh();
167 			
168 			/* add new dockers except of the first and last */
169 			var dockersToAdd = elem.dockers.slice(1,-1);
170 			dockersToAdd.each(function(dockerPoint){
171 				var newDocker = shape.createDocker(undefined, dockerPoint);
172 				newDocker.parent = shape;
173 				newDocker.bounds.centerMoveTo(dockerPoint.x, dockerPoint.y);
174 				/*newDocker.setReferencePoint(dockerPoint);*/
175 				newDocker.update();
176 			});
177 		}		
178 	}
179 });
